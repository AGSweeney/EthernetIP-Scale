35static const char *TAG = "ota_manager";
36static ota_status_info_t s_ota_status = {0};
37static SemaphoreHandle_t s_ota_mutex = NULL;
38static TaskHandle_t s_ota_task_handle = NULL;
39static const esp_partition_t *s_update_partition = NULL; // Store partition being updated
40static size_t s_streaming_total_bytes = 0; // Total bytes written during streaming update
41static size_t s_streaming_expected_size = 0; // Expected total size for streaming update

43static void ota_task(void *pvParameters)
44{
45 const char *url = (const char *)pvParameters;
46
47 esp_http_client_config_t config = {
48 .url = url,
49 .timeout_ms = 5000,
50 };
51
52 esp_https_ota_config_t ota_config = {
53 .http_config = &config,
54 };
55
56 esp_https_ota_handle_t https_ota_handle = NULL;
57 esp_err_t err = esp_https_ota_begin(&ota_config, &https_ota_handle);
58
59 if (err != ESP_OK) {
60 ESP_LOGE(TAG, "OTA begin failed: %s", esp_err_to_name(err));
61 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
62 s_ota_status.status = OTA_STATUS_ERROR;
63 snprintf(s_ota_status.message, sizeof(s_ota_status.message), "OTA begin failed: %s", esp_err_to_name(err));
64 s_ota_status.progress = 0;
65 s_ota_task_handle = NULL;
66 xSemaphoreGive(s_ota_mutex);
67 free((void *)url); // Free the URL copy
68 vTaskDelete(NULL);
69 return;
70 }
71
72 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
73 s_ota_status.status = OTA_STATUS_IN_PROGRESS;
74 s_ota_status.progress = 0;
75 strcpy(s_ota_status.message, "Downloading firmware...");
76 xSemaphoreGive(s_ota_mutex);
77
78 esp_err_t ota_finish_err = ESP_OK;
79 while (1) {
80 err = esp_https_ota_perform(https_ota_handle);
81 if (err != ESP_ERR_HTTPS_OTA_IN_PROGRESS) {
82 break;
83 }
84
85 // Update progress (simplified - just indicate we're downloading)
86 esp_app_desc_t new_app_info;
87 if (esp_https_ota_get_img_desc(https_ota_handle, &new_app_info) == ESP_OK) {
88 // Image header received, update progress indicator
89 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
90 if (s_ota_status.progress < 90) {
91 s_ota_status.progress += 1; // Increment progress gradually
92 }
93 snprintf(s_ota_status.message, sizeof(s_ota_status.message),
94 "Downloading firmware...");
95 xSemaphoreGive(s_ota_mutex);
96 }
97
98 vTaskDelay(pdMS_TO_TICKS(100));
99 }
100
101 if (err == ESP_OK) {
102 ota_finish_err = esp_https_ota_finish(https_ota_handle);
103 if (ota_finish_err == ESP_OK) {
104 ESP_LOGI(TAG, "OTA update successful, rebooting...");
105 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
106 s_ota_status.status = OTA_STATUS_COMPLETE;
107 s_ota_status.progress = 100;
108 strcpy(s_ota_status.message, "Update complete, rebooting...");
109 xSemaphoreGive(s_ota_mutex);
110
111 vTaskDelay(pdMS_TO_TICKS(2000));
112 esp_restart();
113 } else {
114 ESP_LOGE(TAG, "OTA finish failed: %s", esp_err_to_name(ota_finish_err));
115 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
116 s_ota_status.status = OTA_STATUS_ERROR;
117 snprintf(s_ota_status.message, sizeof(s_ota_status.message),
118 "OTA finish failed: %s", esp_err_to_name(ota_finish_err));
119 xSemaphoreGive(s_ota_mutex);
120 }
121 } else {
122 ESP_LOGE(TAG, "OTA perform failed: %s", esp_err_to_name(err));
123 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
124 s_ota_status.status = OTA_STATUS_ERROR;
125 snprintf(s_ota_status.message, sizeof(s_ota_status.message),
126 "OTA perform failed: %s", esp_err_to_name(err));
127 xSemaphoreGive(s_ota_mutex);
128 }
129
130 esp_https_ota_abort(https_ota_handle);
131 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
132 s_ota_task_handle = NULL;
133 xSemaphoreGive(s_ota_mutex);
134 free((void *)url); // Free the URL copy
135 vTaskDelete(NULL);
136}

138bool ota_manager_init(void)
139{
140 if (s_ota_mutex == NULL) {
141 s_ota_mutex = xSemaphoreCreateMutex();
142 if (s_ota_mutex == NULL) {
143 ESP_LOGE(TAG, "Failed to create OTA mutex");
144 return false;
145 }
146 }
147
148 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
149 s_ota_status.status = OTA_STATUS_IDLE;
150 s_ota_status.progress = 0;
151 strcpy(s_ota_status.message, "Ready");
152 xSemaphoreGive(s_ota_mutex);
153
154 ESP_LOGI(TAG, "OTA manager initialized");
155 return true;
156}

158bool ota_manager_start_update(const char *url)
159{
160 if (url == NULL || strlen(url) == 0) {
161 ESP_LOGE(TAG, "Invalid URL");
162 return false;
163 }
164
165 if (s_ota_mutex == NULL) {
166 ESP_LOGE(TAG, "OTA manager not initialized");
167 return false;
168 }
169
170 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
171 if (s_ota_task_handle != NULL) {
172 xSemaphoreGive(s_ota_mutex);
173 ESP_LOGW(TAG, "OTA update already in progress");
174 return false;
175 }
176
177 // Create task for OTA update
178 char *url_copy = strdup(url);
179 if (url_copy == NULL) {
180 xSemaphoreGive(s_ota_mutex);
181 ESP_LOGE(TAG, "Failed to allocate memory for URL");
182 return false;
183 }
184
185 BaseType_t result = xTaskCreate(ota_task, "ota_task", 8192, url_copy, 5, &s_ota_task_handle);
186 if (result != pdPASS) {
187 xSemaphoreGive(s_ota_mutex);
188 ESP_LOGE(TAG, "Failed to create OTA task");
189 free(url_copy);
190 return false;
191 }
192
193 xSemaphoreGive(s_ota_mutex);
194 ESP_LOGI(TAG, "OTA update started from URL: %s", url);
195 return true;
196}

198static void ota_update_from_data_task(void *pvParameters)
199{
200 typedef struct {
201 uint8_t *data;
202 size_t len;
203 } ota_data_t;
204
205 ota_data_t *ota_data = (ota_data_t *)pvParameters;
206
207 const esp_partition_t *update_partition = esp_ota_get_next_update_partition(NULL);
208 if (update_partition == NULL) {
209 ESP_LOGE(TAG, "No OTA partition found");
210 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
211 s_ota_status.status = OTA_STATUS_ERROR;
212 strcpy(s_ota_status.message, "No OTA partition found");
213 s_ota_status.progress = 0;
214 s_ota_task_handle = NULL;
215 xSemaphoreGive(s_ota_mutex);
216 free(ota_data->data);
217 free(ota_data);
218 vTaskDelete(NULL);
219 return;
220 }
221
222 esp_ota_handle_t ota_handle = 0;
223 esp_err_t err = esp_ota_begin(update_partition, ota_data->len, &ota_handle);
224 if (err != ESP_OK) {
225 ESP_LOGE(TAG, "esp_ota_begin failed: %s", esp_err_to_name(err));
226 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
227 s_ota_status.status = OTA_STATUS_ERROR;
228 snprintf(s_ota_status.message, sizeof(s_ota_status.message), "OTA begin failed: %s", esp_err_to_name(err));
229 s_ota_status.progress = 0;
230 s_ota_task_handle = NULL;
231 xSemaphoreGive(s_ota_mutex);
232 free(ota_data->data);
233 free(ota_data);
234 vTaskDelete(NULL);
235 return;
236 }
237
238 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
239 s_ota_status.status = OTA_STATUS_IN_PROGRESS;
240 s_ota_status.progress = 0;
241 strcpy(s_ota_status.message, "Writing firmware...");
242 xSemaphoreGive(s_ota_mutex);
243
244 // Write firmware in chunks
245 const size_t chunk_size = 4096;
246 size_t written = 0;
247 for (size_t offset = 0; offset < ota_data->len; offset += chunk_size) {
248 size_t to_write = (offset + chunk_size > ota_data->len) ? (ota_data->len - offset) : chunk_size;
249 err = esp_ota_write(ota_handle, ota_data->data + offset, to_write);
250 if (err != ESP_OK) {
251 ESP_LOGE(TAG, "esp_ota_write failed at offset %d: %s", offset, esp_err_to_name(err));
252 esp_ota_abort(ota_handle);
253 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
254 s_ota_status.status = OTA_STATUS_ERROR;
255 snprintf(s_ota_status.message, sizeof(s_ota_status.message), "Write failed: %s", esp_err_to_name(err));
256 s_ota_task_handle = NULL;
257 xSemaphoreGive(s_ota_mutex);
258 free(ota_data->data);
259 free(ota_data);
260 vTaskDelete(NULL);
261 return;
262 }
263
264 written += to_write;
265 uint8_t progress = (uint8_t)((written * 100) / ota_data->len);
266
267 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
268 s_ota_status.progress = progress;
269 snprintf(s_ota_status.message, sizeof(s_ota_status.message), "Writing: %d%%", progress);
270 xSemaphoreGive(s_ota_mutex);
271
272 vTaskDelay(pdMS_TO_TICKS(10)); // Small delay to allow status updates
273 }
274
275 err = esp_ota_end(ota_handle);
276 if (err != ESP_OK) {
277 ESP_LOGE(TAG, "esp_ota_end failed: %s", esp_err_to_name(err));
278 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
279 s_ota_status.status = OTA_STATUS_ERROR;
280 snprintf(s_ota_status.message, sizeof(s_ota_status.message), "OTA end failed: %s", esp_err_to_name(err));
281 s_ota_task_handle = NULL;
282 xSemaphoreGive(s_ota_mutex);
283 free(ota_data->data);
284 free(ota_data);
285 vTaskDelete(NULL);
286 return;
287 }
288
289 err = esp_ota_set_boot_partition(update_partition);
290 if (err != ESP_OK) {
291 ESP_LOGE(TAG, "esp_ota_set_boot_partition failed: %s", esp_err_to_name(err));
292 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
293 s_ota_status.status = OTA_STATUS_ERROR;
294 snprintf(s_ota_status.message, sizeof(s_ota_status.message), "Set boot partition failed: %s", esp_err_to_name(err));
295 s_ota_task_handle = NULL;
296 xSemaphoreGive(s_ota_mutex);
297 free(ota_data->data);
298 free(ota_data);
299 vTaskDelete(NULL);
300 return;
301 }
302
303 ESP_LOGI(TAG, "OTA update successful, rebooting...");
304 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
305 s_ota_status.status = OTA_STATUS_COMPLETE;
306 s_ota_status.progress = 100;
307 strcpy(s_ota_status.message, "Update complete, rebooting...");
308 xSemaphoreGive(s_ota_mutex);
309
310 free(ota_data->data);
311 free(ota_data);
312
313 vTaskDelay(pdMS_TO_TICKS(2000));
314 esp_restart();
315}

317bool ota_manager_start_update_from_data(const uint8_t *data, size_t data_len)
318{
319 if (data == NULL || data_len == 0) {
320 ESP_LOGE(TAG, "Invalid firmware data");
321 return false;
322 }
323
324 if (s_ota_mutex == NULL) {
325 ESP_LOGE(TAG, "OTA manager not initialized");
326 return false;
327 }
328
329 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
330 if (s_ota_task_handle != NULL) {
331 xSemaphoreGive(s_ota_mutex);
332 ESP_LOGW(TAG, "OTA update already in progress");
333 return false;
334 }
335
336 // Get next OTA partition to check size
337 const esp_partition_t *update_partition = esp_ota_get_next_update_partition(NULL);
338 if (update_partition == NULL) {
339 ESP_LOGE(TAG, "No OTA partition found");
340 return false;
341 }
342
343 // Check if firmware fits (with some margin for headers)
344 size_t partition_size = update_partition->size;
345 if (data_len > partition_size) {
346 ESP_LOGE(TAG, "Firmware too large: %d bytes (max: %d)", data_len, partition_size);
347 return false;
348 }
349
350 // Create task for OTA update
351 typedef struct {
352 uint8_t *data;
353 size_t len;
354 } ota_data_t;
355
356 ota_data_t *ota_data = malloc(sizeof(ota_data_t));
357 if (ota_data == NULL) {
358 ESP_LOGE(TAG, "Failed to allocate memory for OTA data");
359 return false;
360 }
361
362 ota_data->data = malloc(data_len);
363 if (ota_data->data == NULL) {
364 ESP_LOGE(TAG, "Failed to allocate memory for firmware data");
365 free(ota_data);
366 return false;
367 }
368
369 memcpy(ota_data->data, data, data_len);
370 ota_data->len = data_len;
371
372 BaseType_t result = xTaskCreate(ota_update_from_data_task, "ota_task", 8192, ota_data, 5, &s_ota_task_handle);
373 if (result != pdPASS) {
374 xSemaphoreGive(s_ota_mutex);
375 ESP_LOGE(TAG, "Failed to create OTA task");
376 free(ota_data->data);
377 free(ota_data);
378 return false;
379 }
380
381 xSemaphoreGive(s_ota_mutex);
382 ESP_LOGI(TAG, "OTA update started from uploaded data (%d bytes)", data_len);
383 return true;
384}

386bool ota_manager_get_status(ota_status_info_t *status_info)
387{
388 if (status_info == NULL) {
389 return false;
390 }
391
392 if (s_ota_mutex == NULL) {
393 return false;
394 }
395
396 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
397 memcpy(status_info, &s_ota_status, sizeof(ota_status_info_t));
398 xSemaphoreGive(s_ota_mutex);
399
400 return true;
401}

403// Streaming OTA update functions (no double buffering)
404esp_ota_handle_t ota_manager_start_streaming_update(size_t expected_size)
405{
406 ESP_LOGI(TAG, "Starting streaming OTA update, expected_size: %d", expected_size);
407
408 if (s_ota_mutex == NULL) {
409 ESP_LOGE(TAG, "OTA manager not initialized - call ota_manager_init() first");
410 return 0;
411 }
412
413 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
414 if (s_ota_task_handle != NULL) {
415 xSemaphoreGive(s_ota_mutex);
416 ESP_LOGW(TAG, "OTA update already in progress (task handle: %p)", s_ota_task_handle);
417 return 0;
418 }
419
420 // Get running partition for diagnostics
421 const esp_partition_t *running_partition = esp_ota_get_running_partition();
422 if (running_partition != NULL) {
423 ESP_LOGI(TAG, "Running partition: label=%s, type=%d, subtype=%d, address=0x%x, size=%d",
424 running_partition->label, running_partition->type, running_partition->subtype,
425 running_partition->address, running_partition->size);
426 } else {
427 ESP_LOGW(TAG, "Could not determine running partition");
428 }
429
430 // Get next OTA partition
431 const esp_partition_t *update_partition = esp_ota_get_next_update_partition(NULL);
432
433 // If esp_ota_get_next_update_partition fails, try manual selection
434 if (update_partition == NULL) {
435 ESP_LOGW(TAG, "esp_ota_get_next_update_partition returned NULL, trying manual selection");
436
437 // Try to find OTA partitions manually
438 const esp_partition_t *ota0 = esp_partition_find_first(ESP_PARTITION_TYPE_APP, ESP_PARTITION_SUBTYPE_APP_OTA_0, NULL);
439 const esp_partition_t *ota1 = esp_partition_find_first(ESP_PARTITION_TYPE_APP, ESP_PARTITION_SUBTYPE_APP_OTA_1, NULL);
440 const esp_partition_t *otadata = esp_partition_find_first(ESP_PARTITION_TYPE_DATA, ESP_PARTITION_SUBTYPE_DATA_OTA, NULL);
441
442 if (ota0 != NULL) {
443 ESP_LOGI(TAG, "Found ota_0 partition: label=%s, address=0x%x, size=%d", ota0->label, ota0->address, ota0->size);
444 } else {
445 ESP_LOGW(TAG, "ota_0 partition not found");
446 }
447 if (ota1 != NULL) {
448 ESP_LOGI(TAG, "Found ota_1 partition: label=%s, address=0x%x, size=%d", ota1->label, ota1->address, ota1->size);
449 } else {
450 ESP_LOGW(TAG, "ota_1 partition not found");
451 }
452 if (otadata != NULL) {
453 ESP_LOGI(TAG, "Found otadata partition: label=%s, address=0x%x, size=%d", otadata->label, otadata->address, otadata->size);
454 } else {
455 ESP_LOGW(TAG, "otadata partition not found");
456 }
457
458 // Determine which partition to update based on running partition
459 if (running_partition != NULL) {
460 if (running_partition->subtype == ESP_PARTITION_SUBTYPE_APP_OTA_0 && ota1 != NULL) {
461 update_partition = ota1;
462 ESP_LOGI(TAG, "Running on ota_0, will update ota_1");
463 } else if (running_partition->subtype == ESP_PARTITION_SUBTYPE_APP_OTA_1 && ota0 != NULL) {
464 update_partition = ota0;
465 ESP_LOGI(TAG, "Running on ota_1, will update ota_0");
466 } else if (ota0 != NULL) {
467 // Default to ota_0 if we can't determine
468 update_partition = ota0;
469 ESP_LOGI(TAG, "Defaulting to ota_0 partition");
470 } else if (ota1 != NULL) {
471 update_partition = ota1;
472 ESP_LOGI(TAG, "Defaulting to ota_1 partition");
473 }
474 } else {
475 // No running partition info, try ota_0 first
476 if (ota0 != NULL) {
477 update_partition = ota0;
478 ESP_LOGI(TAG, "No running partition info, using ota_0");
479 } else if (ota1 != NULL) {
480 update_partition = ota1;
481 ESP_LOGI(TAG, "No running partition info, using ota_1");
482 }
483 }
484
485 if (update_partition == NULL) {
486 ESP_LOGE(TAG, "No OTA partition found - check partition table");
487 xSemaphoreGive(s_ota_mutex);
488 return 0;
489 }
490 }
491
492 ESP_LOGI(TAG, "Found OTA update partition: label=%s, type=%d, subtype=%d, address=0x%x, size=%d",
493 update_partition->label, update_partition->type, update_partition->subtype,
494 update_partition->address, update_partition->size);
495
496 // Use partition size as max (esp_ota_begin will validate)
497 size_t partition_size = update_partition->size;
498 size_t ota_size = (expected_size > 0 && expected_size < partition_size) ? expected_size : partition_size;
499
500 ESP_LOGI(TAG, "Starting OTA with size: %d (partition size: %d)", ota_size, partition_size);
501
502 esp_ota_handle_t ota_handle = 0;
503 esp_err_t err = esp_ota_begin(update_partition, ota_size, &ota_handle);
504 if (err != ESP_OK) {
505 ESP_LOGE(TAG, "esp_ota_begin failed: %s (0x%x)", esp_err_to_name(err), err);
506 xSemaphoreGive(s_ota_mutex);
507 return 0;
508 }
509
510 ESP_LOGI(TAG, "esp_ota_begin successful, handle: %d", ota_handle);
511
512 // Store the partition pointer for use in finish
513 s_update_partition = update_partition;
514
515 // Initialize streaming progress tracking
516 s_streaming_total_bytes = 0;
517 s_streaming_expected_size = ota_size; // Use partition size as expected size if not provided
518
519 // Update status
520 s_ota_status.status = OTA_STATUS_IN_PROGRESS;
521 s_ota_status.progress = 0;
522 strcpy(s_ota_status.message, "Uploading firmware...");
523 xSemaphoreGive(s_ota_mutex);
524
525 ESP_LOGI(TAG, "Streaming OTA update started successfully");
526 return ota_handle;
527}

529bool ota_manager_write_streaming_chunk(esp_ota_handle_t ota_handle, const uint8_t *data, size_t len)
530{
531 if (ota_handle == 0 || data == NULL || len == 0) {
532 ESP_LOGE(TAG, "Invalid parameters for write_streaming_chunk");
533 return false;
534 }
535
536 esp_ota_handle_t handle = ota_handle;
537 esp_err_t err = esp_ota_write(handle, data, len);
538
539 if (err != ESP_OK) {
540 ESP_LOGE(TAG, "esp_ota_write failed: %s", esp_err_to_name(err));
541 esp_ota_abort(handle);
542 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
543 s_ota_status.status = OTA_STATUS_ERROR;
544 snprintf(s_ota_status.message, sizeof(s_ota_status.message), "Write failed: %s", esp_err_to_name(err));
545 s_streaming_total_bytes = 0; // Reset streaming counters on error
546 s_streaming_expected_size = 0;
547 xSemaphoreGive(s_ota_mutex);
548 return false;
549 }
550
551 // Update progress tracking
552 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
553 s_streaming_total_bytes += len;
554
555 // Calculate progress percentage (0-100)
556 if (s_streaming_expected_size > 0) {
557 uint8_t progress = (uint8_t)((s_streaming_total_bytes * 100) / s_streaming_expected_size);
558 if (progress > 100) {
559 progress = 100; // Cap at 100%
560 }
561 s_ota_status.progress = progress;
562 snprintf(s_ota_status.message, sizeof(s_ota_status.message),
563 "Uploading firmware... %d%% (%d/%d bytes)",
564 progress, s_streaming_total_bytes, s_streaming_expected_size);
565 } else {
566 // If we don't know expected size, just show bytes written
567 snprintf(s_ota_status.message, sizeof(s_ota_status.message),
568 "Uploading firmware... %d bytes written", s_streaming_total_bytes);
569 }
570 xSemaphoreGive(s_ota_mutex);
571
572 return true;
573}

575bool ota_manager_finish_streaming_update(esp_ota_handle_t ota_handle)
576{
577 if (ota_handle == 0) {
578 ESP_LOGE(TAG, "Invalid OTA handle");
579 return false;
580 }
581
582 esp_ota_handle_t handle = ota_handle;
583
584 esp_err_t err = esp_ota_end(handle);
585 if (err != ESP_OK) {
586 ESP_LOGE(TAG, "esp_ota_end failed: %s", esp_err_to_name(err));
587 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
588 s_ota_status.status = OTA_STATUS_ERROR;
589 snprintf(s_ota_status.message, sizeof(s_ota_status.message), "OTA end failed: %s", esp_err_to_name(err));
590 s_ota_task_handle = NULL;
591 s_update_partition = NULL;
592 s_streaming_total_bytes = 0; // Reset streaming counters on error
593 s_streaming_expected_size = 0;
594 xSemaphoreGive(s_ota_mutex);
595 return false;
596 }
597
598 // Use the partition we stored when starting the update
599 if (s_update_partition == NULL) {
600 ESP_LOGE(TAG, "Update partition not found - internal error");
601 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
602 s_ota_status.status = OTA_STATUS_ERROR;
603 strcpy(s_ota_status.message, "Update partition not found");
604 s_ota_task_handle = NULL;
605 s_streaming_total_bytes = 0; // Reset streaming counters on error
606 s_streaming_expected_size = 0;
607 xSemaphoreGive(s_ota_mutex);
608 return false;
609 }
610
611 err = esp_ota_set_boot_partition(s_update_partition);
612 if (err != ESP_OK) {
613 ESP_LOGE(TAG, "esp_ota_set_boot_partition failed: %s", esp_err_to_name(err));
614 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
615 s_ota_status.status = OTA_STATUS_ERROR;
616 snprintf(s_ota_status.message, sizeof(s_ota_status.message), "Set boot partition failed: %s", esp_err_to_name(err));
617 s_ota_task_handle = NULL;
618 s_update_partition = NULL;
619 s_streaming_total_bytes = 0; // Reset streaming counters on error
620 s_streaming_expected_size = 0;
621 xSemaphoreGive(s_ota_mutex);
622 return false;
623 }
624
625 ESP_LOGI(TAG, "Streaming OTA update successful, rebooting...");
626 xSemaphoreTake(s_ota_mutex, portMAX_DELAY);
627 s_ota_status.status = OTA_STATUS_COMPLETE;
628 s_ota_status.progress = 100;
629 strcpy(s_ota_status.message, "Update complete, rebooting...");
630 s_ota_task_handle = NULL;
631 s_update_partition = NULL; // Clear partition pointer
632 s_streaming_total_bytes = 0; // Reset streaming counters
633 s_streaming_expected_size = 0;
634 xSemaphoreGive(s_ota_mutex);
635
636 // Delay 3 seconds to allow web UI to poll and display completion status
637 vTaskDelay(pdMS_TO_TICKS(3000));
638 esp_restart();
639
640 return true;
641}
