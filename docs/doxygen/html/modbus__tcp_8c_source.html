44static void modbus_tcp_server_task(void *pvParameters)
45{
46 struct sockaddr_in client_addr;
47 socklen_t client_addr_len = sizeof(client_addr);
48 int client_sockets[MODBUS_TCP_MAX_CONNECTIONS];
49 int max_fd;
50 fd_set read_fds;
51 bool running = true;
52
53 memset(client_sockets, -1, sizeof(client_sockets));
54
55 while (running) {
56 // Check running flag with mutex protection
57 if (s_modbus_mutex != NULL) {
58 xSemaphoreTake(s_modbus_mutex, portMAX_DELAY);
59 running = s_running;
60 xSemaphoreGive(s_modbus_mutex);
61 }
62
63 if (!running) {
64 break;
65 }
66 FD_ZERO(&read_fds);
67 FD_SET(s_listen_socket, &read_fds);
68 max_fd = s_listen_socket;
69
70 // Add client sockets to set
71 for (int i = 0; i < MODBUS_TCP_MAX_CONNECTIONS; i++) {
72 if (client_sockets[i] >= 0) {
73 FD_SET(client_sockets[i], &read_fds);
74 if (client_sockets[i] > max_fd) {
75 max_fd = client_sockets[i];
76 }
77 }
78 }
79
80 struct timeval timeout = {.tv_sec = 1, .tv_usec = 0};
81 int activity = select(max_fd + 1, &read_fds, NULL, NULL, &timeout);
82
83 if (activity < 0 && errno != EINTR) {
84 ESP_LOGE(TAG, "Select error: %s", strerror(errno));
85 break;
86 }
87
88 // Check for new connections
89 if (FD_ISSET(s_listen_socket, &read_fds)) {
90 int new_socket = accept(s_listen_socket, (struct sockaddr *)&client_addr, &client_addr_len);
91 if (new_socket >= 0) {
92 // New ModbusTCP connection
93
94 // Disable Nagle's algorithm for low latency
95 int flag = 1;
96 if (setsockopt(new_socket, IPPROTO_TCP, TCP_NODELAY, &flag, sizeof(flag)) < 0) {
97 ESP_LOGW(TAG, "Failed to set TCP_NODELAY on new connection: %s", strerror(errno));
98 }
99
100 // Find empty slot
101 bool added = false;
102 for (int i = 0; i < MODBUS_TCP_MAX_CONNECTIONS; i++) {
103 if (client_sockets[i] < 0) {
104 client_sockets[i] = new_socket;
105 added = true;
106 break;
107 }
108 }
109
110 if (!added) {
111 ESP_LOGE(TAG, "Max connections reached, closing new connection");
112 close(new_socket);
113 }
114 }
115 }
116
117 // Check client sockets for data
118 for (int i = 0; i < MODBUS_TCP_MAX_CONNECTIONS; i++) {
119 if (client_sockets[i] >= 0 && FD_ISSET(client_sockets[i], &read_fds)) {
120 if (!modbus_tcp_handle_request(client_sockets[i])) {
121 // Connection closed or error
122 // Closing ModbusTCP connection
123 close(client_sockets[i]);
124 client_sockets[i] = -1;
125 }
126 }
127 }
128 }
129
130 // Cleanup
131 for (int i = 0; i < MODBUS_TCP_MAX_CONNECTIONS; i++) {
132 if (client_sockets[i] >= 0) {
133 close(client_sockets[i]);
134 }
135 }
136
137 if (s_modbus_mutex != NULL) {
138 xSemaphoreTake(s_modbus_mutex, portMAX_DELAY);
139 s_server_task_handle = NULL;
140 xSemaphoreGive(s_modbus_mutex);
141 } else {
142 s_server_task_handle = NULL;
143 }
144 vTaskDelete(NULL);
145}

147bool modbus_tcp_init(void)
148{
149 if (s_modbus_mutex == NULL) {
150 s_modbus_mutex = xSemaphoreCreateMutex();
151 if (s_modbus_mutex == NULL) {
152 ESP_LOGE(TAG, "Failed to create ModbusTCP mutex");
153 return false;
154 }
155 }
156
157 xSemaphoreTake(s_modbus_mutex, portMAX_DELAY);
158 if (s_listen_socket >= 0) {
159 xSemaphoreGive(s_modbus_mutex);
160 ESP_LOGE(TAG, "ModbusTCP already initialized");
161 return true;
162 }
163
164 s_listen_socket = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
165 if (s_listen_socket < 0) {
166 xSemaphoreGive(s_modbus_mutex);
167 ESP_LOGE(TAG, "Failed to create socket: %s", strerror(errno));
168 return false;
169 }
170
171 int opt = 1;
172 if (setsockopt(s_listen_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt)) < 0) {
173 xSemaphoreGive(s_modbus_mutex);
174 ESP_LOGE(TAG, "Failed to set socket options: %s", strerror(errno));
175 close(s_listen_socket);
176 s_listen_socket = -1;
177 return false;
178 }
179
180 struct sockaddr_in server_addr;
181 memset(&server_addr, 0, sizeof(server_addr));
182 server_addr.sin_family = AF_INET;
183 server_addr.sin_addr.s_addr = INADDR_ANY;
184 server_addr.sin_port = htons(MODBUS_TCP_PORT);
185
186 if (bind(s_listen_socket, (struct sockaddr *)&server_addr, sizeof(server_addr)) < 0) {
187 xSemaphoreGive(s_modbus_mutex);
188 ESP_LOGE(TAG, "Failed to bind socket: %s", strerror(errno));
189 close(s_listen_socket);
190 s_listen_socket = -1;
191 return false;
192 }
193
194 if (listen(s_listen_socket, MODBUS_TCP_MAX_CONNECTIONS) < 0) {
195 xSemaphoreGive(s_modbus_mutex);
196 ESP_LOGE(TAG, "Failed to listen: %s", strerror(errno));
197 close(s_listen_socket);
198 s_listen_socket = -1;
199 return false;
200 }
201
202 xSemaphoreGive(s_modbus_mutex);
203 // ModbusTCP server initialized
204 return true;
205}

207bool modbus_tcp_start(void)
208{
209 if (s_modbus_mutex == NULL) {
210 ESP_LOGE(TAG, "ModbusTCP not initialized");
211 return false;
212 }
213
214 xSemaphoreTake(s_modbus_mutex, portMAX_DELAY);
215 if (s_listen_socket < 0) {
216 xSemaphoreGive(s_modbus_mutex);
217 ESP_LOGE(TAG, "ModbusTCP not initialized");
218 return false;
219 }
220
221 if (s_server_task_handle != NULL) {
222 xSemaphoreGive(s_modbus_mutex);
223 ESP_LOGE(TAG, "ModbusTCP server already running");
224 return true;
225 }
226
227 s_running = true;
228 BaseType_t result = xTaskCreate(modbus_tcp_server_task, "modbus_tcp", 8192, NULL, 5, &s_server_task_handle);
229 if (result != pdPASS) {
230 s_running = false;
231 xSemaphoreGive(s_modbus_mutex);
232 ESP_LOGE(TAG, "Failed to create ModbusTCP server task");
233 return false;
234 }
235
236 xSemaphoreGive(s_modbus_mutex);
237 // ModbusTCP server started
238 return true;
239}

241void modbus_tcp_stop(void)
242{
243 if (s_modbus_mutex == NULL) {
244 return;
245 }
246
247 xSemaphoreTake(s_modbus_mutex, portMAX_DELAY);
248 s_running = false;
249 TaskHandle_t task_handle = s_server_task_handle;
250
251 // Close socket with mutex protection
252 if (s_listen_socket >= 0) {
253 close(s_listen_socket);
254 s_listen_socket = -1;
255 }
256
257 xSemaphoreGive(s_modbus_mutex);
258
259 if (task_handle != NULL) {
260 // Wait for task to finish (with timeout)
261 vTaskDelay(pdMS_TO_TICKS(1000));
262 xSemaphoreTake(s_modbus_mutex, portMAX_DELAY);
263 s_server_task_handle = NULL;
264 xSemaphoreGive(s_modbus_mutex);
265 }
266
267 // ModbusTCP server stopped
268}
