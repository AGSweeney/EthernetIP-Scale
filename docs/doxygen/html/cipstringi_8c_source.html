159void CipStringIDecodeFromMessage(CipStringI *data_to,
160 CipMessageRouterRequest *const message_router_request)
161{
162
163 CipStringI *target_stringI = data_to;
164
165 target_stringI->number_of_strings = GetUsintFromMessage(
166 &message_router_request->data);
167
168 target_stringI->array_of_string_i_structs = CipCalloc(
169 target_stringI->number_of_strings, sizeof(CipStringIStruct) );
170
171 for (size_t i = 0; i < target_stringI->number_of_strings; ++i) {
172
173 target_stringI->array_of_string_i_structs[i].language_char_1 =
174 GetUsintFromMessage(&message_router_request->data);
175 target_stringI->array_of_string_i_structs[i].language_char_2 =
176 GetUsintFromMessage(&message_router_request->data);
177 target_stringI->array_of_string_i_structs[i].language_char_3 =
178 GetUsintFromMessage(&message_router_request->data);
179 target_stringI->array_of_string_i_structs[i].char_string_struct =
180 GetUsintFromMessage(&message_router_request->data);
181 target_stringI->array_of_string_i_structs[i].character_set =
182 GetUintFromMessage(&message_router_request->data);
183
184 switch (target_stringI->array_of_string_i_structs[i].char_string_struct) {
185 case kCipShortString: {
186 target_stringI->array_of_string_i_structs[i].string = CipCalloc(1,
187 sizeof(
188 CipShortString) );
189 CipShortString *short_string =
190 (CipShortString *) (target_stringI->array_of_string_i_structs[i].
191 string);
192 CipUsint length = GetUsintFromMessage(
193 &message_router_request->data);
194 SetCipShortStringByData(short_string, length,
195 message_router_request->data);
196 message_router_request->data += length;
197
198 }
199 break;
200 case kCipString: {
201 target_stringI->array_of_string_i_structs[i].string = CipCalloc(1,
202 sizeof(
203 CipString) );
204 CipString *const string =
205 (CipString *const ) target_stringI->array_of_string_i_structs[i].
206 string;
207 CipUint length = GetUintFromMessage(&message_router_request->data);
208 SetCipStringByData(string, length, message_router_request->data);
209 message_router_request->data += length;
210 }
211 break;
212 case kCipString2: {
213 target_stringI->array_of_string_i_structs[i].string = CipCalloc(1,
214 sizeof(
215 CipString2) );
216 CipString2 *const string =
217 (CipString2 *const ) target_stringI->array_of_string_i_structs[i].
218 string;
219 CipUint length = GetUintFromMessage(&message_router_request->data);
220 SetCipString2ByData(string, length, message_router_request->data);
221 message_router_request->data += length * 2 * sizeof(CipOctet);
222 }
223 break;
224 case kCipStringN: {
225 CipUint size = GetUintFromMessage(&message_router_request->data);
226 CipUint length = GetUintFromMessage(&message_router_request->data);
227
228 target_stringI->array_of_string_i_structs[i].string = CipCalloc(1,
229 sizeof(
230 CipStringN) );
231 CipStringN *const string =
232 (CipStringN *const ) target_stringI->array_of_string_i_structs[i].
233 string;
234 SetCipStringNByData(string, length, size,
235 message_router_request->data);
236 message_router_request->data += length * size;
237 }
238 break;
239 default:
240 OPENER_TRACE_ERR("CIP File: No valid String type received!\n");
241 }

245bool CipStringICompare(const CipStringI *const stringI_1,
246 const CipStringI *const stringI_2) {
247
248 /*loop through struct 1 strings*/
249 for (size_t i = 0; i < stringI_1->number_of_strings; ++i) {
250 // String 1
251 void *string_1 = stringI_1->array_of_string_i_structs[i].string;
252 void *string_1_data = NULL;
253 CipUint len_1 = 0; //size of string-struct in bytes
254
255 switch (stringI_1->array_of_string_i_structs[i].char_string_struct) {
256 case kCipShortString: {
257 len_1 = ((CipShortString *)string_1)->length;
258 string_1_data = ((CipShortString *)string_1)->string;
259 }
260 break;
261 case kCipString: {
262 len_1 = ((CipString *)string_1)->length;
263 string_1_data = ((CipString *)string_1)->string;
264 }
265
266 break;
267 case kCipString2: {
268 len_1 = ((CipString2 *)string_1)->length * 2;
269 string_1_data = ((CipString2 *)string_1)->string;
270 }
271 break;
272 case kCipStringN: {
273 CipUint length = ((CipStringN *)string_1)->length;
274 CipUint size = ((CipStringN *)string_1)->size; //bytes per symbol
275 len_1 = length * size;
276 string_1_data = ((CipStringN *)string_1)->string;
277
278 }
279 break;
280 default:
281 OPENER_TRACE_ERR("CIP File: No valid String type received!\n");
282 }
283
284 /*loop through struct 2 strings*/
285 for (size_t j = 0; j < stringI_2->number_of_strings; ++j) {
286 // String 2
287 void *string_2 = stringI_2->array_of_string_i_structs[j].string;
288 void *string_2_data = NULL;
289 CipUint len_2 = 0; //size of string-struct in bytes
290
291 switch (stringI_2->array_of_string_i_structs[j].char_string_struct) {
292 case kCipShortString: {
293 len_2 = ((CipShortString *)string_2)->length;
294 string_2_data = ((CipShortString *)string_2)->string;
295 }
296 break;
297 case kCipString: {
298 len_2 = ((CipString *)string_2)->length;
299 string_2_data = ((CipString *)string_2)->string;
300 }
301
302 break;
303 case kCipString2: {
304 len_2 = ((CipString2 *)string_2)->length * 2;
305 string_2_data = ((CipString2 *)string_2)->string;
306 }
307 break;
308 case kCipStringN: {
309 CipUint length = ((CipStringN *)string_2)->length;
310 CipUint size = ((CipStringN *)string_2)->size; //bytes per symbol
311 len_2 = length * size;
312 string_2_data = ((CipStringN *)string_2)->string;
313
314 }
315 break;
316 default:
317 OPENER_TRACE_ERR("CIP File: No valid String type received!\n");
318 }
319
320 /*compare strings*/ //TODO: compare works only for same data types
321 if (len_1 == len_2 && string_1_data != NULL && string_2_data != NULL) {
322 if (0 == memcmp(string_1_data, string_2_data, len_1) ) {
323 return true;
324 }
325 }
326 } //end for 1
327 } //end for 2
328
329 return false;
