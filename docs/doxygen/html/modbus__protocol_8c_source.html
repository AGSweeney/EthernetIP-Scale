55static void send_exception(int client_socket, uint16_t transaction_id, uint8_t unit_id,
56 uint8_t function_code, uint8_t exception_code)
57{
58 uint8_t response[9];
59 response[0] = (transaction_id >> 8) & 0xFF;
60 response[1] = transaction_id & 0xFF;
61 response[2] = 0x00; // Protocol ID
62 response[3] = 0x00;
63 response[4] = 0x00; // Length
64 response[5] = 0x03;
65 response[6] = unit_id;
66 response[7] = function_code | 0x80; // Exception flag
67 response[8] = exception_code;
68
69 send(client_socket, response, sizeof(response), 0);
70}

72static bool handle_read_holding_registers(int client_socket, uint16_t transaction_id,
73 uint8_t unit_id, const uint8_t *pdu, int pdu_len)
74{
75 // Read Holding Registers requires: start_addr (2 bytes) + quantity (2 bytes) = 4 bytes
76 if (pdu_len < 4) {
77 send_exception(client_socket, transaction_id, unit_id, MODBUS_FC_READ_HOLDING_REGISTERS,
78 MODBUS_EX_ILLEGAL_DATA_VALUE);
79 return true;
80 }
81
82 uint16_t start_addr = (pdu[0] << 8) | pdu[1];
83 uint16_t quantity = (pdu[2] << 8) | pdu[3];
84
85 // Validate quantity (Modbus spec: 1-125 registers)
86 if (quantity == 0 || quantity > 125) {
87 send_exception(client_socket, transaction_id, unit_id, MODBUS_FC_READ_HOLDING_REGISTERS,
88 MODBUS_EX_ILLEGAL_DATA_VALUE);
89 return true;
90 }
91
92 // Check response buffer size
93 size_t response_data_size = quantity * 2;
94 if (response_data_size > 250) { // Max 125 registers * 2 bytes = 250 bytes
95 send_exception(client_socket, transaction_id, unit_id, MODBUS_FC_READ_HOLDING_REGISTERS,
96 MODBUS_EX_ILLEGAL_DATA_VALUE);
97 return true;
98 }
99
100 uint8_t response[256];
101 int response_len = 9 + response_data_size;
102 response[0] = (transaction_id >> 8) & 0xFF;
103 response[1] = transaction_id & 0xFF;
104 response[2] = 0x00; // Protocol ID
105 response[3] = 0x00;
106 response[4] = ((response_len - 6) >> 8) & 0xFF; // Length
107 response[5] = (response_len - 6) & 0xFF;
108 response[6] = unit_id;
109 response[7] = MODBUS_FC_READ_HOLDING_REGISTERS;
110 response[8] = response_data_size; // Byte count
111
112 // Read registers from map
113 if (!modbus_read_holding_registers(start_addr, quantity, &response[9])) {
114 ESP_LOGE(TAG, "Failed to read holding registers: start_addr=%d, quantity=%d",
115 start_addr, quantity);
116 send_exception(client_socket, transaction_id, unit_id, MODBUS_FC_READ_HOLDING_REGISTERS,
117 MODBUS_EX_ILLEGAL_DATA_ADDRESS);
118 return true;
119 }
120
121 int sent = send(client_socket, response, response_len, 0);
122 if (sent != response_len) {
123 ESP_LOGE(TAG, "Failed to send full response: sent %d of %d bytes", sent, response_len);
124 }
125 return true;
126}

184static bool handle_write_single_register(int client_socket, uint16_t transaction_id,
185 uint8_t unit_id, const uint8_t *pdu, int pdu_len)
186{
187 // Write Single Register requires: address (2 bytes) + value (2 bytes) = 4 bytes
188 if (pdu_len < 4) {
189 send_exception(client_socket, transaction_id, unit_id, MODBUS_FC_WRITE_SINGLE_REGISTER,
190 MODBUS_EX_ILLEGAL_DATA_VALUE);
191 return true;
192 }
193
194 uint16_t address = (pdu[0] << 8) | pdu[1];
195 uint16_t value = (pdu[2] << 8) | pdu[3];
196
197 // Write register to map
198 if (!modbus_write_holding_register(address, value)) {
199 ESP_LOGE(TAG, "Failed to write holding register: address=%d, value=%d", address, value);
200 send_exception(client_socket, transaction_id, unit_id, MODBUS_FC_WRITE_SINGLE_REGISTER,
201 MODBUS_EX_ILLEGAL_DATA_ADDRESS);
202 return true;
203 }
204
205 // Echo back the request
206 uint8_t response[12];
207 response[0] = (transaction_id >> 8) & 0xFF;
208 response[1] = transaction_id & 0xFF;
209 response[2] = 0x00; // Protocol ID
210 response[3] = 0x00;
211 response[4] = 0x00; // Length
212 response[5] = 0x06;
213 response[6] = unit_id;
214 response[7] = MODBUS_FC_WRITE_SINGLE_REGISTER;
215 response[8] = (address >> 8) & 0xFF;
216 response[9] = address & 0xFF;
217 response[10] = (value >> 8) & 0xFF;
218 response[11] = value & 0xFF;
219
220 int sent = send(client_socket, response, sizeof(response), 0);
221 if (sent != sizeof(response)) {
222 ESP_LOGE(TAG, "Failed to send write response: sent %d of %d bytes", sent, sizeof(response));
223 }
224 return true;
225}

227static bool handle_write_multiple_registers(int client_socket, uint16_t transaction_id,
228 uint8_t unit_id, const uint8_t *pdu, int pdu_len)
229{
230 // Write Multiple Registers requires: start_addr (2) + quantity (2) + byte_count (1) + data (N) = at least 6 bytes
231 if (pdu_len < 6) {
232 send_exception(client_socket, transaction_id, unit_id, MODBUS_FC_WRITE_MULTIPLE_REGISTERS,
233 MODBUS_EX_ILLEGAL_DATA_VALUE);
234 return true;
235 }
236
237 uint16_t start_addr = (pdu[0] << 8) | pdu[1];
238 uint16_t quantity = (pdu[2] << 8) | pdu[3];
239 uint8_t byte_count = pdu[4];
240
241 // Validate parameters
242 if (quantity == 0 || quantity > 123 || byte_count != quantity * 2 || pdu_len < 5 + byte_count) {
243 send_exception(client_socket, transaction_id, unit_id, MODBUS_FC_WRITE_MULTIPLE_REGISTERS,
244 MODBUS_EX_ILLEGAL_DATA_VALUE);
245 return true;
246 }
247
248 // Write registers to map
249 if (!modbus_write_holding_registers(start_addr, quantity, &pdu[5])) {
250 ESP_LOGE(TAG, "Failed to write holding registers: start_addr=%d, quantity=%d", start_addr, quantity);
251 send_exception(client_socket, transaction_id, unit_id, MODBUS_FC_WRITE_MULTIPLE_REGISTERS,
252 MODBUS_EX_ILLEGAL_DATA_ADDRESS);
253 return true;
254 }
255
256 // Response
257 uint8_t response[12];
258 response[0] = (transaction_id >> 8) & 0xFF;
259 response[1] = transaction_id & 0xFF;
260 response[2] = 0x00; // Protocol ID
261 response[3] = 0x00;
262 response[4] = 0x00; // Length
263 response[5] = 0x06;
264 response[6] = unit_id;
265 response[7] = MODBUS_FC_WRITE_MULTIPLE_REGISTERS;
266 response[8] = (start_addr >> 8) & 0xFF;
267 response[9] = start_addr & 0xFF;
268 response[10] = (quantity >> 8) & 0xFF;
269 response[11] = quantity & 0xFF;
270
271 int sent = send(client_socket, response, sizeof(response), 0);
272 if (sent != sizeof(response)) {
273 ESP_LOGE(TAG, "Failed to send write response: sent %d of %d bytes", sent, sizeof(response));
274 }
275 return true;
276}

278bool modbus_tcp_handle_request(int client_socket)
279{
280 uint8_t mbap_header[6];
281
282 // Read MBAP header (6 bytes)
283 int received = recv(client_socket, mbap_header, sizeof(mbap_header), 0);
284
285 if (received <= 0) {
286 if (received == 0 || errno == ECONNRESET) {
287 return false; // Connection closed
288 }
289 return true; // Try again
290 }
291
292 if (received < 6) {
293 return true; // Wait for more data
294 }
295
296 uint16_t transaction_id = (mbap_header[0] << 8) | mbap_header[1];
297 uint16_t protocol_id = (mbap_header[2] << 8) | mbap_header[3];
298 uint16_t length = (mbap_header[4] << 8) | mbap_header[5];
299
300 if (protocol_id != 0 || length < 2 || length > 253) {
301 return false;
302 }
303
304 // Read PDU (unit_id + function_code + data)
305 // MBAP length field includes unit_id byte, so PDU is length bytes total
306 uint8_t pdu[256];
307 if (length > sizeof(pdu)) {
308 return false;
309 }
310
311 // Read the full PDU - may need multiple recv calls for large PDUs
312 size_t total_received = 0;
313 while (total_received < length) {
314 received = recv(client_socket, pdu + total_received, length - total_received, 0);
315 if (received <= 0) {
316 if (received == 0 || errno == ECONNRESET) {
317 return false;
318 }
319 if (errno == EAGAIN || errno == EWOULDBLOCK) {
320 // Would block, try again
321 continue;
322 }
323 ESP_LOGE(TAG, "Error reading PDU: %s", strerror(errno));
324 return false;
325 }
326 total_received += received;
327 }
328
329 if (total_received != length || length < 2) {
330 return false;
331 }
332
333 uint8_t unit_id = pdu[0];
334 uint8_t function_code = pdu[1];
335 const uint8_t *pdu_data = &pdu[2];
336 int pdu_data_len = length - 2; // Data portion after unit_id and function_code
337
338 bool result = true;
339 switch (function_code) {
340 case MODBUS_FC_READ_HOLDING_REGISTERS:
341 result = handle_read_holding_registers(client_socket, transaction_id, unit_id, pdu_data, pdu_data_len);
342 break;
343 case MODBUS_FC_READ_INPUT_REGISTERS:
344 result = handle_read_input_registers(client_socket, transaction_id, unit_id, pdu_data, pdu_data_len);
345 break;
346 case MODBUS_FC_WRITE_SINGLE_REGISTER:
347 result = handle_write_single_register(client_socket, transaction_id, unit_id, pdu_data, pdu_data_len);
348 break;
349 case MODBUS_FC_WRITE_MULTIPLE_REGISTERS:
350 result = handle_write_multiple_registers(client_socket, transaction_id, unit_id, pdu_data, pdu_data_len);
351 break;
352 default:
353 send_exception(client_socket, transaction_id, unit_id, function_code, MODBUS_EX_ILLEGAL_FUNCTION);
354 result = true;
355 break;
356 }
357
358 return result;
359}
