31void opener_init(struct netif *netif) {
32 // Create mutex on first call if needed
33 if (opener_init_mutex == NULL) {
34 opener_init_mutex = xSemaphoreCreateMutex();
35 if (opener_init_mutex == NULL) {
36 OPENER_TRACE_ERR("Failed to create opener init mutex\n");
37 return;
38 }
39 }
40
41 // Take mutex to prevent double initialization
42 if (xSemaphoreTake(opener_init_mutex, portMAX_DELAY) != pdTRUE) {
43 OPENER_TRACE_ERR("Failed to take opener init mutex\n");
44 return;
45 }
46
47 // Check if already initialized
48 if (opener_initialized) {
49 OPENER_TRACE_WARN("Opener already initialized, skipping\n");
50 xSemaphoreGive(opener_init_mutex);
51 return;
52 }
53
54 EipStatus eip_status = 0;
55
56 if (IfaceLinkIsUp(netif)) {
57 DoublyLinkedListInitialize(&connection_list,
58 CipConnectionObjectListArrayAllocator,
59 CipConnectionObjectListArrayFree);
60
61 uint8_t iface_mac[6];
62 IfaceGetMacAddress(netif, iface_mac);
63
64 SetDeviceSerialNumber(123456789);
65
66 // Use hardware random number generator instead of rand()
67 EipUint16 unique_connection_id = (EipUint16)(esp_random() & 0xFFFF);
68
69 EipStatus eip_status = CipStackInit(unique_connection_id);
70
71 CipClass *tcp_ip_class = GetCipClass(kCipTcpIpInterfaceClassCode);
72 if (NULL != tcp_ip_class) {
73 InsertGetSetCallback(tcp_ip_class, NvTcpipSetCallback, kNvDataFunc);
74 }
75
76 CipEthernetLinkSetMac(iface_mac);
77
78 GetHostName(netif, &g_tcpip.hostname);
79
80 g_end_stack = 0;
81
82 eip_status = IfaceGetConfiguration(netif, &g_tcpip.interface_configuration);
83 if (eip_status < 0) {
84 OPENER_TRACE_WARN("Problems getting interface configuration\n");
85 }
86
87 eip_status = NetworkHandlerInitialize();
88 }
89 else {
90 OPENER_TRACE_WARN("Network link is down, OpENer not started\n");
91 g_end_stack = 1;
92 }
93 if ((g_end_stack == 0) && (eip_status == kEipStatusOk)) {
94 // Pin OpENer task to Core 0 (same as LWIP TCP/IP task)
95 BaseType_t result = xTaskCreatePinnedToCore(opener_thread,
96 "OpENer",
97 OPENER_STACK_SIZE,
98 netif,
99 OPENER_THREAD_PRIO,
100 &opener_task_handle,
101 0); // Core 0
102 if (result == pdPASS) {
103 opener_initialized = true;
104 OPENER_TRACE_INFO("OpENer: opener_thread started on Core 0, free heap size: %d\n",
105 xPortGetFreeHeapSize());
106 } else {
107 OPENER_TRACE_ERR("Failed to create OpENer task\n");
108 }
109 } else {
110 OPENER_TRACE_ERR("NetworkHandlerInitialize error %d\n", eip_status);
111 }
112
113 xSemaphoreGive(opener_init_mutex);
114}

116static void opener_thread(void *argument) {
117 struct netif *netif = (struct netif*) argument;
118 while (!g_end_stack) {
119 if (kEipStatusOk != NetworkHandlerProcessCyclic()) {
120 OPENER_TRACE_ERR("Error in NetworkHandler loop! Exiting OpENer!\n");
121 g_end_stack = 1;
122 }
123 if (!IfaceLinkIsUp(netif)) {
124 OPENER_TRACE_INFO("Network link is down, exiting OpENer\n");
125 g_end_stack = 1;
126 }
127 }
128 NetworkHandlerFinish();
129 ShutdownCipStack();
130
131 // Mark as not initialized and clear task handle atomically
132 if (opener_init_mutex != NULL) {
133 if (xSemaphoreTake(opener_init_mutex, portMAX_DELAY) == pdTRUE) {
134 opener_initialized = false;
135 opener_task_handle = NULL;
136 xSemaphoreGive(opener_init_mutex);
137 }
138 } else {
139 opener_task_handle = NULL;
140 }
141
142 vTaskDelete(NULL);
143}
