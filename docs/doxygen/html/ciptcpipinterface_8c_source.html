108static bool isalnum_c(const EipByte byte) {
109 return
110 ('a' <= byte && byte <= 'z') ||
111 ('A' <= byte && byte <= 'Z') ||
112 ('0' <= byte && byte <= '9');
113}

128static bool IsValidNameLabel(const EipByte *label) {
129 if (!isalnum_c(*label) ) {
130 return false;
131 }
132 ++label;
133 while ('\0' != *label && (isalnum_c(*label) || '-' == *label) ) {
134 ++label;
135 }
136 return ('\0' == *label && '-' != label[-1]);
137}

152static bool IsValidDomain(EipByte *domain) {
153 bool status = true;
154
155 OPENER_TRACE_INFO("Enter '%s'->", domain);
156 if ('.' == *domain) { /* Forbid leading dot */
157 return false;
158 }
159 EipByte *dot = (EipByte *)strchr( (char *)domain, '.' );
160 if (dot) {
161 bool rc;
162
163 *dot = '\0';
164 status &= rc = IsValidNameLabel(domain);
165 OPENER_TRACE_INFO("Checked %d '%s'\n", rc, domain);
166 if ('\0' != dot[1]) {
167 status &= IsValidDomain(dot + 1);
168 }
169 else { /* Forbid trailing dot */
170 status = false;
171 }
172 *dot = '.';
173 }
174 else {
175 status = IsValidNameLabel(domain);
176 OPENER_TRACE_INFO("Check end %d '%s'\n", status, domain);
177 }
178 return status;
179}

190static bool IsValidNetmask(in_addr_t netmask) {
191 in_addr_t v = ntohl(netmask);
192
193 v = ~v; /* Create the host mask */
194 ++v; /* This must be a power of 2 then */
195 bool valid = v && !(v & (v - 1) ); /* Check if it is a power of 2 */
196
197 return valid && (INADDR_BROADCAST != netmask);
198}

207static bool IsInClassAbc(in_addr_t ip_addr) {
208 in_addr_t ip = ntohl(ip_addr);
209 return IN_CLASSA(ip) || IN_CLASSB(ip) || IN_CLASSC(ip);
210}

220static bool IsOnLoopbackNetwork(in_addr_t ip_addr) {
221 in_addr_t ip = ntohl(ip_addr);
222 return (ip & IN_CLASSA_NET) == (INADDR_LOOPBACK & IN_CLASSA_NET);
223}

235static bool IsNetworkOrBroadcastIp(in_addr_t ip_addr,
236 in_addr_t net_mask) {
237 return ( (ip_addr & net_mask) == ip_addr ) || /* is network address */
238 ( (ip_addr | ~net_mask) == ip_addr ); /* is broadcast address */
239}

263CipBool CipTcpIpIsValidNetworkConfig(const CipTcpIpInterfaceConfiguration *if_cfg) {
264 if (INADDR_ANY == ntohl(if_cfg->ip_address) ) { /* N0 */
265 return false;
266 }
267 if (INADDR_ANY != ntohl(if_cfg->network_mask) && /* MASK */
268 !IsValidNetmask(if_cfg->network_mask) ) {
269 return false;
270 }
271 if (!IsInClassAbc(if_cfg->ip_address) || /* ABC */
272 !IsInClassAbc(if_cfg->gateway) ||
273 !IsInClassAbc(if_cfg->name_server) ||
274 !IsInClassAbc(if_cfg->name_server_2) ) {
275 return false;
276 }
277 if (IsOnLoopbackNetwork(if_cfg->ip_address) || /* NLCL */
278 IsOnLoopbackNetwork(if_cfg->gateway) ) {
279 return false;
280 }
281 /* Check NB */
282 if (IsNetworkOrBroadcastIp(if_cfg->ip_address, if_cfg->network_mask) ||
283 (INADDR_ANY != ntohl(if_cfg->gateway) &&
284 IsNetworkOrBroadcastIp(if_cfg->gateway, if_cfg->network_mask) ) ) {
285 return false;
286 }
287 if (INADDR_ANY != ntohl(if_cfg->gateway) &&
288 INADDR_ANY != ntohl(if_cfg->network_mask) ) {
289 /* gateway is configured. Check if it is reachable. */
290 if ( (if_cfg->network_mask & if_cfg->ip_address) !=
291 (if_cfg->network_mask & if_cfg->gateway) ) {
292 return false;
293 }
294 }
295 return true;
296}

311 }
312
313 return false;

320/* MODIFICATION: Storage structure for EtherNet/IP TCP/IP Interface Object Attribute #11
321 * Added by: Adam G. Sweeney <agsweeney@gmail.com>
322 * This structure stores the last ACD conflict data (activity status, MAC address,
323 * and raw ARP frame) as required by EtherNet/IP specification for Attribute #11
324 * "Last Conflict Detected"

327 CipUsint activity;
328 CipUsint remote_mac[6];

332static void TcpipSetLastConflictState(CipUsint activity) {
333 s_tcpip_last_conflict.activity = activity;
334 /* MAC and raw_data persist across activity changes to maintain conflict history.
335 * They are only set when a conflict is detected (via CipTcpIpSetLastAcdMac/RawData)
336 * and should remain available for diagnostic purposes even after the conflict
337 * is resolved or ACD transitions to a different state.
338 */
339}

372void CipTcpIpSetLastAcdRawData(const uint8_t *data, size_t length) {
373 if (NULL == data) {
374 memset(s_tcpip_last_conflict.raw_data, 0, sizeof(s_tcpip_last_conflict.raw_data));

376 }
377 if (length > sizeof(s_tcpip_last_conflict.raw_data)) {
378 length = sizeof(s_tcpip_last_conflict.raw_data);
379 }
380 memcpy(s_tcpip_last_conflict.raw_data, data, length);

435/* MODIFICATION: Encode EtherNet/IP TCP/IP Interface Object Attribute #11
436 * Added by: Adam G. Sweeney <agsweeney@gmail.com>
437 * This function encodes the last ACD conflict data (activity status, MAC address,
438 * and raw ARP frame) for EtherNet/IP Attribute #11 "Last Conflict Detected"
439 */
440void EncodeCipLastConflictDetected(const void *const data,
441 ENIPMessage *const outgoing_message) {
442 (void)data;
443
444 outgoing_message->current_message_position[0] = s_tcpip_last_conflict.activity;
445 MoveMessageNOctets(1, outgoing_message);
446
447 memcpy(outgoing_message->current_message_position,
448 s_tcpip_last_conflict.remote_mac,
449 sizeof(s_tcpip_last_conflict.remote_mac));
450 MoveMessageNOctets(sizeof(s_tcpip_last_conflict.remote_mac), outgoing_message);
451
452 memcpy(outgoing_message->current_message_position,
453 s_tcpip_last_conflict.raw_data,
454 sizeof(s_tcpip_last_conflict.raw_data));
455 MoveMessageNOctets(sizeof(s_tcpip_last_conflict.raw_data), outgoing_message);
456}

459int DecodeTcpIpInterfaceConfigurationControl( /* Attribute 3 */
460 void *const data,
461 CipMessageRouterRequest *const message_router_request,
462 CipMessageRouterResponse *const message_router_response) {
463
464 int number_of_decoded_bytes = -1;
465
466 CipDword configuration_control_received = GetDintFromMessage(
467 &(message_router_request->data));
468 if ((configuration_control_received & kTcpipCfgCtrlMethodMask) >= 0x03
469 || (configuration_control_received & ~kTcpipCfgCtrlMethodMask)) {
470 message_router_response->general_status =
471 kCipErrorInvalidAttributeValue;
472
473 } else {
474
475 /* Set reserved bits to zero on reception. */
476 configuration_control_received &= (kTcpipCfgCtrlMethodMask
477 | kTcpipCfgCtrlDnsEnable);
478
479 *(CipDword *)data = configuration_control_received;
480 number_of_decoded_bytes = 4;
481 message_router_response->general_status = kCipErrorSuccess;
482 }
483
484 return number_of_decoded_bytes;
485}

490int DecodeCipTcpIpInterfaceConfiguration( /* Attribute 5 */
491 CipTcpIpInterfaceConfiguration *const data, //kCipUdintUdintUdintUdintUdintString
492 CipMessageRouterRequest *const message_router_request,
493 CipMessageRouterResponse *const message_router_response) {
494
495 int number_of_decoded_bytes = -1;
496
497 CipTcpIpInterfaceConfiguration if_cfg;
498 CipUdint tmp_ip;
499
500 if (IsIOConnectionActive()) {
501 OPENER_TRACE_WARN("Applying TCP/IP config while IO connection active\n");
502 }
503 bool switch_to_static = (kTcpipCfgCtrlStaticIp
504 != (g_tcpip.config_control & kTcpipCfgCtrlMethodMask));
505 memset(&if_cfg, 0, sizeof if_cfg);
506 tmp_ip = GetUdintFromMessage(&(message_router_request->data));
507 if_cfg.ip_address = htonl(tmp_ip);
508 tmp_ip = GetUdintFromMessage(&(message_router_request->data));
509 if_cfg.network_mask = htonl(tmp_ip);
510 tmp_ip = GetUdintFromMessage(&(message_router_request->data));
511 if_cfg.gateway = htonl(tmp_ip);
512 tmp_ip = GetUdintFromMessage(&(message_router_request->data));
513 if_cfg.name_server = htonl(tmp_ip);
514 tmp_ip = GetUdintFromMessage(&(message_router_request->data));
515 if_cfg.name_server_2 = htonl(tmp_ip);
516
517 CipUint domain_name_length = GetUintFromMessage(
518 &(message_router_request->data));
519 if (domain_name_length > kTcpipMaxDomainLength) { /* see Vol. 2, Table 5-4.3 Instance Attributes */
520 message_router_response->general_status = kCipErrorTooMuchData;
521 return number_of_decoded_bytes;
522 }
523 if (NULL == SetCipStringByData(&if_cfg.domain_name, domain_name_length,
524 message_router_request->data) ) {
525 message_router_response->general_status = kCipErrorNotEnoughData;
526 return number_of_decoded_bytes;
527 }
528 domain_name_length = (domain_name_length + 1) & (~0x0001u); /* Align for possible pad byte */
529 const char *domain_str = (if_cfg.domain_name.length > 0 && NULL != if_cfg.domain_name.string)
530 ? (const char *)if_cfg.domain_name.string : "";
531 OPENER_TRACE_INFO("Domain: ds %hu '%s'\n",
532 domain_name_length,
533 domain_str);
534
535 bool domain_valid = true;
536 if (if_cfg.domain_name.length > 0) {
537 char domain_buf[kTcpipMaxDomainLength + 1] = {0};
538 CipUint len = if_cfg.domain_name.length;
539 if (len > kTcpipMaxDomainLength) {
540 len = kTcpipMaxDomainLength;
541 }
542 memcpy(domain_buf, if_cfg.domain_name.string, len);
543 domain_valid = IsValidDomain((EipByte *)domain_buf);
544 }
545
546 if (!CipTcpIpIsValidNetworkConfig(&if_cfg) || !domain_valid) {
547 message_router_response->general_status =
548 kCipErrorInvalidAttributeValue;
549 return number_of_decoded_bytes;
550 }
551
552 if (switch_to_static) {
553 g_tcpip.config_control &= ~kTcpipCfgCtrlMethodMask;
554 g_tcpip.config_control |= kTcpipCfgCtrlStaticIp;
555 }
556
557 *data = if_cfg; //write data to attribute
558 number_of_decoded_bytes = 20 + domain_name_length;
559
560 /* Tell that this configuration change becomes active after a reset */
561 g_tcpip.status |= kTcpipStatusIfaceCfgPend;
562 message_router_response->general_status = kCipErrorSuccess;
563
564 return number_of_decoded_bytes;
565
566}

568int DecodeCipTcpIpInterfaceHostName( /* Attribute 6 */
569 CipString *const data,
570 CipMessageRouterRequest *const message_router_request,
571 CipMessageRouterResponse *const message_router_response) {
572
573 int number_of_decoded_bytes = -1;
574
575 CipString tmp_host_name = {
576 .length = 0u,
577 .string = NULL
578 };
579 CipUint host_name_length =
580 GetUintFromMessage(&(message_router_request->data) );
581 if (host_name_length > kTcpipMaxHostnameLength) { /* see RFC 1123 on more details */
582 message_router_response->general_status = kCipErrorTooMuchData;
583 return number_of_decoded_bytes;
584 }
585 if (NULL == SetCipStringByData(&tmp_host_name,
586 host_name_length,
587 message_router_request->data) ) {
588 message_router_response->general_status = kCipErrorNotEnoughData;
589 return number_of_decoded_bytes;
590 }
591 CipUint padded_length = (host_name_length + 1) & (~0x0001u); /* Align for possible pad byte */
592 const char *host_str = (tmp_host_name.length > 0 && NULL != tmp_host_name.string)
593 ? (const char *)tmp_host_name.string : "";
594 OPENER_TRACE_INFO("Host Name: ds %hu '%s'\n",
595 padded_length,
596 host_str);
597
598 bool hostname_valid = true;
599 if (host_name_length > 0) {
600 char host_buf[kTcpipMaxHostnameLength + 1] = {0};
601 CipUint len = host_name_length;
602 if (len > kTcpipMaxHostnameLength) {
603 len = kTcpipMaxHostnameLength;
604 }
605 memcpy(host_buf, tmp_host_name.string, len);
606 hostname_valid = IsValidDomain((EipByte *)host_buf);
607 }
608
609 if (!hostname_valid) {
610 message_router_response->general_status =
611 kCipErrorInvalidAttributeValue;
612 return number_of_decoded_bytes;
613 }
614
615 *data = tmp_host_name; //write data to attribute
616
617 /* Tell that this configuration change becomes active after a reset */
618 g_tcpip.status |= kTcpipStatusIfaceCfgPend;
619 message_router_response->general_status = kCipErrorSuccess;
620
621 return number_of_decoded_bytes;
622
623}

646 CipMessageRouterRequest *const message_router_request,
647 CipMessageRouterResponse *const message_router_response) {
648 if (message_router_request->request_data_size < 1) {
649 message_router_response->general_status = kCipErrorNotEnoughData;

656 return -1;

660 message_router_response->general_status = kCipErrorSuccess;

662 g_tcpip.status &= ~(kTcpipStatusAcdStatus | kTcpipStatusAcdFault);
663 (void)NvTcpipStore(&g_tcpip);
664
665 return 1;

694EipStatus CipTcpIpInterfaceInit() {
695 CipClass *tcp_ip_class = NULL;
696
697 if ( ( tcp_ip_class = CreateCipClass(kCipTcpIpInterfaceClassCode, /* class code */
698 0, /* # class attributes */
699 7, /* # highest class attribute number */
700 2, /* # class services */
701 13, /* # instance attributes */
702 13, /* # highest instance attribute number */
703 3, /* # instance services */
704 1, /* # instances */
705 "TCP/IP interface", 4, /* # class revision */
706 NULL /* # function pointer for initialization */
707 ) ) == 0 ) {
708 return kEipStatusError;
709 }
710
711 CipInstance *instance = GetCipInstance(tcp_ip_class, 1); /* bind attributes to the instance #1 that was created above */
712
713 InsertAttribute(instance,
714 1,
715 kCipDword,
716 EncodeCipDword,
717 NULL,
718 &g_tcpip.status,
719 kGetableSingleAndAll);
720 InsertAttribute(instance,
721 2,
722 kCipDword,
723 EncodeCipDword,
724 NULL,
725 &g_tcpip.config_capability,
726 kGetableSingleAndAll);

863void CipTcpIpCalculateMulticastIp(CipTcpIpObject *const tcpip) {
864 /* Multicast base address according to spec: 239.192.1.0 */
865 static const CipUdint cip_mcast_base_addr = 0xEFC00100;
866
867 /* Calculate the CIP multicast address. The multicast address is calculated, not input */
868 CipUdint host_id = ntohl(tcpip->interface_configuration.ip_address) &
869 ~ntohl(tcpip->interface_configuration.network_mask);
870 host_id -= 1;
871 host_id &= 0x3ff;
872
873 tcpip->mcast_config.starting_multicast_address =
874 htonl(cip_mcast_base_addr + (host_id << 5) );
875}
