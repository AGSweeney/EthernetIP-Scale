127CipConnectionObject *GetIoConnectionForConnectionData(
128 CipConnectionObject *const RESTRICT connection_object,
129 EipUint16 *const extended_error) {
130
131 *extended_error = 0;
132
133 CipConnectionObject *io_connection = GetExclusiveOwnerConnection(
134 connection_object,
135 extended_error);
136 if (NULL == io_connection) {
137 if (kConnectionManagerExtendedStatusCodeSuccess == *extended_error) {
138 /* we found no connection and don't have an error so try input only next */
139 io_connection = GetInputOnlyConnection(connection_object, extended_error);
140 if (NULL == io_connection) {
141 if (kConnectionManagerExtendedStatusCodeSuccess == *extended_error) {
142 /* we found no connection and don't have an error so try listen only next */
143 io_connection = GetListenOnlyConnection(connection_object,
144 extended_error);
145 if ( (NULL == io_connection) &&
146 (kCipErrorSuccess == *extended_error) ) {
147 /* no application connection type was found that suits the given data */
148 *extended_error =
149 kConnectionManagerExtendedStatusCodeInconsistentApplicationPathCombo;
150 } else {
151 ConnectionObjectSetInstanceType(connection_object,
152 kConnectionObjectInstanceTypeIOListenOnly);
153 OPENER_TRACE_INFO("IO Listen only connection requested\n");
154 //Is listen only connection
155 }
156 }
157 } else {
158 ConnectionObjectSetInstanceType(connection_object,
159 kConnectionObjectInstanceTypeIOInputOnly);
160 OPENER_TRACE_INFO("IO Input only connection requested\n");
161 //is Input only connection
162 }
163 }
164 } else {
165 ConnectionObjectSetInstanceType(connection_object,
166 kConnectionObjectInstanceTypeIOExclusiveOwner);
167 OPENER_TRACE_INFO("IO Exclusive Owner connection requested\n");
168 //Is exclusive owner connection
169 }
170
171 if (NULL != io_connection) {
172 ConnectionObjectDeepCopy(io_connection, connection_object);
173 }
174
175 return io_connection;
176}

178CipConnectionObject *GetExclusiveOwnerConnection(
179 const CipConnectionObject *const RESTRICT connection_object,
180 EipUint16 *const extended_error) {
181
182 for (size_t i = 0; i < OPENER_CIP_NUM_EXLUSIVE_OWNER_CONNS; ++i) {
183 if ( (g_exlusive_owner_connections[i].output_assembly ==
184 connection_object->consumed_path.instance_id)
185 && (g_exlusive_owner_connections[i].input_assembly ==
186 connection_object->produced_path.instance_id)
187 && ( (g_exlusive_owner_connections[i].config_assembly ==
188 connection_object->configuration_path.instance_id)
189 || (0 == connection_object->configuration_path.instance_id) ) ) {
190
191 /* check if on other connection point with the same output assembly is currently connected */
192 const CipConnectionObject *const exclusive_owner =
193 GetConnectedOutputAssembly(
194 connection_object->produced_path.instance_id);
195 if ( NULL
196 != exclusive_owner ) {
197 if(kConnectionObjectStateEstablished ==
198 ConnectionObjectGetState(exclusive_owner) ) {
199 *extended_error =
200 kConnectionManagerExtendedStatusCodeErrorOwnershipConflict;
201 OPENER_TRACE_INFO("Hit an Ownership conflict in appcontype.c:198\n");
202 break;
203 }
204 if(kConnectionObjectStateTimedOut ==
205 ConnectionObjectGetState(exclusive_owner)
206 && ConnectionObjectEqualOriginator(connection_object,
207 exclusive_owner) ) {
208 g_exlusive_owner_connections[i].connection_data.
209 connection_close_function(&(g_exlusive_owner_connections[i].
210 connection_data) );
211 return &(g_exlusive_owner_connections[i].connection_data);
212 } else {
213 *extended_error =
214 kConnectionManagerExtendedStatusCodeErrorOwnershipConflict;
215 OPENER_TRACE_INFO(
216 "Hit an Ownership conflict with timed out connection");
217 break;
218 }
219 }
220 return &(g_exlusive_owner_connections[i].connection_data);
221 }
222 }
223 return NULL;
224}

226CipConnectionObject *GetInputOnlyConnection(
227 const CipConnectionObject *const RESTRICT connection_object,
228 EipUint16 *const extended_error) {
229 EipUint16 err = 0;
230
231 for (size_t i = 0; i < OPENER_CIP_NUM_INPUT_ONLY_CONNS; ++i) {
232 if (g_input_only_connections[i].output_assembly
233 == connection_object->consumed_path.instance_id) { /* we have the same output assembly */
234 if (g_input_only_connections[i].input_assembly
235 != connection_object->produced_path.instance_id) {
236 err = kConnectionManagerExtendedStatusCodeInvalidProducingApplicationPath;
237 continue;
238 }
239 if (g_input_only_connections[i].config_assembly
240 != connection_object->configuration_path.instance_id
241 && 0 != connection_object->configuration_path.instance_id) {
242 err = kConnectionManagerExtendedStatusCodeInconsistentApplicationPathCombo;
243 continue;
244 }
245
246 for (size_t j = 0; j < OPENER_CIP_NUM_INPUT_ONLY_CONNS_PER_CON_PATH;
247 ++j) {
248 if (kConnectionObjectStateTimedOut
249 == ConnectionObjectGetState(&(g_input_only_connections[i].
250 connection_data[j]) )
251 && ConnectionObjectEqualOriginator(connection_object,
252 &(g_input_only_connections[i].
253 connection_data[j]))) {
254 g_input_only_connections[i].connection_data[j].
255 connection_close_function(
256 &g_input_only_connections[i].connection_data[j]);
257 return &(g_input_only_connections[i].connection_data[j]);
258 }
259 }
260
261 for (size_t j = 0; j < OPENER_CIP_NUM_INPUT_ONLY_CONNS_PER_CON_PATH;
262 ++j) {
263 if (kConnectionObjectStateNonExistent
264 == ConnectionObjectGetState(&(g_input_only_connections[i].
265 connection_data[j]) ) ) {
266 return &(g_input_only_connections[i].connection_data[j]);
267 }
268 }
269 err = kConnectionManagerExtendedStatusCodeTargetObjectOutOfConnections;
270 break;
271 }
272 }
273
274 *extended_error = err;
275 return NULL;
276}

278CipConnectionObject *GetListenOnlyConnection(
279 const CipConnectionObject *const RESTRICT connection_object,
280 EipUint16 *const extended_error) {
281 EipUint16 err = 0;
282
283 for (size_t i = 0; i < OPENER_CIP_NUM_LISTEN_ONLY_CONNS; i++) {
284 if (g_listen_only_connections[i].output_assembly
285 == connection_object->consumed_path.instance_id) { /* we have the same output assembly */
286 if (g_listen_only_connections[i].input_assembly
287 != connection_object->produced_path.instance_id) {
288 err = kConnectionManagerExtendedStatusCodeInvalidProducingApplicationPath;
289 continue;
290 }
291 if (g_listen_only_connections[i].config_assembly
292 != connection_object->configuration_path.instance_id
293 && 0 != connection_object->configuration_path.instance_id) {
294 err = kConnectionManagerExtendedStatusCodeInconsistentApplicationPathCombo;
295 continue;
296 }
297
298 /* Here we look for both Point-to-Point and Multicast IO connections */
299 if ( NULL == GetExistingProducerIoConnection(false,
300 connection_object->
301 produced_path.instance_id)) {
302 err = kConnectionManagerExtendedStatusCodeNonListenOnlyConnectionNotOpened;
303 break;
304 }
305
306 for (size_t j = 0; j < OPENER_CIP_NUM_LISTEN_ONLY_CONNS_PER_CON_PATH;
307 ++j) {
308 if (kConnectionObjectStateTimedOut
309 == ConnectionObjectGetState(&(g_listen_only_connections[i].
310 connection_data[j]) )
311 && ConnectionObjectEqualOriginator(connection_object,
312 &(g_listen_only_connections[i].
313 connection_data[j]))) {
314 g_listen_only_connections[i].connection_data[j].
315 connection_close_function(
316 &g_listen_only_connections[i].connection_data[j]);
317 return &(g_listen_only_connections[i].connection_data[j]);
318 }
319 }
320
321 for (size_t j = 0; j < OPENER_CIP_NUM_LISTEN_ONLY_CONNS_PER_CON_PATH;
322 j++) {
323 if (kConnectionObjectStateNonExistent
324 == ConnectionObjectGetState(&(g_listen_only_connections[i].
325 connection_data[j]) ) ) {
326 return &(g_listen_only_connections[i].connection_data[j]);
327 }
328 }
329 err = kConnectionManagerExtendedStatusCodeTargetObjectOutOfConnections;
330 break;
331 }
332 }
333
334 *extended_error = err;
335 return NULL;

338CipConnectionObject *GetExistingProducerIoConnection(
339 const bool multicast_only,
340 const EipUint32 input_point) {
341 const DoublyLinkedListNode *node = connection_list.first;
342
343 while (NULL != node) {
344 CipConnectionObject *producer_io_connection = node->data;
345 if (ConnectionObjectIsTypeIOConnection(producer_io_connection) &&
346 (input_point == producer_io_connection->produced_path.instance_id) &&
347 (kEipInvalidSocket !=
348 producer_io_connection->socket[kUdpCommuncationDirectionProducing]) )
349 {
350 ConnectionObjectConnectionType cnxn_type =
351 ConnectionObjectGetTToOConnectionType(producer_io_connection);
352 /* we have a connection that produces the same input assembly,
353 * and manages the connection.
354 */
355 if (kConnectionObjectConnectionTypeMulticast == cnxn_type) {
356 return producer_io_connection;
357 }
358 if (!multicast_only &&
359 kConnectionObjectConnectionTypePointToPoint == cnxn_type)
360 {
361 return producer_io_connection;
362 }
363 }
364 node = node->next;
365 }
366 return NULL;
367}

385 next_non_control_master_connection)

400void CloseAllConnectionsForInputWithSameType(const EipUint32 input_point,
401 const ConnectionObjectInstanceType instance_type)
402{
403
404 OPENER_TRACE_INFO("Close all instance type %d only connections\n",
405 instance_type);
406 const DoublyLinkedListNode *node = connection_list.first;
407 while (NULL != node) {
408 CipConnectionObject *const connection = node->data;
409 node = node->next;
410 if ( (instance_type == ConnectionObjectGetInstanceType(connection) )
411 && (input_point == connection->produced_path.instance_id) ) {
412 CipConnectionObject *connection_to_delete = connection;
413 CheckIoConnectionEvent(
414 connection_to_delete->consumed_path.instance_id,
415 connection_to_delete->produced_path.instance_id,
416 kIoConnectionEventClosed);
417
418 assert(connection_to_delete->connection_close_function != NULL);
419 connection_to_delete->connection_close_function(connection_to_delete);
420 }
421 }
