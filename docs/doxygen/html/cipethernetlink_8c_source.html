212static EipStatus GetAttributeAllEthernetLink(
213 CipInstance *instance,
214 CipMessageRouterRequest *message_router_request,
215 CipMessageRouterResponse *message_router_response,
216 const struct sockaddr *originator_address,
217 const CipSessionHandle encapsulation_session) {
218 (void)originator_address;
219 (void)encapsulation_session;
220
221 InitializeENIPMessage(&message_router_response->message);
222
223 if (0 == instance->cip_class->number_of_attributes) {
224 message_router_response->reply_service =
225 (0x80 | message_router_request->service);
226 message_router_response->general_status = kCipErrorServiceNotSupported;
227 message_router_response->size_of_additional_status = 0;
228 return kEipStatusOkSend;
229 }
230
231 GenerateGetAttributeSingleHeader(message_router_request,
232 message_router_response);
233 message_router_response->general_status = kCipErrorSuccess;
234
235 CipAttributeStruct *attribute = instance->attributes;
236 for (size_t j = 0; j < instance->cip_class->number_of_attributes; ++j) {
237 EipUint16 attribute_number = attribute->attribute_number;
238 if ( (instance->cip_class->get_all_bit_mask[CalculateIndex(attribute_number)])
239 & (1 << (attribute_number % 8)) ) {
240 message_router_request->request_path.attribute_number = attribute_number;
241
242 if ((attribute->attribute_flags & kPreGetFunc) &&
243 NULL != instance->cip_class->PreGetCallback) {
244 instance->cip_class->PreGetCallback(instance,
245 attribute,
246 message_router_request->service);
247 }
248
249 attribute->encode(attribute->data, &message_router_response->message);
250
251 if ((attribute->attribute_flags & kPostGetFunc) &&
252 NULL != instance->cip_class->PostGetCallback) {
253 instance->cip_class->PostGetCallback(instance,
254 attribute,
255 message_router_request->service);
256 }
257 }
258 ++attribute;
259 }
260
261 return kEipStatusOkSend;
262}

273EipStatus CipEthernetLinkInit(void) {
274 CipClass *ethernet_link_class = CreateCipClass(kCipEthernetLinkClassCode,
275 0,
276 /* # class attributes*/
277 7,
278 /* # highest class attribute number*/
279 2,
280 /* # class services*/
281 11,
282 /* # instance attributes*/
283 11,
284 /* # highest instance attribute number*/
285 /* # instance services follow */
286 2 + OPENER_ETHLINK_CNTRS_ENABLE + OPENER_ETHLINK_IFACE_CTRL_ENABLE,
287 OPENER_ETHLINK_INSTANCE_CNT,
288 /* # instances*/
289 "Ethernet Link",
290 /* # class name */
291 4,
292 /* # class revision*/
293 NULL); /* # function pointer for initialization*/
294
295 /* set attributes to initial values */
296 for (size_t idx = 0; idx < OPENER_ETHLINK_INSTANCE_CNT; ++idx) {
297 g_ethernet_link[idx].interface_speed = 100;
298 /* successful speed and duplex neg, full duplex active link.
299 * TODO: in future it should be checked if link is active */
300 g_ethernet_link[idx].interface_flags = 0xF;
301
302 g_ethernet_link[idx].interface_type = kEthLinkIfTypeTwistedPair;
303 if (2 == idx) {
304 g_ethernet_link[idx].interface_type = kEthLinkIfTypeInternal;
305 }
306 s_interface_state[idx] = kEthLinkInterfaceStateUnknown;
307#if defined(OPENER_ETHLINK_LABEL_ENABLE) && 0 != OPENER_ETHLINK_LABEL_ENABLE
308 g_ethernet_link[idx].interface_label = iface_label_table[idx];
309#endif
310 g_ethernet_link[idx].interface_caps.capability_bits = kEthLinkCapAutoNeg;
311 g_ethernet_link[idx].interface_caps.speed_duplex_selector =
312 kEthLinkSpeedDpx_100_FD;
313#if defined(OPENER_ETHLINK_IFACE_CTRL_ENABLE) && \
314 0 != OPENER_ETHLINK_IFACE_CTRL_ENABLE
315 g_ethernet_link[idx].interface_control.control_bits =
316 kEthLinkIfCntrlAutonegotiate;
317 g_ethernet_link[idx].interface_control.forced_interface_speed = 0U;
318#endif
319 }
320
321 if (ethernet_link_class != NULL) {
322 /* add services to the class */
323 InsertService(ethernet_link_class, kGetAttributeSingle,
324 &GetAttributeSingle,
325 "GetAttributeSingle");
326 InsertService(ethernet_link_class, kGetAttributeAll,
327 &GetAttributeAllEthernetLink,
328 "GetAttributeAll");
329
330#if defined(OPENER_ETHLINK_CNTRS_ENABLE) && 0 != OPENER_ETHLINK_CNTRS_ENABLE
331 InsertService(ethernet_link_class, kEthLinkGetAndClear,
332 &GetAndClearEthernetLink, "GetAndClear");
333#endif
334#if defined(OPENER_ETHLINK_IFACE_CTRL_ENABLE) && \
335 0 != OPENER_ETHLINK_IFACE_CTRL_ENABLE
336 InsertService(ethernet_link_class, kSetAttributeSingle,
337 &SetAttributeSingle,
338 "SetAttributeSingle");
339#endif
340
341 /* bind attributes to the instance */
342 for (CipInstanceNum idx = 0; idx < OPENER_ETHLINK_INSTANCE_CNT; ++idx) {
343 CipInstance *ethernet_link_instance =
344 GetCipInstance( ethernet_link_class, (CipInstanceNum)(idx + 1) );
345
346 InsertAttribute(ethernet_link_instance,
347 1,
348 kCipUdint,
349 EncodeCipUdint,
350 NULL,
351 &g_ethernet_link[idx].interface_speed,
352 kGetableSingleAndAll);
353 InsertAttribute(ethernet_link_instance,
354 2,
355 kCipDword,
356 EncodeCipDword,
357 NULL,
358 &g_ethernet_link[idx].interface_flags,
359 kGetableSingleAndAll);
360 InsertAttribute(ethernet_link_instance,
361 3,
362 kCip6Usint,
363 EncodeCipEthernetLinkPhyisicalAddress,
364 NULL,
365 &g_ethernet_link[idx].physical_address,
366 kGetableSingleAndAll);
367#if defined(OPENER_ETHLINK_CNTRS_ENABLE) && 0 != OPENER_ETHLINK_CNTRS_ENABLE
368 InsertAttribute(ethernet_link_instance,
369 4,
370 kCipAny,
371 EncodeCipEthernetLinkInterfaceCounters,
372 NULL,
373 &g_ethernet_link[idx].interface_cntrs,
374 kGetableSingleAndAll);
375 InsertAttribute(ethernet_link_instance,
376 5,
377 kCipAny,
378 EncodeCipEthernetLinkMediaCounters,
379 NULL,
380 &g_ethernet_link[idx].media_cntrs,
381 kGetableSingleAndAll);
382#else
383 InsertAttribute(ethernet_link_instance,
384 4,
385 kCipAny,
386 EncodeCipEthernetLinkInterfaceCounters,
387 NULL,
388 &dummy_attribute_udint,
389 kGetableSingleAndAll);
390 InsertAttribute(ethernet_link_instance,
391 5,
392 kCipAny,
393 EncodeCipEthernetLinkMediaCounters,
394 NULL,
395 &dummy_attribute_udint,
396 kGetableSingleAndAll);
397#endif /* ... && 0 != OPENER_ETHLINK_CNTRS_ENABLE */
398#if defined(OPENER_ETHLINK_IFACE_CTRL_ENABLE) && \
399 0 != OPENER_ETHLINK_IFACE_CTRL_ENABLE
400 if (2 == idx) {
401 /* Interface control of internal switch port is never settable. */
402 InsertAttribute(ethernet_link_instance,
403 6,
404 kCipAny,
405 EncodeCipEthernetLinkInterfaceControl,
406 DecodeCipEthernetLinkInterfaceControl,
407 &g_ethernet_link[idx].interface_control,
408 IFACE_CTRL_ACCESS_MODE & ~kSetable);
409 } else {
410 InsertAttribute(ethernet_link_instance,
411 6,
412 kCipAny,
413 EncodeCipEthernetLinkInterfaceControl,
414 DecodeCipEthernetLinkInterfaceControl,
415 &g_ethernet_link[idx].interface_control,
416 IFACE_CTRL_ACCESS_MODE);
417 }
418#else
419 InsertAttribute(ethernet_link_instance,
420 6,
421 kCipAny,
422 EncodeCipEthernetLinkInterfaceControl,
423 NULL,
424 &s_interface_control,
425 kGetableAll);
426#endif
427 InsertAttribute(ethernet_link_instance,
428 7,
429 kCipUsint,
430 EncodeCipUsint,
431 NULL,
432 &g_ethernet_link[idx].interface_type,
433 kGetableSingleAndAll);
434 InsertAttribute(ethernet_link_instance,
435 8,
436 kCipUsint,
437 EncodeCipUsint,
438 NULL,
439 &s_interface_state[idx],
440 kGetableAllDummy);
441 InsertAttribute(ethernet_link_instance,
442 9,
443 kCipUsint,
444 EncodeCipUsint,
445 NULL,
446 &dummy_attribute_usint, kGetableAllDummy);
447 InsertAttribute(ethernet_link_instance,
448 10,
449 kCipShortString,
450 EncodeCipShortString,
451 NULL,
452 &g_ethernet_link[idx].interface_label,
453 IFACE_LABEL_ACCESS_MODE);
454 InsertAttribute(ethernet_link_instance,
455 11,
456 kCipAny,
457 EncodeCipEthernetLinkInterfaceCaps,
458 NULL,
459 &g_ethernet_link[idx].interface_caps,
460 kGetableSingleAndAll);
461 }
462 } else {
463 return kEipStatusError;
464 }
465
466 return kEipStatusOk;
467}

536#define NELEMENTS(x) ( (sizeof(x) / sizeof(x[0]) ) )
537static void EncodeCipEthernetLinkInterfaceCaps(const void *const data,
538 ENIPMessage *const outgoing_message)
539{
540 const CipEthernetLinkMetaInterfaceCapability *const interface_caps = data;
541 EncodeCipDword(&interface_caps->capability_bits, outgoing_message);
542 uint16_t selected = interface_caps->speed_duplex_selector;
543 CipUsint count = 0;
544 while(selected) { /* count # of bits set */
545 selected &= selected - 1U; /* clear the least significant bit set */
546 count++;
547 }
548 EncodeCipUsint(&count, outgoing_message);
549
550 for (size_t i = 0; i < NELEMENTS(speed_duplex_table); i++) {
551 if (interface_caps->speed_duplex_selector &
552 (1U << i) ) {
553 EncodeCipUint(&speed_duplex_table[i].interface_speed, outgoing_message);
554 EncodeCipUsint(&speed_duplex_table[i].interface_duplex_mode,
555 outgoing_message);
556 }
557 }
558}

606#if defined(OPENER_ETHLINK_IFACE_CTRL_ENABLE) && \
607 0 != OPENER_ETHLINK_IFACE_CTRL_ENABLE
608static bool IsIfaceControlAllowed(CipInstanceNum instance_id,
609 CipEthernetLinkInterfaceControl const *iface_cntrl)
610{
611 const CipUsint duplex_mode =
612 (iface_cntrl->control_bits & kEthLinkIfCntrlForceDuplexFD) ? 1 : 0;
613 for (size_t i = 0; i < NELEMENTS(speed_duplex_table); i++) {
614 if (g_ethernet_link[instance_id - 1].interface_caps.speed_duplex_selector &
615 (1U << i) ) {
616 if (duplex_mode == speed_duplex_table[i].interface_duplex_mode &&
617 iface_cntrl->forced_interface_speed ==
618 speed_duplex_table[i].interface_speed) {
619 return true;
620 }
621 }
622 }
623 return false;
624}

626int DecodeCipEthernetLinkInterfaceControl(
627 CipEthernetLinkInterfaceControl *const data,
628 CipMessageRouterRequest *const message_router_request,
629 CipMessageRouterResponse *const message_router_response) {
630
631 CipInstance *const instance = GetCipInstance(
632 GetCipClass(message_router_request->request_path.class_id),
633 message_router_request->request_path.instance_number);
634
635 int number_of_decoded_bytes = -1;
636
637 CipEthernetLinkInterfaceControl if_cntrl;
638
639 DecodeCipWord(&if_cntrl.control_bits, message_router_request,
640 message_router_response);
641 DecodeCipUint(&if_cntrl.forced_interface_speed, message_router_request,
642 message_router_response);
643
644 if (if_cntrl.control_bits > kEthLinkIfCntrlMaxValid) {
645 message_router_response->general_status =
646 kCipErrorInvalidAttributeValue;
647 return number_of_decoded_bytes;
648
649 } else {
650 if ((0 != (if_cntrl.control_bits & kEthLinkIfCntrlAutonegotiate))
651 && ((0 != (if_cntrl.control_bits & kEthLinkIfCntrlForceDuplexFD))
652 || (0 != if_cntrl.forced_interface_speed))) {
653 message_router_response->general_status =
654 kCipErrorObjectStateConflict;
655 return number_of_decoded_bytes;
656 } else {
657 if (0 == (if_cntrl.control_bits & kEthLinkIfCntrlAutonegotiate)) {
658 /* Need to check if a supported mode is forced. */
659 if (!IsIfaceControlAllowed(instance->instance_number,
660 &if_cntrl)) {
661 message_router_response->general_status =
662 kCipErrorInvalidAttributeValue;
663 return number_of_decoded_bytes;
664 }
665 }
666 *data = if_cntrl; //write data to attribute
667 message_router_response->general_status = kCipErrorSuccess;
668 number_of_decoded_bytes = 4;
669 }
670 }
671 return number_of_decoded_bytes;
672}
