38// LEDC configuration for PWM output
39#define LEDC_TIMER LEDC_TIMER_0
40#define LEDC_MODE LEDC_LOW_SPEED_MODE
41#define LEDC_CHANNEL LEDC_CHANNEL_0
42#define LEDC_DUTY_RES LEDC_TIMER_13_BIT // 8192 steps (13-bit resolution)
43#define LEDC_FREQUENCY 1000 // 1kHz PWM frequency
44
45// Current limits
46#define CURRENT_MIN_MA 4.0f
47#define CURRENT_MAX_MA 20.0f
48#define CURRENT_RANGE_MA 16.0f // 20.0 - 4.0
49
50// Voltage ranges
51#define VOLTAGE_MAX_V 2.5f // Maximum input to XTR111
52#define ESP32_VOLTAGE_MAX_V 3.3f // ESP32 GPIO voltage

55static int s_pwm_gpio = -1;
56static float s_current_ma = CURRENT_MIN_MA;

58bool current_loop_4_20ma_init(int pwm_gpio)
59{
60 if (pwm_gpio < 0 || pwm_gpio > 54) {
61 ESP_LOGE(TAG, "Invalid GPIO pin: %d", pwm_gpio);
62 return false;
63 }
64
65 s_pwm_gpio = pwm_gpio;
66
67 // Configure LEDC timer
68 ledc_timer_config_t ledc_timer = {
69 .speed_mode = LEDC_MODE,
70 .duty_resolution = LEDC_DUTY_RES,
71 .timer_num = LEDC_TIMER,
72 .freq_hz = LEDC_FREQUENCY,
73 .clk_cfg = LEDC_AUTO_CLK
74 };
75 esp_err_t err = ledc_timer_config(&ledc_timer);
76 if (err != ESP_OK) {
77 ESP_LOGE(TAG, "Failed to configure LEDC timer: %s", esp_err_to_name(err));
78 return false;
79 }
80
81 // Configure LEDC channel
82 ledc_channel_config_t ledc_channel = {
83 .speed_mode = LEDC_MODE,
84 .channel = LEDC_CHANNEL,
85 .timer_sel = LEDC_TIMER,
86 .intr_type = LEDC_INTR_DISABLE,
87 .gpio_num = pwm_gpio,
88 .duty = 0,
89 .hpoint = 0
90 };
91 err = ledc_channel_config(&ledc_channel);
92 if (err != ESP_OK) {
93 ESP_LOGE(TAG, "Failed to configure LEDC channel: %s", esp_err_to_name(err));
94 return false;
95 }
96
97 // Set initial output to 4mA (minimum)
98 current_loop_4_20ma_set(CURRENT_MIN_MA);
99
100 ESP_LOGI(TAG, "4-20mA current loop initialized on GPIO%d", pwm_gpio);
101 return true;
102}

104bool current_loop_4_20ma_set(float current_ma)
105{
106 if (s_pwm_gpio < 0) {
107 ESP_LOGE(TAG, "4-20mA not initialized");
108 return false;
109 }
110
111 // Clamp to valid range
112 if (current_ma < CURRENT_MIN_MA) {
113 ESP_LOGW(TAG, "Current %.2f mA below minimum, clamping to %.2f mA",
114 current_ma, CURRENT_MIN_MA);
115 current_ma = CURRENT_MIN_MA;
116 } else if (current_ma > CURRENT_MAX_MA) {
117 ESP_LOGW(TAG, "Current %.2f mA above maximum, clamping to %.2f mA",
118 current_ma, CURRENT_MAX_MA);
119 current_ma = CURRENT_MAX_MA;
120 }
121
122 s_current_ma = current_ma;
123
124 // Convert 4-20mA to voltage (0-2.5V for XTR111)
125 // Formula: V = (I - 4mA) / 16mA × 2.5V
126 float voltage = ((current_ma - CURRENT_MIN_MA) / CURRENT_RANGE_MA) * VOLTAGE_MAX_V;
127
128 // Convert voltage to PWM duty cycle
129 // ESP32-P4 PWM: 0-3.3V range, 13-bit resolution (0-8191)
130 // Note: If using voltage divider, adjust this calculation accordingly
131 uint32_t duty = (uint32_t)((voltage / ESP32_VOLTAGE_MAX_V) * ((1 << LEDC_DUTY_RES) - 1));
132
133 // Clamp duty to valid range
134 uint32_t max_duty = (1 << LEDC_DUTY_RES) - 1;
135 if (duty > max_duty) {
136 duty = max_duty;
137 }
138
139 // Set PWM duty
140 esp_err_t err = ledc_set_duty(LEDC_MODE, LEDC_CHANNEL, duty);
141 if (err != ESP_OK) {
142 ESP_LOGE(TAG, "Failed to set PWM duty: %s", esp_err_to_name(err));
143 return false;
144 }
145
146 err = ledc_update_duty(LEDC_MODE, LEDC_CHANNEL);
147 if (err != ESP_OK) {
148 ESP_LOGE(TAG, "Failed to update PWM duty: %s", esp_err_to_name(err));
149 return false;
150 }
151
152 ESP_LOGD(TAG, "Set 4-20mA output: %.2f mA (voltage: %.3fV, duty: %lu/%lu)",
153 current_ma, voltage, duty, max_duty);
154
155 return true;
156}

163void current_loop_4_20ma_update_from_assembly(uint8_t assembly_byte)
164{
165 // Map byte value (0-255) to current (4-20mA)
166 // Linear mapping: 0 -> 4mA, 255 -> 20mA
167 // Formula: Current (mA) = 4.0 + (byte / 255.0) × 16.0
168 float current_ma = CURRENT_MIN_MA + ((float)assembly_byte / 255.0f) * CURRENT_RANGE_MA;
169 current_loop_4_20ma_set(current_ma);
170}
