108static inline void MergeStatusAndExtStatus(void) {
109 CipWord status_flags = g_identity.status & (~kExtStatusMask);
110 CipWord ext_status = g_identity.ext_status & kExtStatusMask;
111
112 /* Any major fault will override the current extended status with kMajorFault.
113 See comment on Major Fault at Vol. 1, Table 5A-2.4. */
114 if(0 !=
115 (status_flags & (kMajorRecoverableFault | kMajorUnrecoverableFault) ) ) {
116 ext_status = kMajorFault;
117 }
118 g_identity.status = status_flags | ext_status;
119}

168EipStatus IdentityObjectPreResetCallback(
169 CipInstance *RESTRICT const instance,
170 const CipMessageRouterRequest *const message_router_request,
171 CipMessageRouterResponse *const message_router_response
172) {
173 (void) instance;
174
175 EipStatus eip_status = kEipStatusOkSend;
176
177 if(message_router_request->request_data_size > 1) {
178 message_router_response->general_status = kCipErrorTooMuchData;
179 } else {
180 CipOctet reset_type = 0; /* The default type if type parameter was omitted. */
181 if(message_router_request->request_data_size == 1) {
182 reset_type = message_router_request->data[0];
183 }
184 switch(reset_type) {
185 case 0: /* Reset type 0 -> Emulate power cycle */
186 if(kEipStatusError == ResetDevice() ) {
187 message_router_response->general_status = kCipErrorInvalidParameter;
188 }
189 break;
190
191 case 1: /* Reset type 1 -> Return to factory defaults & power cycle*/
192 if(kEipStatusError == ResetDeviceToInitialConfiguration() ) {
193 message_router_response->general_status = kCipErrorInvalidParameter;
194 }
195 break;
196
197 /* case 2: Not supported Reset type 2 ->
198 Return to factory defaults except communications parameters & power cycle*/
199
200 default:
201 message_router_response->general_status = kCipErrorInvalidParameter;
202 break;
203 }
204 }
205 return eip_status;
206 }

208void InitializeCipIdentity(CipClass *class) {
209
210 CipClass *meta_class = class->class_instance.cip_class;
211
212 InsertAttribute( (CipInstance *) class, 1, kCipUint, EncodeCipUint, NULL,
213 (void *) &class->revision, kGetableSingleAndAll ); /* revision */
214 InsertAttribute( (CipInstance *) class, 2, kCipUint, EncodeCipUint, NULL,
215 (void *) &class->number_of_instances, kGetableSingleAndAll ); /* largest instance number */
216 InsertAttribute( (CipInstance *) class, 3, kCipUint, EncodeCipUint, NULL,
217 (void *) &class->number_of_instances, kGetableSingle ); /* number of instances currently existing*/
218 InsertAttribute( (CipInstance *) class, 4, kCipUint, EncodeCipUint, NULL,
219 (void *) &kCipUintZero, kNotSetOrGetable ); /* optional attribute list - default = 0 */
220 InsertAttribute( (CipInstance *) class, 5, kCipUint, EncodeCipUint, NULL,
221 (void *) &kCipUintZero, kNotSetOrGetable ); /* optional service list - default = 0 */
222 InsertAttribute( (CipInstance *) class, 6, kCipUint, EncodeCipUint, NULL,
223 (void *) &meta_class->highest_attribute_number,
224 kGetableSingleAndAll ); /* max class attribute number*/
225 InsertAttribute( (CipInstance *) class, 7, kCipUint, EncodeCipUint, NULL,
226 (void *) &class->highest_attribute_number,
227 kGetableSingleAndAll ); /* max instance attribute number*/
228
229 InsertService(meta_class,
230 kGetAttributeAll,
231 &GetAttributeAll,
232 "GetAttributeAll"); /* bind instance services to the metaclass*/
233 InsertService(meta_class,
234 kGetAttributeSingle,
235 &GetAttributeSingle,
236 "GetAttributeSingle");
237
238 // add Callback function pointers
239 class->PreResetCallback = &IdentityObjectPreResetCallback;
240
241}

250EipStatus CipIdentityInit() {
251
252 CipClass *class = CreateCipClass(kCipIdentityClassCode, 0, /* # of non-default class attributes */
253 7, /* # highest class attribute number*/
254 2, /* # of class services*/
255 8, /* # of instance attributes*/
256 8, /* # highest instance attribute number*/
257 5, /* # of instance services*/
258 1, /* # of instances*/
259 "identity", /* # class name (for debug)*/
260 1, /* # class revision*/ //TODO: change revision to 2 - check
261 &InitializeCipIdentity); /* # function pointer for initialization*/
262
263 if(class == 0) {
264 return kEipStatusError;
265 }
266
267 if (g_identity.product_name.length == 0)
268 SetDeviceProductName(OPENER_DEVICE_NAME);
269
270 CipInstance *instance = GetCipInstance(class, 1);
271 InsertAttribute(instance, 1, kCipUint, EncodeCipUint,
272 NULL, &g_identity.vendor_id, kGetableSingleAndAll);
273 InsertAttribute(instance, 2, kCipUint, EncodeCipUint,
274 NULL, &g_identity.device_type, kGetableSingleAndAll);
275 InsertAttribute(instance, 3, kCipUint, EncodeCipUint,
276 NULL, &g_identity.product_code, kGetableSingleAndAll);
277 InsertAttribute(instance, 4, kCipUsintUsint, EncodeRevision,
278 NULL, &g_identity.revision, kGetableSingleAndAll);
279 InsertAttribute(instance, 5, kCipWord, EncodeCipWord,
280 NULL, &g_identity.status, kGetableSingleAndAll);
281 InsertAttribute(instance, 6, kCipUdint, EncodeCipUdint,
282 NULL, &g_identity.serial_number, kGetableSingleAndAll);
283 InsertAttribute(instance, 7, kCipShortString, EncodeCipShortString,
284 NULL, &g_identity.product_name, kGetableSingleAndAll);
285 InsertAttribute(instance, 8, kCipUsint, EncodeCipUsint,
286 NULL, &g_identity.state, kGetableSingleAndAll);
287
288 InsertService(class,
289 kGetAttributeSingle,
290 &GetAttributeSingle,
291 "GetAttributeSingle");
292 InsertService(class, kGetAttributeAll, &GetAttributeAll, "GetAttributeAll");
293 InsertService(class, kReset, &CipResetService, "Reset");
294 InsertService(class, kGetAttributeList, &GetAttributeList,
295 "GetAttributeList");
296 InsertService(class, kSetAttributeList, &SetAttributeList,
297 "SetAttributeList");
298
299 return kEipStatusOk;
300}
