98int32_t lsm6ds3_reset_set(lsm6ds3_ctx_t *ctx, lsm6ds3_property_t val)
99{
100 uint8_t reg;
101 if (lsm6ds3_read_reg(ctx, LSM6DS3_CTRL3_C, &reg, 1)) {
102 return -1;
103 }
104 reg = (reg & ~0x01) | (uint8_t)val;
105 return lsm6ds3_write_reg(ctx, LSM6DS3_CTRL3_C, &reg, 1);
106}

208int32_t lsm6ds3_status_reg_get(lsm6ds3_ctx_t *ctx, lsm6ds3_status_reg_t *val)
209{
210 uint8_t reg;
211 if (lsm6ds3_read_reg(ctx, LSM6DS3_STATUS_REG, &reg, 1)) {
212 return -1;
213 }
214 val->xlda = (reg >> 0) & 0x01;
215 val->gda = (reg >> 1) & 0x01;
216 val->tda = (reg >> 2) & 0x01;
217 val->xlda_gy = (reg >> 4) & 0x01;
218 val->gda_gy = (reg >> 5) & 0x01;
219 val->den_lh = (reg >> 7) & 0x01;
220 return 0;
221}
