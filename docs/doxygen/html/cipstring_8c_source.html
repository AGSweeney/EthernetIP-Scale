52CipStringN *SetCipStringNByData(CipStringN *const cip_string,
53 CipUint str_len,
54 CipUint size,
55 const CipOctet *const data) {
56 CipStringN *result = cip_string;
57
58 (void) ClearCipStringN(cip_string);
59
60 if(0 != str_len) {
61 /* No trailing '\0' character! */
62 cip_string->length = str_len;
63 cip_string->size = size;
64 cip_string->string = CipCalloc(cip_string->length,
65 cip_string->size * sizeof(CipOctet) );
66 if(NULL == cip_string->string) {
67 result = NULL;
68 cip_string->length = 0;
69 cip_string->size = 0;
70 } else {
71 memcpy(cip_string->string,
72 data,
73 cip_string->length * cip_string->size * sizeof(CipOctet) );
74 }
75 }
76 return result;
77}

79CipStringN *SetCipStringNByCstr(CipStringN *const cip_string,
80 const char *const string,

82 if(0 != strlen(string) % symbol_size) {
83 OPENER_TRACE_ERR("Not enough octets for %d width StringN\n", symbol_size);
84 return cip_string;

87 return SetCipStringNByData(cip_string,

89 symbol_size,
90 (const CipOctet *) string);

267/* Ensures buf is NUL terminated, provided initial validation is successful */
268int GetCstrFromCipShortString(CipShortString *const string, char *buf, size_t len) {
269 size_t num;
270 int rc = 0;
271
272 if (!string || !buf || len < 1)
273 return -1;
274
275 num = (size_t)string->length;
276 if (len <= num) {
277 rc = (int)(num - len - 1);
278 num = len - 1;
279 }
280 len = num;
281
282 memcpy(buf, string->string, num);
283 buf[len] = 0;
284
285 return rc;
286}
