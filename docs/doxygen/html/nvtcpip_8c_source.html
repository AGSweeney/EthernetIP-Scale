64static esp_err_t TcpipNvOpen(nvs_handle_t *handle, nvs_open_mode_t mode) {
65 esp_err_t err = nvs_open(TCPIP_NVS_NAMESPACE, mode, handle);
66 if (ESP_ERR_NVS_NOT_FOUND == err && mode == NVS_READWRITE) {
67 /* Namespace does not exist yet - create by opening in read/write */
68 err = nvs_open(TCPIP_NVS_NAMESPACE, NVS_READWRITE, handle);
69 }
70 if (ESP_OK != err) {
71 ESP_LOGE(kTag, "nvs_open failed (%s)", esp_err_to_name(err));
72 }

81EipStatus NvTcpipLoad(CipTcpIpObject *p_tcp_ip) {
82 nvs_handle_t handle;
83 esp_err_t err = TcpipNvOpen(&handle, NVS_READONLY);
84 if (ESP_ERR_NVS_NOT_FOUND == err) {
85 ESP_LOGI(kTag, "No stored TCP/IP configuration found, using defaults");
86 return kEipStatusError;
87 } else if (ESP_OK != err) {
88 return kEipStatusError;
89 }
90
91 uint8_t raw_blob[sizeof(TcpipNvBlob)] = {0};
92 size_t length = sizeof(raw_blob);
93 err = nvs_get_blob(handle, TCPIP_NVS_KEY, raw_blob, &length);
94 nvs_close(handle);
95 if (ESP_ERR_NVS_NOT_FOUND == err) {
96 ESP_LOGI(kTag, "No stored TCP/IP configuration found, using defaults");
97 return kEipStatusError;
98 }
99 if (ESP_OK != err) {
100 ESP_LOGE(kTag, "Failed to load TCP/IP configuration (%s)", esp_err_to_name(err));
101 return kEipStatusError;
102 }
103
104 const TcpipNvBlob *blob_v2 = NULL;
105 TcpipNvBlobV1 blob_v1 = {0};
106
107 if (length == sizeof(TcpipNvBlob)) {
108 blob_v2 = (const TcpipNvBlob *)raw_blob;
109 if (blob_v2->version != TCPIP_NV_VERSION) {
110 ESP_LOGW(kTag, "Stored TCP/IP configuration has unexpected version %" PRIu32,
111 blob_v2->version);
112 return kEipStatusError;
113 }
114 } else if (length == sizeof(TcpipNvBlobV1)) {
115 memcpy(&blob_v1, raw_blob, sizeof(blob_v1));
116 if (blob_v1.version != 1U) {
117 ESP_LOGW(kTag, "Stored TCP/IP configuration has incompatible format");
118 return kEipStatusError;
119 }
120 } else {
121 ESP_LOGW(kTag, "Stored TCP/IP configuration has incompatible format");
122 return kEipStatusError;
123 }
124
125 if (blob_v2 != NULL) {
126 p_tcp_ip->config_control = blob_v2->config_control;
127 p_tcp_ip->interface_configuration.ip_address = blob_v2->ip_address;
128 p_tcp_ip->interface_configuration.network_mask = blob_v2->network_mask;
129 p_tcp_ip->interface_configuration.gateway = blob_v2->gateway;
130 p_tcp_ip->interface_configuration.name_server = blob_v2->name_server;
131 p_tcp_ip->interface_configuration.name_server_2 = blob_v2->name_server2;
132 p_tcp_ip->select_acd = (blob_v2->select_acd != 0u);
133 } else {
134 p_tcp_ip->config_control = blob_v1.config_control;
135 p_tcp_ip->interface_configuration.ip_address = blob_v1.ip_address;
136 p_tcp_ip->interface_configuration.network_mask = blob_v1.network_mask;
137 p_tcp_ip->interface_configuration.gateway = blob_v1.gateway;
138 p_tcp_ip->interface_configuration.name_server = blob_v1.name_server;
139 p_tcp_ip->interface_configuration.name_server_2 = blob_v1.name_server2;
140 p_tcp_ip->select_acd = false;
141 }
142
143 ip4_addr_t nv_ip = { .addr = p_tcp_ip->interface_configuration.ip_address };
144 ip4_addr_t nv_mask = { .addr = p_tcp_ip->interface_configuration.network_mask };
145 ip4_addr_t nv_gw = { .addr = p_tcp_ip->interface_configuration.gateway };
146 ip4_addr_t nv_dns1 = { .addr = p_tcp_ip->interface_configuration.name_server };
147 ip4_addr_t nv_dns2 = { .addr = p_tcp_ip->interface_configuration.name_server_2 };
148
149 char ip_str[16];
150 char mask_str[16];
151 char gw_str[16];
152 char dns1_str[16];
153 char dns2_str[16];
154
155 const char *dns1_print = ip4_addr_isany_val(nv_dns1) ?
156 "none" :
157 ip4addr_ntoa_r(&nv_dns1, dns1_str, sizeof dns1_str);
158 const char *dns2_print = ip4_addr_isany_val(nv_dns2) ?
159 "none" :
160 ip4addr_ntoa_r(&nv_dns2, dns2_str, sizeof dns2_str);
161
162 ESP_LOGI(kTag,
163 "NV config ctrl=0x%08lx ip=%s mask=%s gw=%s dns=%s/%s",
164 (unsigned long)p_tcp_ip->config_control,
165 ip4addr_ntoa_r(&nv_ip, ip_str, sizeof ip_str),
166 ip4addr_ntoa_r(&nv_mask, mask_str, sizeof mask_str),
167 ip4addr_ntoa_r(&nv_gw, gw_str, sizeof gw_str),
168 dns1_print,
169 dns2_print);
170
171 ClearCipString(&p_tcp_ip->interface_configuration.domain_name);
172 uint16_t domain_length = blob_v2 ? blob_v2->domain_length : blob_v1.domain_length;
173 if (domain_length > TCPIP_DOMAIN_MAX_LEN) {
174 domain_length = TCPIP_DOMAIN_MAX_LEN;
175 }
176 if (domain_length > 0u) {
177 const uint8_t *domain_src = blob_v2 ? blob_v2->domain : blob_v1.domain;
178 if (NULL == SetCipStringByData(&p_tcp_ip->interface_configuration.domain_name,
179 domain_length,
180 domain_src)) {
181 ESP_LOGE(kTag, "Failed to restore domain name");
182 return kEipStatusError;
183 }
184 }
185
186 ClearCipString(&p_tcp_ip->hostname);
187 uint16_t hostname_length = blob_v2 ? blob_v2->hostname_length : blob_v1.hostname_length;
188 if (hostname_length > TCPIP_HOSTNAME_MAX_LEN) {
189 hostname_length = TCPIP_HOSTNAME_MAX_LEN;
190 }
191 if (hostname_length > 0u) {
192 const uint8_t *hostname_src = blob_v2 ? blob_v2->hostname : blob_v1.hostname;
193 if (NULL == SetCipStringByData(&p_tcp_ip->hostname,
194 hostname_length,
195 hostname_src)) {
196 ESP_LOGE(kTag, "Failed to restore host name");
197 return kEipStatusError;
198 }
199 }
200
201 /* Mark configuration as applied */
202 p_tcp_ip->status |= 0x01;
203 p_tcp_ip->status &= ~kTcpipStatusIfaceCfgPend;
204
205 if ((p_tcp_ip->config_control & kTcpipCfgCtrlMethodMask) == kTcpipCfgCtrlStaticIp) {
206 if (!CipTcpIpIsValidNetworkConfig(&p_tcp_ip->interface_configuration)) {
207 ESP_LOGW(kTag, "Stored static configuration invalid, switching to DHCP");
208 p_tcp_ip->config_control &= ~kTcpipCfgCtrlMethodMask;
209 p_tcp_ip->config_control |= kTcpipCfgCtrlDhcp;
210 p_tcp_ip->interface_configuration.ip_address = 0u;
211 p_tcp_ip->interface_configuration.network_mask = 0u;
212 p_tcp_ip->interface_configuration.gateway = 0u;
213 p_tcp_ip->interface_configuration.name_server = 0u;
214 p_tcp_ip->interface_configuration.name_server_2 = 0u;
215 p_tcp_ip->status &= ~(kTcpipStatusAcdStatus | kTcpipStatusAcdFault);
216 (void)NvTcpipStore(p_tcp_ip);
217 }
218 }
219
220 if (blob_v2 == NULL) {
221 /* Upgrade legacy blob to latest format */
222 (void)NvTcpipStore(p_tcp_ip);
223 }
224
225 ESP_LOGI(kTag, "Restored TCP/IP configuration (method=%s)",
226 ( (p_tcp_ip->config_control & kTcpipCfgCtrlMethodMask) == kTcpipCfgCtrlDhcp) ?
227 "DHCP" : "Static");
228
229 return kEipStatusOk;
230}

237EipStatus NvTcpipStore(const CipTcpIpObject *p_tcp_ip) {
238 nvs_handle_t handle;
239 esp_err_t err = TcpipNvOpen(&handle, NVS_READWRITE);
240 if (ESP_OK != err) {
241 return kEipStatusError;
242 }
243
244 TcpipNvBlob blob = {0};
245 blob.version = TCPIP_NV_VERSION;
246 blob.config_control = p_tcp_ip->config_control;
247 blob.ip_address = p_tcp_ip->interface_configuration.ip_address;
248 blob.network_mask = p_tcp_ip->interface_configuration.network_mask;
249 blob.gateway = p_tcp_ip->interface_configuration.gateway;
250 blob.name_server = p_tcp_ip->interface_configuration.name_server;
251 blob.name_server2 = p_tcp_ip->interface_configuration.name_server_2;
252
253 if (p_tcp_ip->interface_configuration.domain_name.length > TCPIP_DOMAIN_MAX_LEN) {
254 blob.domain_length = TCPIP_DOMAIN_MAX_LEN;
255 } else {
256 blob.domain_length = p_tcp_ip->interface_configuration.domain_name.length;
257 }
258 if (blob.domain_length > 0u && NULL != p_tcp_ip->interface_configuration.domain_name.string) {
259 memcpy(blob.domain,
260 p_tcp_ip->interface_configuration.domain_name.string,
261 blob.domain_length);
262 }
263
264 if (p_tcp_ip->hostname.length > TCPIP_HOSTNAME_MAX_LEN) {
265 blob.hostname_length = TCPIP_HOSTNAME_MAX_LEN;
266 } else {
267 blob.hostname_length = p_tcp_ip->hostname.length;
268 }
269 if (blob.hostname_length > 0u && NULL != p_tcp_ip->hostname.string) {
270 memcpy(blob.hostname, p_tcp_ip->hostname.string, blob.hostname_length);
271 }
272
273 blob.select_acd = p_tcp_ip->select_acd ? 1u : 0u;
274
275 err = nvs_set_blob(handle, TCPIP_NVS_KEY, &blob, sizeof(blob));
276 if (ESP_OK == err) {
277 err = nvs_commit(handle);
278 }
279 nvs_close(handle);
280
281 if (ESP_OK != err) {
282 ESP_LOGE(kTag, "Failed to store TCP/IP configuration (%s)", esp_err_to_name(err));
283 return kEipStatusError;
284 }
285
286 ESP_LOGI(kTag, "Stored TCP/IP configuration (method=%s)",
287 ((p_tcp_ip->config_control & kTcpipCfgCtrlMethodMask) == kTcpipCfgCtrlDhcp) ?
288 "DHCP" : "Static");
289
290 return kEipStatusOk;
291}
