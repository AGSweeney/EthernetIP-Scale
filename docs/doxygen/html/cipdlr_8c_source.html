116EipStatus CipDlrInit(void) {
117 CipClass *dlr_class = NULL;
118
119 dlr_class = CreateCipClass(kCipDlrClassCode,
120 0, /* # class attributes */
121 7, /* # highest class attribute number */
122 2, /* # class services */
123 11,/* # instance attributes */
124 12,/* # of highest instance attribute */
125 2, /* # instance services */
126 1, /* # instances */
127 "DLR", /* object class name */
128 DLR_CLASS_REVISION, /* # class revision */
129 NULL /* function pointer for initialization */
130 );
131
132 if (NULL == dlr_class) {
133 return kEipStatusError;
134 }
135
136 /* Add services to the class */
137 InsertService(dlr_class, kGetAttributeSingle,
138 GetAttributeSingle, "GetAttributeSingle");
139 InsertService(dlr_class, kGetAttributeAll,
140 GetAttributeAll, "GetAttributeAll");
141
142 /* Bind attributes to the instance */
143 CipInstance *dlr_instance = GetCipInstance(dlr_class, 1u);
144
145 InsertAttribute(dlr_instance,
146 1,
147 kCipUsint,
148 EncodeCipUsint,
149 NULL,
150 &g_dlr.network_topology,
151 kGetableSingleAndAll);
152 InsertAttribute(dlr_instance,
153 2,
154 kCipUsint,
155 EncodeCipUsint,
156 NULL,
157 &g_dlr.network_status,
158 kGetableSingleAndAll);
159 InsertAttribute(dlr_instance,
160 3,
161 kCipUsint,
162 EncodeCipUsint,
163 NULL,
164 (void *)&s_0xFF_default,
165 kGetableAll);
166 InsertAttribute(dlr_instance,
167 4,
168 kCipAny,
169 EncodeCipRingSupervisorConfig,
170 NULL,
171 (void *)&s_0x00000000_default,
172 kGetableAllDummy);
173 InsertAttribute(dlr_instance,
174 5,
175 kCipUint,
176 EncodeCipUint,
177 NULL,
178 (void *)&s_0x0000_default,
179 kGetableAll);
180 InsertAttribute(dlr_instance,
181 6,
182 kCipAny,
183 EncodeCipNodeAddress,
184 NULL,
185 (void *)&s_zero_node,
186 kGetableAll);
187 InsertAttribute(dlr_instance,
188 7,
189 kCipAny,
190 EncodeCipNodeAddress,
191 NULL,
192 (void *)&s_zero_node,
193 kGetableAll);
194 InsertAttribute(dlr_instance,
195 8,
196 kCipUint,
197 EncodeCipUint,
198 NULL,
199 (void *)&s_0xFFFF_default,
200 kGetableAll);
201 /* Attribute #9 is not implemented and also NOT part of the GetAttributesAll
202 * response. Therefore it is not added here! */
203 InsertAttribute(dlr_instance,
204 10,
205 kCipAny,
206 EncodeCipNodeAddress,
207 NULL,
208 &g_dlr.active_supervisor_address,
209 kGetableSingleAndAll);
210 InsertAttribute(dlr_instance,
211 11,
212 kCipUsint,
213 EncodeCipUsint,
214 NULL,
215 (void *)&s_0x00_default,
216 kGetableAll);
217 InsertAttribute(dlr_instance,
218 12,
219 kCipDword,
220 EncodeCipDword,
221 NULL,
222 &g_dlr.capability_flags,
223 kGetableSingleAndAll);
224
225 /* Set attributes to initial values */
226 /* Assume beacon based DLR device. Also all Revision 3 and higher devices
227 * are required to support the Flush_Tables and Learning_Update frames
228 * (see Vol. 2 Section 5-6.2 Revision History of the DLR object).*/
229 g_dlr.capability_flags = (kDlrCapBeaconBased | kDlrCapFlushTableFrame);
230
231 return kEipStatusOk;
232}
