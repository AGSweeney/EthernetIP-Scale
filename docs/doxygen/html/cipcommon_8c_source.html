102EipStatus NotifyClass(const CipClass *RESTRICT const cip_class,
103 CipMessageRouterRequest *const message_router_request,
104 CipMessageRouterResponse *const message_router_response,
105 const struct sockaddr *originator_address,
106 const CipSessionHandle encapsulation_session) {
107
108 /* find the instance: if instNr==0, the class is addressed, else find the instance */
109 CipInstanceNum instance_number =
110 message_router_request->request_path.instance_number; /* get the instance number */
111 CipInstance *instance = GetCipInstance(cip_class, instance_number); /* look up the instance (note that if inst==0 this will be the class itself) */
112 if(instance) /* if instance is found */
113 {
114 OPENER_TRACE_INFO("notify: found instance %d%s\n",
115 instance_number,
116 instance_number == 0 ? " (class object)" : "");
117
118 CipServiceStruct *service = instance->cip_class->services; /* get pointer to array of services */
119 if(NULL != service) /* if services are defined */
120 {
121 for(size_t i = 0; i < instance->cip_class->number_of_services; i++) /* seach the services list */
122 {
123 if(message_router_request->service == service->service_number) /* if match is found */
124 {
125 /* call the service, and return what it returns */
126 OPENER_TRACE_INFO("notify: calling %s service\n", service->name);
127 OPENER_ASSERT(NULL != service->service_function);
128 return service->service_function(instance,
129 message_router_request,
130 message_router_response,
131 originator_address,
132 encapsulation_session);
133 } else {
134 service++;
135 }
136 }
137 } OPENER_TRACE_WARN(
138 "notify: service 0x%x not supported\n", message_router_request->service);
139 message_router_response->general_status = kCipErrorServiceNotSupported; /* if no services or service not found, return an error reply*/
140 } else {
141 OPENER_TRACE_WARN("notify: instance number %d unknown\n", instance_number);
142 /* if instance not found, return an error reply */
143 message_router_response->general_status = kCipErrorPathDestinationUnknown;
144 /* according to the test tool this is the correct error flag instead of CIP_ERROR_OBJECT_DOES_NOT_EXIST */
145 }
146
147 /* handle error replies*/
148 message_router_response->size_of_additional_status = 0; /* fill in the rest of the reply with not much of anything*/
149 InitializeENIPMessage(&message_router_response->message);
150 message_router_response->reply_service =
151 (0x80 | message_router_request->service); /* except the reply code is an echo of the command + the reply flag */
152
153 return kEipStatusOkSend;
154}

156CipUint GetMaxInstanceNumber(CipClass *RESTRICT const cip_class) {
157 CipUint max_instance = 0;
158 CipInstance *instance = cip_class->instances;
159 while (NULL != instance) { /* loop trough all instances of class */
160 if(instance->instance_number > max_instance) {
161 max_instance = instance->instance_number;
162 }
163 instance = instance->next;
164 }
165 return max_instance;
166}

168CipInstance *AddCipInstances(CipClass *RESTRICT const cip_class,
169 const CipInstanceNum number_of_instances) {
170 CipInstance **next_instance = NULL;
171 CipInstance *first_instance = NULL; /* Initialize to error result */
172 CipInstanceNum instance_number = 1; /* the first instance is number 1 */
173 int new_instances = 0;
174
175 OPENER_TRACE_INFO("adding %d instances to class %s\n",
176 number_of_instances,
177 cip_class->class_name);
178
179 /* Allocate and initialize all needed instances one by one. */
180 for(new_instances = 0; new_instances < number_of_instances; new_instances++) {
181
182 /* Find next free instance number */
183 CipBool found_free_number = false;
184
185 while (!found_free_number) {
186 next_instance = &cip_class->instances; /* set pointer to head of existing instances chain */
187
188 found_free_number = true; /* anticipate instance_number is not in use*/
189
190 /* loop through existing instances */
191 while (*next_instance) /* as long as what next_instance points to is not zero */
192 {
193 /* check if instance number in use */
194 if(instance_number == (*next_instance)->instance_number) {
195 found_free_number = false; /* instance number exists already */
196 break;
197 }
198 next_instance = &(*next_instance)->next; /* get next instance in instances chain*/
199 }
200
201 if(!found_free_number) {
202 instance_number++; /* try next instance_number and loop again through existing instances */
203 }
204
205 }
206
207 CipInstance *current_instance =
208 (CipInstance *) CipCalloc( 1, sizeof(CipInstance) );
209 OPENER_ASSERT(NULL != current_instance); /* fail if run out of memory */
210 if(NULL == current_instance) {
211 break;
212 }
213 if(NULL == first_instance) {
214 first_instance = current_instance; /* remember the first allocated instance */
215 }
216
217 current_instance->instance_number = instance_number; /* assign the next sequential instance number */
218 current_instance->cip_class = cip_class; /* point each instance to its class */
219
220 if(cip_class->number_of_attributes) /* if the class calls for instance attributes */
221 { /* then allocate storage for the attribute array */
222 current_instance->attributes = (CipAttributeStruct *) CipCalloc(
223 cip_class->number_of_attributes,
224 sizeof(CipAttributeStruct) );
225 OPENER_ASSERT(NULL != current_instance->attributes);/* fail if run out of memory */
226 if(NULL == current_instance->attributes) {
227 break;
228 }
229 }
230
231 *next_instance = current_instance; /* link the previous pointer to this new node */
232 next_instance = &current_instance->next; /* update pp to point to the next link of the current node */
233 cip_class->number_of_instances += 1; /* update the total number of instances recorded by the class */
234 instance_number++; /* update to the number of the next node*/
235 }
236
237 cip_class->max_instance = GetMaxInstanceNumber(cip_class); /* update largest instance number (class Attribute 2) */
238
239 if(new_instances != number_of_instances) {
240 /* TODO: Free again all attributes and instances allocated so far in this call. */
241 OPENER_TRACE_ERR(
242 "ERROR: Allocated only %d instances of requested %d for class %s\n",
243 new_instances,
244 number_of_instances,
245 cip_class->class_name);
246 first_instance = NULL; /* failed to allocate all instances / attributes */
247 }
248 return first_instance;
249}

251CipInstance *AddCipInstance(CipClass *RESTRICT const cip_class,
252 const CipInstanceNum instance_id) {
253 CipInstance *instance = GetCipInstance(cip_class, instance_id);
254
255 if(NULL == instance) { /*we have no instance with given id*/
256 instance = AddCipInstances(cip_class, 1);
257 instance->instance_number = instance_id;
258 }
259
260 cip_class->max_instance = GetMaxInstanceNumber(cip_class); /* update largest instance number (class Attribute 2) */
261
262 return instance;
263}

265CipClass *CreateCipClass(const CipUdint class_code,
266 const int number_of_class_attributes,
267 const EipUint32 highest_class_attribute_number,
268 const int number_of_class_services,
269 const int number_of_instance_attributes,
270 const EipUint32 highest_instance_attribute_number,
271 const int number_of_instance_services,
272 const CipInstanceNum number_of_instances,
273 const char *const name,
274 const EipUint16 revision,
275 InitializeCipClass initializer) {
276
277 OPENER_TRACE_INFO("creating class '%s' with code: 0x%" PRIX32 "\n", name,
278 class_code);
279
280 OPENER_ASSERT( NULL == GetCipClass(class_code) ); /* check if an class with the ClassID already exists */
281 /* should never try to redefine a class*/
282
283 /* a metaClass is a class that holds the class attributes and services
284 CIP can talk to an instance, therefore an instance has a pointer to its class
285 CIP can talk to a class, therefore a class struct is a subclass of the instance struct,
286 and contains a pointer to a metaclass
287 CIP never explicitly addresses a metaclass*/
288
289 CipClass *const cip_class = (CipClass *) CipCalloc( 1, sizeof(CipClass) ); /* create the class object*/
290 CipClass *const meta_class = (CipClass *) CipCalloc( 1, sizeof(CipClass) ); /* create the metaclass object*/
291
292 /* initialize the class-specific fields of the Class struct*/
293 cip_class->class_code = class_code; /* the class remembers the class ID */
294 cip_class->revision = revision; /* the class remembers the class ID */
295 cip_class->max_instance = 0; /* the largest instance number of a created object in this class */
296 cip_class->number_of_instances = 0; /* the number of instances initially zero (more created below) */
297 cip_class->instances = 0;
298 cip_class->number_of_attributes = number_of_instance_attributes; /* the class remembers the number of instances of that class */
299 cip_class->highest_attribute_number = highest_instance_attribute_number; /* indicate which attributes are included in instance getAttributeAll */
300 cip_class->number_of_services = number_of_instance_services; /* the class manages the behavior of the instances */
301 cip_class->services = 0;
302
303 /* Allocate and initialize the class name string. */
304 OPENER_ASSERT(NULL != name);
305 const size_t name_len = strlen(name); /* Length does not include termination byte. */
306 OPENER_ASSERT(0 < name_len); /* Cannot be an empty string. */
307 cip_class->class_name = CipCalloc(name_len + 1, 1); /* Allocate length plus termination byte. */
308 OPENER_ASSERT(NULL != cip_class->class_name);
309
310 /*
311 * memcpy is used here, instead of a strcpy variant, to avoid Visual Studio
312 * depreciation warnings arising from strcpy use, and the recommended
313 * alternatives, e.g. strcpy_s, are not available on all systems. In this
314 * case the size of the source string is already known, so memcpy() is
315 * suitable.
316 */
317 memcpy(cip_class->class_name, name, name_len + 1);
318
319 /* initialize the class-specific fields of the metaClass struct */
320 meta_class->class_code = 0xffffffff; /* set metaclass ID (this should never be referenced) */
321 meta_class->number_of_instances = 1; /* the class object is the only instance of the metaclass */
322 meta_class->instances = (CipInstance *) cip_class;
323 meta_class->number_of_attributes = number_of_class_attributes + 7; /* the metaclass remembers how many class attributes exist*/
324 meta_class->highest_attribute_number = highest_class_attribute_number; /* indicate which attributes are included in class getAttributeAll*/
325 meta_class->number_of_services = number_of_class_services; /* the metaclass manages the behavior of the class itself */
326 meta_class->class_name = (char *) CipCalloc(1, strlen(name) + 6); /* fabricate the name "meta<classname>"*/
327 snprintf(meta_class->class_name, strlen(name) + 6, "meta-%s", name);
328
329 /* initialize the instance-specific fields of the Class struct*/
330 cip_class->class_instance.instance_number = 0; /* the class object is instance zero of the class it describes (weird, but that's the spec)*/
331 cip_class->class_instance.attributes = 0; /* this will later point to the class attibutes*/
332 cip_class->class_instance.cip_class = meta_class; /* the class's class is the metaclass (like SmallTalk)*/
333 cip_class->class_instance.next = 0; /* the next link will always be zero, since there is only one instance of any particular class object */
334
335 meta_class->class_instance.instance_number = 0xffff; /*the metaclass object does not really have a valid instance number*/
336 meta_class->class_instance.attributes = NULL;/* the metaclass has no attributes*/
337 meta_class->class_instance.cip_class = NULL; /* the metaclass has no class*/
338 meta_class->class_instance.next = NULL; /* the next link will always be zero, since there is only one instance of any particular metaclass object*/
339
340 /* further initialization of the class object*/
341
342 cip_class->class_instance.attributes = (CipAttributeStruct *) CipCalloc(
343 meta_class->number_of_attributes,
344 sizeof(CipAttributeStruct) );
345 /* TODO -- check that we didn't run out of memory?*/
346
347 meta_class->services = (CipServiceStruct *) CipCalloc(
348 meta_class->number_of_services,
349 sizeof(CipServiceStruct) );
350
351 cip_class->services = (CipServiceStruct *) CipCalloc(
352 cip_class->number_of_services,
353 sizeof(CipServiceStruct) );
354
355 if(number_of_instances > 0) {
356 AddCipInstances(cip_class, number_of_instances); /*TODO handle return value and clean up if necessary*/
357 }
358
359 if(RegisterCipClass(cip_class) == kEipStatusError) {/* no memory to register class in Message Router */
360 return 0; /*TODO handle return value and clean up if necessary*/
361 }
362
363 AllocateAttributeMasks(meta_class); /* Allocation of bitmasks for Class Attributes */
364 AllocateAttributeMasks(cip_class); /* Allocation of bitmasks for Instance Attributes */
365
366 if(NULL == initializer) {
367 InsertAttribute( (CipInstance *) cip_class, 1, kCipUint, EncodeCipUint,
368 NULL, (void *) &cip_class->revision,
369 kGetableSingleAndAll ); /* revision */
370 InsertAttribute( (CipInstance *) cip_class, 2, kCipUint, EncodeCipUint,
371 NULL, (void *) &cip_class->max_instance,
372 kGetableSingleAndAll ); /* #2 Max instance no. */
373 InsertAttribute( (CipInstance *) cip_class, 3, kCipUint, EncodeCipUint,
374 NULL, (void *) &cip_class->number_of_instances,
375 kGetableSingleAndAll ); /* number of instances currently existing*/
376 InsertAttribute( (CipInstance *) cip_class, 4, kCipUint, EncodeCipUint,
377 NULL, (void *) &kCipUintZero, kGetableAllDummy ); /* optional attribute list - default = 0 */
378 InsertAttribute( (CipInstance *) cip_class, 5, kCipUint, EncodeCipUint,
379 NULL, (void *) &kCipUintZero, kNotSetOrGetable ); /* optional service list - default = 0 */
380 InsertAttribute( (CipInstance *) cip_class, 6, kCipUint, EncodeCipUint,
381 NULL, (void *) &meta_class->highest_attribute_number,
382 kGetableSingle ); /* max class attribute number*/
383 InsertAttribute( (CipInstance *) cip_class, 7, kCipUint, EncodeCipUint,
384 NULL, (void *) &cip_class->highest_attribute_number,
385 kGetableSingle ); /* max instance attribute number*/
386 if(number_of_class_services > 0) {
387 if(number_of_class_services > 1) { /*only if the mask has values add the get_attribute_all service */
388 InsertService(meta_class,
389 kGetAttributeAll,
390 &GetAttributeAll,
391 "GetAttributeAll"); /* bind instance services to the metaclass*/
392 }
393 InsertService(meta_class,
394 kGetAttributeSingle,
395 &GetAttributeSingle,
396 "GetAttributeSingle");
397 }
398 } else {
399 initializer(cip_class);
400 }
401
402 /* create the standard class services*/
403 return cip_class;
404}

406void InsertAttribute(CipInstance *const instance,
407 const EipUint16 attribute_number,
408 const EipUint8 cip_type,
409 CipAttributeEncodeInMessage encode_function,
410 CipAttributeDecodeFromMessage decode_function,
411 void *const data,
412 const EipByte cip_flags) {
413
414 OPENER_ASSERT(NULL != data); /* Its not allowed to push a NULL pointer, as this marks an unused attribute struct */
415
416 CipAttributeStruct *attribute = instance->attributes;
417 CipClass *cip_class = instance->cip_class;
418
419 OPENER_ASSERT(NULL != attribute);
420 /* adding a attribute to a class that was not declared to have any attributes is not allowed */
421 for(int i = 0; i < instance->cip_class->number_of_attributes; i++) {
422 if(attribute->data == NULL) { /* found non set attribute */
423 attribute->attribute_number = attribute_number;
424 attribute->type = cip_type;
425 attribute->encode = encode_function;
426 attribute->decode = decode_function;
427 attribute->attribute_flags = cip_flags;
428 attribute->data = data;
429
430 OPENER_ASSERT(attribute_number <= cip_class->highest_attribute_number);
431
432 size_t index = CalculateIndex(attribute_number);
433
434 cip_class->get_single_bit_mask[index] |=
435 (cip_flags & kGetableSingle) ? 1 << (attribute_number) % 8 : 0;
436 cip_class->get_all_bit_mask[index] |=
437 ( cip_flags & (kGetableAll | kGetableAllDummy) ) ? 1 <<
438 (attribute_number) % 8 : 0;
439 cip_class->set_bit_mask[index] |= ( (cip_flags & kSetable) ? 1 : 0 ) <<
440 ( (attribute_number) % 8 );
441
442 return;
443 }
444 attribute++;
445 } OPENER_TRACE_ERR(
446 "Tried to insert too many attributes into class: %" PRIu32 " '%s', instance %" PRIu32 "\n",
447 cip_class->class_code,
448 cip_class->class_name,
449 (EipUint32)instance->instance_number);
450 OPENER_ASSERT(false);
451 /* trying to insert too many attributes*/
452}

454void InsertService(const CipClass *const cip_class,
455 const EipUint8 service_number,
456 const CipServiceFunction service_function,
457 char *const service_name) {
458
459 CipServiceStruct *service = cip_class->services; /* get a pointer to the service array*/
460 OPENER_TRACE_INFO("%s, number of services:%d, service number:%d\n",
461 cip_class->class_name, cip_class->number_of_services,
462 service_number);
463 OPENER_ASSERT(service != NULL);
464 /* adding a service to a class that was not declared to have services is not allowed*/
465 for(int i = 0; i < cip_class->number_of_services; i++) /* Iterate over all service slots attached to the class */
466 {
467 if(service->service_number == service_number ||
468 service->service_function == NULL) /* found undefined service slot*/
469 {
470 service->service_number = service_number; /* fill in service number*/
471 service->service_function = service_function; /* fill in function address*/
472 service->name = service_name;
473 return;
474 }
475 ++service;
476 }
477 OPENER_ASSERT(false);
478 /* adding more services than were declared is a no-no*/
479}

528EipStatus GetAttributeSingle(CipInstance *RESTRICT const instance,
529 CipMessageRouterRequest *const message_router_request,
530 CipMessageRouterResponse *const message_router_response,
531 const struct sockaddr *originator_address,
532 const CipSessionHandle encapsulation_session) {
533 /* Suppress unused parameter compiler warning. */
534 (void) originator_address;
535 (void) encapsulation_session;
536
537 /* Mask for filtering get-ability */
538
539 CipAttributeStruct *attribute = GetCipAttribute(instance,
540 message_router_request->request_path.attribute_number);
541
542 GenerateGetAttributeSingleHeader(message_router_request,
543 message_router_response);
544
545 EipUint16 attribute_number =
546 message_router_request->request_path.attribute_number;
547
548 if( (NULL != attribute) && (NULL != attribute->data) ) {
549 uint8_t get_bit_mask =
550 (instance->cip_class->get_single_bit_mask[CalculateIndex(attribute_number)
551 ]);
552 if( 0 != ( get_bit_mask & ( 1 << (attribute_number % 8) ) ) ) {
553 OPENER_TRACE_INFO("getAttribute %d\n",
554 message_router_request->request_path.attribute_number); /* create a reply message containing the data*/
555
556 /* Call the PreGetCallback if enabled for this attribute and the class provides one. */
557 if( (attribute->attribute_flags & kPreGetFunc) &&
558 NULL != instance->cip_class->PreGetCallback ) {
559 instance->cip_class->PreGetCallback(instance,
560 attribute,
561 message_router_request->service);
562 }
563
564 OPENER_ASSERT(NULL != attribute);
565 attribute->encode(attribute->data, &message_router_response->message);
566 message_router_response->general_status = kCipErrorSuccess;
567
568 /* Call the PostGetCallback if enabled for this attribute and the class provides one. */
569 if( (attribute->attribute_flags & kPostGetFunc) &&
570 NULL != instance->cip_class->PostGetCallback ) {
571 instance->cip_class->PostGetCallback(instance,
572 attribute,
573 message_router_request->service);
574 }
575 }
576 }
577
578 return kEipStatusOkSend;
579}

782EipStatus SetAttributeSingle(CipInstance *RESTRICT const instance,
783 CipMessageRouterRequest *const message_router_request,
784 CipMessageRouterResponse *const message_router_response,
785 const struct sockaddr *originator_address,
786 const CipSessionHandle encapsulation_session) {
787 /* Suppress unused parameter compiler warning. */
788 (void)originator_address;
789 (void)encapsulation_session;
790
791 CipAttributeStruct *attribute = GetCipAttribute(instance,
792 message_router_request->request_path.attribute_number);
793
794 GenerateSetAttributeSingleHeader(message_router_request,
795 message_router_response);
796
797 EipUint16 attribute_number =
798 message_router_request->request_path.attribute_number;
799
800 /* Mask for filtering set-ability */
801 if( (NULL != attribute) && (NULL != attribute->data) ) {
802
803 if( (attribute->attribute_flags == kGetableAllDummy) ||
804 (attribute->attribute_flags == kNotSetOrGetable) ||
805 (attribute->attribute_flags == kGetableAll) ) {
806 OPENER_TRACE_WARN("SetAttributeSingle: Attribute %d not supported!\n\r",
807 attribute_number);
808 } else {
809 uint8_t set_bit_mask =
810 (instance->cip_class->set_bit_mask[CalculateIndex(attribute_number)]);
811 if( 0 != ( set_bit_mask & ( 1 << (attribute_number % 8) ) ) ) {
812 OPENER_TRACE_INFO("setAttribute %d\n", attribute_number);
813
814 /* Call the PreSetCallback if enabled for this attribute and the class provides one. */
815 if( (attribute->attribute_flags & kPreSetFunc) &&
816 NULL != instance->cip_class->PreSetCallback ) {
817 instance->cip_class->PreSetCallback(instance,
818 attribute,
819 message_router_request->service);
820 }
821
822 OPENER_ASSERT(NULL != attribute);
823
824 attribute->decode(attribute->data,
825 message_router_request,
826 message_router_response); //writes data to attribute, sets resonse status
827
828 /* Call the PostSetCallback if enabled for this attribute and the class provides one. */
829 if( ( attribute->attribute_flags & (kPostSetFunc | kNvDataFunc) ) &&
830 NULL != instance->cip_class->PostSetCallback ) {
831 instance->cip_class->PostSetCallback(instance,
832 attribute,
833 message_router_request->service);
834 }
835 } else {
836 message_router_response->general_status = kCipErrorAttributeNotSetable;
837 OPENER_TRACE_WARN("SetAttributeSingle: Attribute %d not setable!\n\r",
838 attribute_number);
839 }
840
841 }
842 }
843
844 return kEipStatusOkSend;
845}
