44// Custom vprintf that captures logs to buffer
45static int log_buffer_vprintf(const char *fmt, va_list args)
46{
47 // Capture to buffer first (before args is consumed)
48 if (s_enabled && s_log_buffer && s_buffer_mutex) {
49 if (xSemaphoreTake(s_buffer_mutex, pdMS_TO_TICKS(10)) == pdTRUE) {
50 // Format log message to temporary buffer
51 va_list args_copy;
52 va_copy(args_copy, args);
53 char temp_buffer[512];
54 int len = vsnprintf(temp_buffer, sizeof(temp_buffer), fmt, args_copy);
55 va_end(args_copy);
56
57 if (len > 0) {
58 size_t bytes_to_write = (len < (int)sizeof(temp_buffer)) ? len : sizeof(temp_buffer) - 1;
59
60 // Write to circular buffer
61 if (s_write_pos + bytes_to_write <= s_buffer_size) {
62 // Simple case: fits in remaining buffer
63 memcpy(&s_log_buffer[s_write_pos], temp_buffer, bytes_to_write);
64 s_write_pos += bytes_to_write;
65 } else {
66 // Wraps around
67 size_t first_part = s_buffer_size - s_write_pos;
68 memcpy(&s_log_buffer[s_write_pos], temp_buffer, first_part);
69 memcpy(s_log_buffer, &temp_buffer[first_part], bytes_to_write - first_part);
70 s_write_pos = bytes_to_write - first_part;
71 s_wrapped = true;
72 }
73
74 s_total_written += bytes_to_write;
75 }
76
77 xSemaphoreGive(s_buffer_mutex);
78 }
79 }
80
81 // Call original vprintf (to UART) - args is still valid here
82 int ret = 0;
83 if (s_original_vprintf) {
84 ret = s_original_vprintf(fmt, args);
85 }
86
87 return ret;
88}

90bool log_buffer_init(size_t buffer_size)
91{
92 if (s_enabled) {
93 ESP_LOGW(TAG, "Log buffer already initialized");
94 return true;
95 }
96
97 if (buffer_size == 0) {
98 buffer_size = DEFAULT_BUFFER_SIZE;
99 }
100
101 // Create mutex
102 s_buffer_mutex = xSemaphoreCreateMutex();
103 if (s_buffer_mutex == NULL) {
104 ESP_LOGE(TAG, "Failed to create log buffer mutex");
105 return false;
106 }
107
108 // Allocate buffer
109 s_log_buffer = (char *)malloc(buffer_size);
110 if (s_log_buffer == NULL) {
111 ESP_LOGE(TAG, "Failed to allocate log buffer (%zu bytes)", buffer_size);
112 vSemaphoreDelete(s_buffer_mutex);
113 s_buffer_mutex = NULL;
114 return false;
115 }
116
117 s_buffer_size = buffer_size;
118 s_write_pos = 0;
119 s_total_written = 0;
120 s_wrapped = false;
121 memset(s_log_buffer, 0, buffer_size);
122
123 // Install custom vprintf
124 s_original_vprintf = esp_log_set_vprintf(log_buffer_vprintf);
125
126 s_enabled = true;
127 ESP_LOGI(TAG, "Log buffer initialized (%zu bytes)", buffer_size);
128
129 return true;
130}

132size_t log_buffer_get(char *buffer, size_t buffer_size)
133{
134 if (!s_enabled || !s_log_buffer || !buffer || buffer_size == 0) {
135 return 0;
136 }
137
138 if (s_buffer_mutex == NULL) {
139 return 0;
140 }
141
142 if (xSemaphoreTake(s_buffer_mutex, portMAX_DELAY) != pdTRUE) {
143 return 0;
144 }
145
146 size_t bytes_to_copy = 0;
147
148 if (s_wrapped) {
149 // Buffer has wrapped - copy from write_pos to end, then from start to write_pos
150 size_t first_part = s_buffer_size - s_write_pos;
151 size_t second_part = s_write_pos;
152
153 if (buffer_size >= s_buffer_size) {
154 // Can copy entire buffer
155 memcpy(buffer, &s_log_buffer[s_write_pos], first_part);
156 memcpy(&buffer[first_part], s_log_buffer, second_part);
157 bytes_to_copy = s_buffer_size;
158 } else {
159 // Copy what fits, starting from oldest data
160 if (buffer_size <= first_part) {
161 memcpy(buffer, &s_log_buffer[s_write_pos], buffer_size);
162 bytes_to_copy = buffer_size;
163 } else {
164 memcpy(buffer, &s_log_buffer[s_write_pos], first_part);
165 size_t remaining = buffer_size - first_part;
166 if (remaining <= second_part) {
167 memcpy(&buffer[first_part], s_log_buffer, remaining);
168 bytes_to_copy = buffer_size;
169 } else {
170 memcpy(&buffer[first_part], s_log_buffer, second_part);
171 bytes_to_copy = first_part + second_part;
172 }
173 }
174 }
175 } else {
176 // Buffer hasn't wrapped - simple copy
177 bytes_to_copy = (s_write_pos < buffer_size) ? s_write_pos : buffer_size;
178 memcpy(buffer, s_log_buffer, bytes_to_copy);
179 }
180
181 // Ensure null termination
182 if (bytes_to_copy < buffer_size) {
183 buffer[bytes_to_copy] = '\0';
184 } else {
185 buffer[buffer_size - 1] = '\0';
186 bytes_to_copy = buffer_size - 1;
187 }
188
189 xSemaphoreGive(s_buffer_mutex);
190
191 return bytes_to_copy;
192}

194size_t log_buffer_get_size(void)
195{
196 if (!s_enabled || !s_log_buffer) {
197 return 0;
198 }
199
200 if (s_buffer_mutex == NULL) {
201 return 0;
202 }
203
204 if (xSemaphoreTake(s_buffer_mutex, portMAX_DELAY) != pdTRUE) {
205 return 0;
206 }
207
208 size_t size = s_wrapped ? s_buffer_size : s_write_pos;
209
210 xSemaphoreGive(s_buffer_mutex);
211
212 return size;
213}

215void log_buffer_clear(void)
216{
217 if (!s_enabled || !s_log_buffer || !s_buffer_mutex) {
218 return;
219 }
220
221 if (xSemaphoreTake(s_buffer_mutex, portMAX_DELAY) == pdTRUE) {
222 s_write_pos = 0;
223 s_total_written = 0;
224 s_wrapped = false;
225 memset(s_log_buffer, 0, s_buffer_size);
226 xSemaphoreGive(s_buffer_mutex);
227 ESP_LOGI(TAG, "Log buffer cleared");
228 }
229}
