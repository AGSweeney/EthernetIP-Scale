99/* @brief Initializes session list and interface information. */
100void EncapsulationInit(void) {
101
102 DetermineEndianess();
103
104 /*initialize random numbers for random delayed response message generation
105 * we use the ip address as seed as suggested in the spec */
106 srand(g_tcpip.interface_configuration.ip_address);
107
108 /* initialize Sessions to invalid == free session */
109 for(size_t i = 0; i < OPENER_NUMBER_OF_SUPPORTED_SESSIONS; i++) {
110 g_registered_sessions[i] = kEipInvalidSocket;
111 }
112
113 for(size_t i = 0; i < ENCAP_NUMBER_OF_SUPPORTED_DELAYED_ENCAP_MESSAGES; i++) {
114 g_delayed_encapsulation_messages[i].socket = kEipInvalidSocket;
115 }
116
117 /*TODO make the service information configurable*/

119 g_service_information.type_code = kCipItemIdListServiceResponse;
120 g_service_information.length = sizeof(g_service_information);
121 g_service_information.encapsulation_protocol_version = 1;
122 g_service_information.capability_flags = kCapabilityFlagsCipTcp | kCapabilityFlagsCipUdpClass0or1;
123 snprintf((char*) g_service_information.name_of_service, sizeof(g_service_information.name_of_service), "Communications");
124}

126EipStatus HandleReceivedExplictTcpData(int socket, EipUint8 *buffer, size_t length, int *number_of_remaining_bytes, struct sockaddr *originator_address,
127 ENIPMessage *const outgoing_message) {
128 OPENER_TRACE_INFO("Handles data for TCP socket: %d\n", socket);
129 EipStatus return_value = kEipStatusOk;
130 EncapsulationData encapsulation_data = { 0 };
131 /* eat the encapsulation header*/
132 /* the structure contains a pointer to the encapsulated data*/
133 /* returns how many bytes are left after the encapsulated data*/
134 const int remaining_bytes = CreateEncapsulationStructure(buffer, length, &encapsulation_data);
135
136 if(remaining_bytes >= 0) {
137 *number_of_remaining_bytes = remaining_bytes;
138 } else {
139 OPENER_TRACE_ERR("Fragmented packet detected! Fragmented packets are not supported!\n");
140 *number_of_remaining_bytes = 0;
141 return kEipStatusError;
142 }
143
144 if(kEncapsulationHeaderOptionsFlag == encapsulation_data.options) /*TODO generate appropriate error response*/
145 {
146 if(*number_of_remaining_bytes >= 0) /* check if the message is corrupt: header size + claimed payload size > than what we actually received*/
147 {
148 /* full package or more received */
149 encapsulation_data.status = kEncapsulationProtocolSuccess;
150 return_value = kEipStatusOkSend;
151 /* most of these functions need a reply to be send */
152 switch(encapsulation_data.command_code){
153 case (kEncapsulationCommandNoOperation):
154 OPENER_TRACE_INFO("NOP\n");
155 /* NOP needs no reply and does nothing */
156 return_value = kEipStatusOk;
157 break;
158
159 case (kEncapsulationCommandListServices):
160 OPENER_TRACE_INFO("List services\n");
161 HandleReceivedListServicesCommand(&encapsulation_data, outgoing_message);
162 break;
163
164 case (kEncapsulationCommandListIdentity):
165 OPENER_TRACE_INFO("List identity\n");
166 HandleReceivedListIdentityCommandTcp(&encapsulation_data, outgoing_message);
167 break;
168
169 case (kEncapsulationCommandListInterfaces):
170 OPENER_TRACE_INFO("List interfaces\n");
171 HandleReceivedListInterfacesCommand(&encapsulation_data, outgoing_message);
172 break;
173
174 case (kEncapsulationCommandRegisterSession):
175 OPENER_TRACE_INFO("Register session\n");
176 HandleReceivedRegisterSessionCommand(socket, &encapsulation_data, outgoing_message);
177 break;
178
179 case (kEncapsulationCommandUnregisterSession):
180 OPENER_TRACE_INFO("unregister session\n");
181 return_value = HandleReceivedUnregisterSessionCommand(&encapsulation_data, outgoing_message);
182 break;
183
184 case (kEncapsulationCommandSendRequestReplyData):
185 OPENER_TRACE_INFO("Send Request/Reply Data\n");
186 return_value = HandleReceivedSendRequestResponseDataCommand(&encapsulation_data, originator_address, outgoing_message);
187 break;
188
189 case (kEncapsulationCommandSendUnitData):
190 OPENER_TRACE_INFO("Send Unit Data\n");
191 return_value = HandleReceivedSendUnitDataCommand(&encapsulation_data, originator_address, outgoing_message);
192 break;
193
194 default:
195 return_value = HandleReceivedInvalidCommand(&encapsulation_data, outgoing_message);
196 break;
197 }
198 }
199 }
200
201 return return_value;
202}

277void GenerateEncapsulationHeader(const EncapsulationData *const receive_data, const size_t command_specific_data_length, const CipSessionHandle session_handle,
278 const EncapsulationProtocolErrorCode encapsulation_protocol_status, ENIPMessage *const outgoing_message) {
279 AddIntToMessage(receive_data->command_code, outgoing_message);
280 AddIntToMessage(command_specific_data_length, outgoing_message);
281 AddDintToMessage(session_handle, outgoing_message); //Session handle
282 AddDintToMessage(encapsulation_protocol_status, outgoing_message); //Status
283 memcpy(outgoing_message->current_message_position, receive_data->sender_context, kSenderContextSize); // sender context
284 outgoing_message->current_message_position += kSenderContextSize;
285 outgoing_message->used_message_length += kSenderContextSize;
286 AddDintToMessage(0, outgoing_message); // options
287}

330void HandleReceivedListIdentityCommandUdp(const int socket,
331 const struct sockaddr_in *const from_address,
332 const EncapsulationData *const receive_data)
333{
334 DelayedEncapsulationMessage *delayed_message_buffer = NULL;
335 ENIPMessage *p_outgoing_message = NULL;
336
337 for(size_t i = 0; i < ENCAP_NUMBER_OF_SUPPORTED_DELAYED_ENCAP_MESSAGES; i++) {
338 if(kEipInvalidSocket == g_delayed_encapsulation_messages[i].socket) {
339 delayed_message_buffer = &(g_delayed_encapsulation_messages[i]);
340
341 p_outgoing_message = &(delayed_message_buffer->outgoing_message);
342 InitializeENIPMessage(p_outgoing_message);
343 break;
344 }
345 }
346
347 if(NULL != delayed_message_buffer) {
348 delayed_message_buffer->socket = socket;
349 memcpy((&delayed_message_buffer->receiver), from_address, sizeof(struct sockaddr_in));
350
351 DetermineDelayTime(receive_data->communication_buffer_start, delayed_message_buffer);
352
353 EncapsulateListIdentityResponseMessage(receive_data, p_outgoing_message);
354 }
355}

363 /* Item ID*/
364 const CipUint kItemIDCipIdentity = 0x0C;
365 AddIntToMessage(kItemIDCipIdentity, outgoing_message);
366
367 AddIntToMessage(ListIdentityGetCipIdentityItemLength(), outgoing_message);

377 AddIntToMessage(g_identity.device_type, outgoing_message);

381 AddIntToMessage(g_identity.status, outgoing_message);
382 AddDintToMessage(g_identity.serial_number, outgoing_message);

387 outgoing_message->used_message_length += g_identity.product_name.length;

389 AddSintToMessage(g_identity.state, outgoing_message);

405void DetermineDelayTime(const EipByte *buffer_start, DelayedEncapsulationMessage *const delayed_message_buffer) {
406
407 buffer_start += 12; /* start of the sender context */
408 EipUint16 maximum_delay_time = GetUintFromMessage((const EipUint8** const ) &buffer_start);
409
410 if(0 == maximum_delay_time) {
411 maximum_delay_time = kListIdentityDefaultDelayTime;
412 } else if(kListIdentityMinimumDelayTime > maximum_delay_time) { /* if maximum_delay_time is between 1 and 500ms set it to 500ms */
413 maximum_delay_time = kListIdentityMinimumDelayTime;
414 }
415
416 delayed_message_buffer->time_out = rand() % maximum_delay_time;
417}

431/* @brief Check supported protocol, generate session handle, send replay back to originator.
432 * @param socket Socket this request is associated to. Needed for double register check
433 * @param receive_data Pointer to received data with request/response.
434 */
435void HandleReceivedRegisterSessionCommand(int socket, const EncapsulationData *const receive_data, ENIPMessage *const outgoing_message) {
436 int session_index = 0;
437 CipSessionHandle session_handle = 0;
438 EncapsulationProtocolErrorCode encapsulation_protocol_status = kEncapsulationProtocolSuccess;
439
440 EipUint16 protocol_version = GetUintFromMessage((const EipUint8** const ) &receive_data->current_communication_buffer_position);
441 EipUint16 option_flag = GetUintFromMessage((const EipUint8** const ) &receive_data->current_communication_buffer_position);
442
443 /* check if requested protocol version is supported and the register session option flag is zero*/
444 if((0 < protocol_version) && (protocol_version <= kSupportedProtocolVersion) && (0 == option_flag)) { /*Option field should be zero*/
445 /* check if the socket has already a session open */
446 for(unsigned int i = 0; i < OPENER_NUMBER_OF_SUPPORTED_SESSIONS; ++i) {
447 if(g_registered_sessions[i] == socket) {
448 /* the socket has already registered a session this is not allowed*/
449 OPENER_TRACE_INFO(
450 "Error: A session is already registered at socket %d\n",
451 socket);
452 session_handle = i + 1; /*return the already assigned session back, the cip spec is not clear about this needs to be tested*/
453 encapsulation_protocol_status = kEncapsulationProtocolInvalidCommand;
454 session_index = kSessionStatusInvalid;
455 break;
456 }
457 }
458
459 if(kSessionStatusInvalid != session_index) {
460 session_index = GetFreeSessionIndex();
461 if(kSessionStatusInvalid == session_index) /* no more sessions available */
462 {
463 encapsulation_protocol_status = kEncapsulationProtocolInsufficientMemory;
464 } else { /* successful session registered */
465 SocketTimer *socket_timer = SocketTimerArrayGetEmptySocketTimer(g_timestamps,
466 OPENER_NUMBER_OF_SUPPORTED_SESSIONS);
467 SocketTimerSetSocket(socket_timer, socket);
468 SocketTimerSetLastUpdate(socket_timer, g_actual_time);
469 g_registered_sessions[session_index] = socket; /* store associated socket */
470 session_handle = (CipSessionHandle)(session_index + 1);
471 encapsulation_protocol_status = kEncapsulationProtocolSuccess;
472 }
473 }
474 } else { /* protocol not supported */
475 encapsulation_protocol_status = kEncapsulationProtocolUnsupportedProtocol;
476 }
477
478 EncapsulateRegisterSessionCommandResponseMessage(receive_data, session_handle, encapsulation_protocol_status, outgoing_message);
479
480}

489EipStatus HandleReceivedUnregisterSessionCommand(const EncapsulationData *const receive_data, ENIPMessage *const outgoing_message) {
490 OPENER_TRACE_INFO("encap.c: Unregister Session Command\n");
491 if((0 < receive_data->session_handle) && (receive_data->session_handle <=
492 OPENER_NUMBER_OF_SUPPORTED_SESSIONS)) {
493 CipSessionHandle i = receive_data->session_handle - 1;
494 if(kEipInvalidSocket != g_registered_sessions[i]) {
495 CloseTcpSocket(g_registered_sessions[i]);
496 g_registered_sessions[i] = kEipInvalidSocket;
497 CloseClass3ConnectionBasedOnSession(i + 1);
498 return kEipStatusOk;
499 }
500 }
501
502 /* no such session registered */
503 GenerateEncapsulationHeader(receive_data, 0, receive_data->session_handle, kEncapsulationProtocolInvalidSessionHandle, outgoing_message);
504 return kEipStatusOkSend;
505}

512EipStatus HandleReceivedSendUnitDataCommand(const EncapsulationData *const receive_data, const struct sockaddr *const originator_address,

514 EipStatus return_value = kEipStatusOkSend;
515 /*EipStatus*/return_value = kEipStatusOk; /* TODO: Shouldn't this be kEipStatusOk cause we must not send any response if data_length < 6? */

518 /* Command specific data UDINT .. Interface Handle, UINT .. Timeout, CPF packets */
519 /* don't use the data yet */
520 GetDintFromMessage((const EipUint8** const ) &receive_data->current_communication_buffer_position); /* skip over null interface handle*/
521 GetIntFromMessage((const EipUint8** const ) &receive_data->current_communication_buffer_position); /* skip over unused timeout value*/

528 InitializeENIPMessage(outgoing_message);
529 GenerateEncapsulationHeader(receive_data, 0, receive_data->session_handle, kEncapsulationProtocolInvalidSessionHandle, outgoing_message);
530 return_value = kEipStatusOkSend; /* TODO: Needs to be here if line with first TODO of this function is adjusted. */
531 }
532 }
533 return return_value;

598int_fast32_t CreateEncapsulationStructure(const EipUint8 *receive_buffer,
599 size_t receive_buffer_length,
600 EncapsulationData *const encapsulation_data)
601{
602 encapsulation_data->communication_buffer_start = (EipUint8*) receive_buffer;
603 encapsulation_data->command_code = GetUintFromMessage(&receive_buffer);
604 encapsulation_data->data_length = GetUintFromMessage(&receive_buffer);
605 encapsulation_data->session_handle = GetUdintFromMessage(&receive_buffer);
606 encapsulation_data->status = GetUdintFromMessage(&receive_buffer);
607
