71int DecodeCipQoSAttribute(void *const data,
72 CipMessageRouterRequest *const message_router_request,
73 CipMessageRouterResponse *const message_router_response) {
74
75 const EipUint8 **const cip_message = &(message_router_request->data);
76
77 int number_of_decoded_bytes = -1;
78
79 if (NULL != cip_message) {
80
81 CipUsint attribute_value_received = GetUsintFromMessage(cip_message);
82 if (attribute_value_received < 64U) {
83
84 *(CipUsint *)data = attribute_value_received; //write value to attribute
85
86 message_router_response->general_status = kCipErrorSuccess;
87 number_of_decoded_bytes = 1;
88
89 } else {
90 message_router_response->general_status =
91 kCipErrorInvalidAttributeValue;
92 }
93 } else {
94
95 message_router_response->general_status = kCipErrorNotEnoughData;
96 OPENER_TRACE_INFO("CIP QoS not enough data\n");
97 }
98
99 return number_of_decoded_bytes;
100
101}

103CipUsint CipQosGetDscpPriority(ConnectionObjectPriority priority) {
104
105 CipUsint priority_value;
106 switch (priority) {
107 case kConnectionObjectPriorityLow:
108 priority_value = s_active_dscp.low;
109 break;
110 case kConnectionObjectPriorityHigh:
111 priority_value = s_active_dscp.high;
112 break;
113 case kConnectionObjectPriorityScheduled:
114 priority_value = s_active_dscp.scheduled;
115 break;
116 case kConnectionObjectPriorityUrgent:
117 priority_value = s_active_dscp.urgent;
118 break;
119 case kConnectionObjectPriorityExplicit: /* fall through */
120 default:
121 priority_value = s_active_dscp.explicit_msg;
122 break;
123 }
124 return priority_value;
125}

127EipStatus CipQoSInit() {
128
129 CipClass *qos_class = NULL;
130
131 if( ( qos_class = CreateCipClass(kCipQoSClassCode,
132 7, /* # class attributes */
133 7, /* # highest class attribute number */
134 2, /* # class services */
135 8, /* # instance attributes */
136 8, /* # highest instance attribute number */
137 2, /* # instance services */
138 1, /* # instances */
139 "Quality of Service",
140 1, /* # class revision */
141 NULL /* # function pointer for initialization */
142 ) ) == 0 ) {
143
144 return kEipStatusError;
145 }
146
147 CipInstance *instance = GetCipInstance(qos_class, 1); /* bind attributes to the instance #1 that was created above */
148
149 InsertAttribute(instance,
150 1,
151 kCipUsint,
152 EncodeCipUsint,
153 NULL,
154 (void *) &g_qos.q_frames_enable,
155 kNotSetOrGetable);
156 InsertAttribute(instance,
157 2,
158 kCipUsint,
159 EncodeCipUsint,
160 NULL,
161 (void *) &g_qos.dscp.event,
162 kNotSetOrGetable);
163 InsertAttribute(instance,
164 3,
165 kCipUsint,
166 EncodeCipUsint,
167 NULL,
168 (void *) &g_qos.dscp.general,
169 kNotSetOrGetable);
170 InsertAttribute(instance,
171 4,
172 kCipUsint,
173 EncodeCipUsint,
174 DecodeCipQoSAttribute,
175 (void *) &g_qos.dscp.urgent,
176 kGetableSingle | kSetable | kNvDataFunc);
177 InsertAttribute(instance,
178 5,
179 kCipUsint,
180 EncodeCipUsint,
181 DecodeCipQoSAttribute,
182 (void *) &g_qos.dscp.scheduled,
183 kGetableSingle | kSetable | kNvDataFunc);
184 InsertAttribute(instance,
185 6,
186 kCipUsint,
187 EncodeCipUsint,
188 DecodeCipQoSAttribute,
189 (void *) &g_qos.dscp.high,
190 kGetableSingle | kSetable | kNvDataFunc);
191 InsertAttribute(instance,
192 7,
193 kCipUsint,
194 EncodeCipUsint,
195 DecodeCipQoSAttribute,
196 (void *) &g_qos.dscp.low,
197 kGetableSingle | kSetable | kNvDataFunc);
198 InsertAttribute(instance,
199 8,
200 kCipUsint,
201 EncodeCipUsint,
202 DecodeCipQoSAttribute,
203 (void *) &g_qos.dscp.explicit_msg,
204 kGetableSingle | kSetable | kNvDataFunc);
205
206 InsertService(qos_class, kGetAttributeSingle, &GetAttributeSingle,
207 "GetAttributeSingle");
208 InsertService(qos_class, kSetAttributeSingle, &SetAttributeSingle,
209 "SetAttributeSingle");
210
211 return kEipStatusOk;
212}
