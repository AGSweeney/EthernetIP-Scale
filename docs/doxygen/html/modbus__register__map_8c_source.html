47static uint16_t bytes_to_big_endian_uint16(const uint8_t *bytes)
48{
49 return (bytes[0] << 8) | bytes[1];
50}

52static void uint16_to_big_endian_bytes(uint16_t value, uint8_t *bytes)
53{
54 bytes[0] = (value >> 8) & 0xFF;
55 bytes[1] = value & 0xFF;
56}

58static uint16_t big_endian_to_little_endian(uint16_t be_value)
59{
60 return ((be_value & 0xFF) << 8) | ((be_value >> 8) & 0xFF);
61}

68bool modbus_read_input_registers(uint16_t start_addr, uint16_t quantity, uint8_t *data)
69{
70 // Check bounds: start_addr must be within range and quantity must not overflow
71 if (start_addr > INPUT_REG_END || quantity == 0 || start_addr + quantity > INPUT_REG_END + 1) {
72 ESP_LOGE(TAG, "Invalid input register range: %d-%d", start_addr, start_addr + quantity - 1);
73 return false;
74 }
75
76 SemaphoreHandle_t assembly_mutex = get_assembly_mutex();
77 if (assembly_mutex == NULL) {
78 ESP_LOGE(TAG, "Assembly mutex not initialized");
79 return false;
80 }
81
82 xSemaphoreTake(assembly_mutex, portMAX_DELAY);
83
84 // Map Input Assembly 100 (32 bytes = 16 registers) to Modbus Input Registers 0-15
85 // Assembly data is stored as little-endian bytes [low_byte, high_byte]
86 // Modbus requires big-endian bytes [high_byte, low_byte]
87 for (uint16_t i = 0; i < quantity; i++) {
88 uint16_t reg_addr = start_addr + i;
89 uint16_t byte_offset = reg_addr * 2;
90
91 if (byte_offset + 1 < sizeof(g_assembly_data064)) {
92 // Read little-endian from assembly: [low_byte, high_byte]
93 uint8_t low_byte = g_assembly_data064[byte_offset];
94 uint8_t high_byte = g_assembly_data064[byte_offset + 1];
95
96 // Write as big-endian for Modbus: [high_byte, low_byte]
97 data[i * 2] = high_byte;
98 data[i * 2 + 1] = low_byte;
99
100 } else {
101 // Out of range, return zero
102 data[i * 2] = 0;
103 data[i * 2 + 1] = 0;
104 }
105 }
106
107 xSemaphoreGive(assembly_mutex);
108 return true;
109}

111bool modbus_read_holding_registers(uint16_t start_addr, uint16_t quantity, uint8_t *data)
112{
113 // Check if address is in output assembly range (100-115)
114 if (start_addr >= HOLDING_REG_OUTPUT_START && start_addr + quantity <= HOLDING_REG_OUTPUT_END + 1) {
115 SemaphoreHandle_t assembly_mutex = get_assembly_mutex();
116 if (assembly_mutex == NULL) {
117 return false;
118 }
119
120 xSemaphoreTake(assembly_mutex, portMAX_DELAY);
121
122 // Map Output Assembly 150 (32 bytes = 16 registers) to Modbus Holding Registers 100-115
123 // Assembly data is stored as little-endian bytes [low_byte, high_byte]
124 // Modbus requires big-endian bytes [high_byte, low_byte]
125 uint16_t reg_offset = start_addr - HOLDING_REG_OUTPUT_START;
126 for (uint16_t i = 0; i < quantity; i++) {
127 uint16_t byte_offset = (reg_offset + i) * 2;
128
129 if (byte_offset + 1 < sizeof(g_assembly_data096)) {
130 // Read little-endian from assembly: [low_byte, high_byte]
131 uint8_t low_byte = g_assembly_data096[byte_offset];
132 uint8_t high_byte = g_assembly_data096[byte_offset + 1];
133
134 // Write as big-endian for Modbus: [high_byte, low_byte]
135 data[i * 2] = high_byte;
136 data[i * 2 + 1] = low_byte;
137
138 } else {
139 // Out of range, return zero
140 data[i * 2] = 0;
141 data[i * 2 + 1] = 0;
142 }
143 }
144
145 xSemaphoreGive(assembly_mutex);
146 return true;
147 }
148
149 // Check if address is in config assembly range (150-154)
150 if (start_addr >= HOLDING_REG_CONFIG_START && start_addr + quantity <= HOLDING_REG_CONFIG_END + 1) {
151 SemaphoreHandle_t assembly_mutex = get_assembly_mutex();
152 if (assembly_mutex == NULL) {
153 return false;
154 }
155
156 xSemaphoreTake(assembly_mutex, portMAX_DELAY);
157
158 // Map Config Assembly 151 (10 bytes = 5 registers) to Modbus Holding Registers 150-154
159 // Assembly data is stored as little-endian bytes [low_byte, high_byte]
160 // Modbus requires big-endian bytes [high_byte, low_byte]
161 uint16_t reg_offset = start_addr - HOLDING_REG_CONFIG_START;
162 for (uint16_t i = 0; i < quantity; i++) {
163 uint16_t byte_offset = (reg_offset + i) * 2;
164
165 if (byte_offset + 1 < sizeof(g_assembly_data097)) {
166 // Read little-endian from assembly: [low_byte, high_byte]
167 uint8_t low_byte = g_assembly_data097[byte_offset];
168 uint8_t high_byte = g_assembly_data097[byte_offset + 1];
169
170 // Write as big-endian for Modbus: [high_byte, low_byte]
171 data[i * 2] = high_byte;
172 data[i * 2 + 1] = low_byte;
173 } else {
174 // Out of range, return zero
175 data[i * 2] = 0;
176 data[i * 2 + 1] = 0;
177 }
178 }
179
180 xSemaphoreGive(assembly_mutex);
181 return true;
182 }
183
184 ESP_LOGE(TAG, "Invalid holding register range: %d-%d", start_addr, start_addr + quantity - 1);
185 return false;
186}

195bool modbus_write_holding_registers(uint16_t start_addr, uint16_t quantity, const uint8_t *data)
196{
197 // Check if address is in output assembly range (100-115)
198 if (start_addr >= HOLDING_REG_OUTPUT_START && start_addr + quantity <= HOLDING_REG_OUTPUT_END + 1) {
199 SemaphoreHandle_t assembly_mutex = get_assembly_mutex();
200 if (assembly_mutex == NULL) {
201 return false;
202 }
203
204 xSemaphoreTake(assembly_mutex, portMAX_DELAY);
205
206 // Map Modbus Holding Registers 100-115 to Output Assembly 150 (32 bytes = 16 registers)
207 uint16_t reg_offset = start_addr - HOLDING_REG_OUTPUT_START;
208 for (uint16_t i = 0; i < quantity; i++) {
209 uint16_t byte_offset = (reg_offset + i) * 2;
210
211 if (byte_offset + 1 < sizeof(g_assembly_data096)) {
212 // Convert big-endian from Modbus to little-endian for assembly
213 uint16_t be_value = bytes_to_big_endian_uint16(&data[i * 2]);
214 uint16_t le_value = big_endian_to_little_endian(be_value);
215 g_assembly_data096[byte_offset] = le_value & 0xFF;
216 g_assembly_data096[byte_offset + 1] = (le_value >> 8) & 0xFF;
217 }
218 }
219
220 xSemaphoreGive(assembly_mutex);
221
222 return true;
223 }
224
225 // Check if address is in config assembly range (150-154)
226 if (start_addr >= HOLDING_REG_CONFIG_START && start_addr + quantity <= HOLDING_REG_CONFIG_END + 1) {
227 SemaphoreHandle_t assembly_mutex = get_assembly_mutex();
228 if (assembly_mutex == NULL) {
229 return false;
230 }
231
232 xSemaphoreTake(assembly_mutex, portMAX_DELAY);
233
234 // Map Modbus Holding Registers 150-154 to Config Assembly 151 (10 bytes = 5 registers)
235 uint16_t reg_offset = start_addr - HOLDING_REG_CONFIG_START;
236 for (uint16_t i = 0; i < quantity; i++) {
237 uint16_t byte_offset = (reg_offset + i) * 2;
238
239 if (byte_offset + 1 < sizeof(g_assembly_data097)) {
240 uint16_t be_value = bytes_to_big_endian_uint16(&data[i * 2]);
241 uint16_t le_value = big_endian_to_little_endian(be_value);
242 g_assembly_data097[byte_offset] = le_value & 0xFF;
243 g_assembly_data097[byte_offset + 1] = (le_value >> 8) & 0xFF;
244 }
245 }
246
247 xSemaphoreGive(assembly_mutex);
248 return true;
249 }
250
251 ESP_LOGE(TAG, "Invalid holding register range for write: %d-%d", start_addr, start_addr + quantity - 1);
252 return false;
253}
