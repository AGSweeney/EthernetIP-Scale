46CipClass *CreateAssemblyClass(void) {
47 /* create the CIP Assembly object with zero instances */
48 CipClass *assembly_class = CreateCipClass(kCipAssemblyClassCode, 0, /* # class attributes*/
49 7, /* # highest class attribute number*/
50 1, /* # class services*/
51 2, /* # instance attributes*/
52 4, /* # highest instance attribute number*/
53 2, /* # instance services*/
54 0, /* # instances*/
55 "assembly", /* name */
56 2, /* Revision, according to the CIP spec currently this has to be 2 */
57 NULL); /* # function pointer for initialization*/
58 if(NULL != assembly_class) {
59 InsertService(assembly_class,
60 kGetAttributeSingle,
61 &GetAttributeSingle,
62 "GetAttributeSingle");

65 kSetAttributeSingle,
66 &SetAttributeSingle,
67 "SetAttributeSingle");
68
69 InsertGetSetCallback(assembly_class, AssemblyPreGetCallback, kPreGetFunc);
70 InsertGetSetCallback(assembly_class, AssemblyPostSetCallback, kPostSetFunc);
71 }

98CipInstance *CreateAssemblyObject(const CipInstanceNum instance_id,
99 EipByte *const data,
100 const EipUint16 data_length) {
101 CipClass *assembly_class = GetCipClass(kCipAssemblyClassCode);
102 if(NULL == assembly_class) {
103 assembly_class = CreateAssemblyClass();
104 }
105
106 if(NULL == assembly_class) {
107 return NULL;
108 }
109
110 CipInstance *const instance = AddCipInstance(assembly_class, instance_id); /* add instances (always succeeds (or asserts))*/
111
112 CipByteArray *const assembly_byte_array = (CipByteArray *) CipCalloc(1,
113 sizeof(
114 CipByteArray) );
115 if(assembly_byte_array == NULL) {
116 return NULL; /*TODO remove assembly instance in case of error*/
117 }
118
119 assembly_byte_array->length = data_length;
120 assembly_byte_array->data = data;
121
122 InsertAttribute(instance,
123 3,
124 kCipByteArray,
125 EncodeCipByteArray,
126 DecodeCipAssemblyAttribute3,
127 assembly_byte_array,
128 kSetAndGetAble | kPreGetFunc | kPostSetFunc);
129 /* Attribute 4 Number of bytes in Attribute 3 */
130
131 InsertAttribute(instance, 4, kCipUint, EncodeCipUint,
132 NULL, &(assembly_byte_array->length), kGetableSingle);
133
134 return instance;

155int DecodeCipAssemblyAttribute3(void *const data,
156 CipMessageRouterRequest *const message_router_request,
157 CipMessageRouterResponse *const message_router_response)
158{
159 CipInstance *const instance =
160 GetCipInstance(GetCipClass(
161 message_router_request->request_path.class_id),
162 message_router_request->request_path.instance_number);
163
164 int number_of_decoded_bytes = -1;
165 OPENER_TRACE_INFO(" -> set Assembly attribute byte array\r\n");
166 CipByteArray *cip_byte_array = (CipByteArray *)data;
167
168 if(message_router_request->request_data_size < cip_byte_array->length) {
169 OPENER_TRACE_INFO(
170 "DecodeCipByteArray: not enough data received.\n");
171 message_router_response->general_status = kCipErrorNotEnoughData;
172 return number_of_decoded_bytes;
173 }
174 if(message_router_request->request_data_size > cip_byte_array->length) {
175 OPENER_TRACE_INFO(
176 "DecodeCipByteArray: too much data received.\n");
177 message_router_response->general_status = kCipErrorTooMuchData;
178 return number_of_decoded_bytes;
179 }
180
181 // data-length is correct
182 memcpy(cip_byte_array->data,
183 message_router_request->data,
184 cip_byte_array->length);
185
186 if(AfterAssemblyDataReceived(instance) != kEipStatusOk) {
187 /* punt early without updating the status... though I don't know
188 * how much this helps us here, as the attribute's data has already
189 * been overwritten.
190 *
191 * however this is the task of the application side which will
192 * take the data. In addition we have to inform the sender that the
193 * data was not ok.
194 */
195 message_router_response->general_status = kCipErrorInvalidAttributeValue;
196 } else {
197 message_router_response->general_status = kCipErrorSuccess;
198 }
199
200 number_of_decoded_bytes = cip_byte_array->length;
201
202 return number_of_decoded_bytes;
203}
