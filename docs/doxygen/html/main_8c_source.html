234// Function to trigger LSM6DS3 calibration (called from API)
235// Returns ESP_OK on success, ESP_FAIL if sensor not initialized
236esp_err_t sample_application_calibrate_lsm6ds3(uint32_t samples, uint32_t sample_delay_ms)
237{
238 if (s_imu_state_mutex == NULL) {
239 imu_state_mutex_init();
240 }
241
242 // Check initialization status with mutex protection
243 bool is_initialized = false;
244 if (s_imu_state_mutex != NULL && xSemaphoreTake(s_imu_state_mutex, pdMS_TO_TICKS(100)) == pdTRUE) {
245 is_initialized = s_lsm6ds3_initialized;
246 xSemaphoreGive(s_imu_state_mutex);
247 } else {
248 // Fallback: check without mutex
249 is_initialized = s_lsm6ds3_initialized;
250 }
251
252 if (!is_initialized) {
253 return ESP_FAIL;
254 }
255
256 // Perform calibration with mutex protection
257 esp_err_t err = ESP_FAIL;
258 if (s_imu_state_mutex != NULL && xSemaphoreTake(s_imu_state_mutex, pdMS_TO_TICKS(1000)) == pdTRUE) {
259 if (s_lsm6ds3_initialized && s_i2c_bus_handle != NULL) {
260 err = lsm6ds3_calibrate_gyro(&s_lsm6ds3_handle, samples, sample_delay_ms);
261 if (err == ESP_OK) {
262 // Save calibration to NVS
263 esp_err_t save_err = lsm6ds3_save_calibration_to_nvs(&s_lsm6ds3_handle, "system");
264 if (save_err != ESP_OK) {
265 ESP_LOGW(TAG, "LSM6DS3: Calibration complete but failed to save to NVS: %s", esp_err_to_name(save_err));
266 }
267 }
268 }
269 xSemaphoreGive(s_imu_state_mutex);
270 }
271
272 return err;
273}

520static void tcpip_acd_conflict_callback(struct netif *netif, acd_callback_enum_t state) {
521 ESP_LOGI(TAG, "ACD callback received: state=%d (0=IP_OK, 1=RESTART_CLIENT, 2=DECLINE)", (int)state);
522 s_acd_last_state = state;
523 s_acd_callback_received = true; // Mark that callback was actually received
524 switch (state) {
525 case ACD_IP_OK:
526 g_tcpip.status &= ~(kTcpipStatusAcdStatus | kTcpipStatusAcdFault);
527 // ACD_IP_OK means probe phase completed successfully and IP is assigned.
528 // ACD now enters ONGOING state for periodic defense, so set activity = 1.
529 CipTcpIpSetLastAcdActivity(1);
530 // Resume LED blinking when IP is OK (no conflict)
531 user_led_start_flash();
532 ESP_LOGI(TAG, "ACD: IP OK - no conflict detected, entering ongoing defense phase");
533#if CONFIG_OPENER_ACD_RETRY_ENABLED
534 // Reset retry count on successful IP assignment
535 s_acd_retry_count = 0;
536 // Stop retry timer if running
537 if (s_acd_retry_timer != NULL) {
538 xTimerStop(s_acd_retry_timer, portMAX_DELAY);
539 }
540#endif
541 /* Legacy mode: Assign IP if it hasn't been assigned yet (callback fired after timeout) */
542 if (s_acd_probe_pending && netif != NULL) {
543 esp_netif_t *esp_netif = esp_netif_get_handle_from_netif_impl(netif);
544 if (esp_netif != NULL && s_pending_static_ip_cfg.ip.addr != 0) {
545 ESP_LOGI(TAG, "Legacy ACD: Assigning IP " IPSTR " after callback confirmation", IP2STR(&s_pending_static_ip_cfg.ip));
546 esp_netif_set_ip_info(esp_netif, &s_pending_static_ip_cfg);
547 opener_configure_dns(esp_netif);
548 s_acd_probe_pending = false;
549 }
550 }
551 break;
552 case ACD_DECLINE:
553 case ACD_RESTART_CLIENT:
554 g_tcpip.status |= kTcpipStatusAcdStatus;
555 g_tcpip.status |= kTcpipStatusAcdFault;
556 CipTcpIpSetLastAcdActivity(3);
557 // Stop LED blinking and turn solid on ACD conflict
558 user_led_stop_flash();
559 user_led_set(true); // Turn LED on solid
560 ESP_LOGW(TAG, "ACD: Conflict detected (state=%d) - LED set to solid", (int)state);
561#if CONFIG_OPENER_ACD_RETRY_ENABLED
562 // Retry logic: On conflict, remove IP and schedule retry after delay
563 if (netif != NULL) {
564 esp_netif_t *esp_netif = esp_netif_get_handle_from_netif_impl(netif);
565 if (esp_netif != NULL) {
566 // Check if we should retry
567 if (CONFIG_OPENER_ACD_RETRY_MAX_ATTEMPTS == 0 ||
568 s_acd_retry_count < CONFIG_OPENER_ACD_RETRY_MAX_ATTEMPTS) {
569 ESP_LOGW(TAG, "ACD: Scheduling retry (attempt %d/%d) after %dms",
570 s_acd_retry_count + 1,
571 CONFIG_OPENER_ACD_RETRY_MAX_ATTEMPTS == 0 ? 999 : CONFIG_OPENER_ACD_RETRY_MAX_ATTEMPTS,
572 CONFIG_OPENER_ACD_RETRY_DELAY_MS);
573 tcpip_acd_start_retry(esp_netif, netif);
574 } else {
575 ESP_LOGE(TAG, "ACD: Max retry attempts (%d) reached - giving up",
576 CONFIG_OPENER_ACD_RETRY_MAX_ATTEMPTS);
577 }
578 }
579 }
580#endif
581 break;
582 default:
583 g_tcpip.status |= kTcpipStatusAcdStatus;
584 g_tcpip.status |= kTcpipStatusAcdFault;
585 break;
586 }
587 if (s_acd_sem != NULL) {
588 xSemaphoreGive(s_acd_sem);
589 }
590}

600static void tcpip_acd_start_cb(void *arg) {
601 ESP_LOGI(TAG, "tcpip_acd_start_cb: CALLBACK EXECUTING - arg=%p", arg);
602 AcdStartContext *ctx = (AcdStartContext *)arg;
603 if (ctx == NULL) {
604 ESP_LOGE(TAG, "tcpip_acd_start_cb: NULL context");
605 // Signal semaphore even on error so caller doesn't hang
606 if (s_acd_registration_sem != NULL) {
607 xSemaphoreGive(s_acd_registration_sem);
608 }
609 return;
610 }
611 ESP_LOGI(TAG, "tcpip_acd_start_cb: Context valid - netif=%p, ip=" IPSTR,
612 ctx->netif, IP2STR(&ctx->ip));
613 ctx->err = ERR_OK;
614
615 // NULL check: netif may be invalidated between context creation and callback execution
616 if (ctx->netif == NULL) {
617 ESP_LOGD(TAG, "tcpip_acd_start_cb: NULL netif - ACD probe cancelled");
618 ctx->err = ERR_IF;
619 free(ctx);
620 return;
621 }
622
623 // If probe phase is complete, still register ACD for ongoing conflict detection
624 bool probe_was_pending = s_acd_probe_pending;
625
626 if (!s_acd_registered) {
627 ctx->netif->acd_list = NULL;
628 memset(&s_static_ip_acd, 0, sizeof(s_static_ip_acd));
629 err_t add_err = acd_add(ctx->netif, &s_static_ip_acd, tcpip_acd_conflict_callback);
630 if (add_err == ERR_OK) {
631 s_acd_registered = true;
632 ESP_LOGD(TAG, "tcpip_acd_start_cb: ACD client registered");
633 } else {
634 ESP_LOGE(TAG, "tcpip_acd_start_cb: acd_add() failed with err=%d", (int)add_err);
635 ctx->err = ERR_IF;
636 // Signal registration semaphore even on failure so caller doesn't hang
637 if (s_acd_registration_sem != NULL) {
638 xSemaphoreGive(s_acd_registration_sem);
639 }
640 free(ctx);
641 return;
642 }
643 }
644
645 // Signal registration semaphore to allow tcpip_perform_acd to wait for completion
646 if (s_acd_registration_sem != NULL) {
647 xSemaphoreGive(s_acd_registration_sem);
648 }
649
650 // If probe phase was skipped (IP already assigned), manually transition to ONGOING state
651 // Otherwise, ACD will naturally transition: PROBING -> ANNOUNCING -> ONGOING
652 if (!probe_was_pending) {
653 // IP already assigned - manually transition to ONGOING state for periodic defensive ARPs
654 acd_stop(&s_static_ip_acd); // Stop current state first
655 s_static_ip_acd.state = ACD_STATE_ONGOING;
656 s_static_ip_acd.ipaddr = ctx->ip;
657 s_static_ip_acd.sent_num = 0;
658 s_static_ip_acd.lastconflict = 0;
659 s_static_ip_acd.num_conflicts = 0;
660
661 // Re-add to netif's acd_list so timer processes it
662 acd_add(ctx->netif, &s_static_ip_acd, tcpip_acd_conflict_callback);
663
664 // Set activity = 1 (OngoingDetection) since we're entering ONGOING state
665 CipTcpIpSetLastAcdActivity(1);
666
667 // Set ttw to defense interval so timer counts down before first probe
668#ifdef CONFIG_OPENER_ACD_PERIODIC_DEFEND_INTERVAL_MS
669 if (CONFIG_OPENER_ACD_PERIODIC_DEFEND_INTERVAL_MS > 0) {
670 const uint16_t timer_interval_ms = 100;
671 s_static_ip_acd.ttw = (uint16_t)((CONFIG_OPENER_ACD_PERIODIC_DEFEND_INTERVAL_MS + timer_interval_ms - 1) / timer_interval_ms);
672 } else {
673 s_static_ip_acd.ttw = 0;
674 }
675#else
676 s_static_ip_acd.ttw = 100; // Default 10 seconds
677#endif
678 }
679 // If probe_was_pending, ACD is already running via acd_start() - don't stop it
680 // It will naturally transition: PROBING -> ANNOUNCING -> ONGOING
681 ctx->err = ERR_OK;
682
683 // Free heap-allocated context (allocated on heap to prevent stack corruption)
684 free(ctx);
685}

727 if (netif == NULL) {
728 ESP_LOGW(TAG, "ACD requested but no netif available");
729 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
730 CipTcpIpSetLastAcdActivity(3);
731 return false;
732 }
733
734 if (s_acd_sem == NULL) {
735 s_acd_sem = xSemaphoreCreateBinary();
736 if (s_acd_sem == NULL) {
737 ESP_LOGE(TAG, "Failed to create ACD semaphore");
738 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
739 CipTcpIpSetLastAcdActivity(3);
740 return false;
741 }
742 }
743
744 while (xSemaphoreTake(s_acd_sem, 0) == pdTRUE) {
745 /* flush any stale signals */
746 }
747
748 // Check if probe is still pending before creating context (prevents invalid context if cancelled)
749 if (!s_acd_probe_pending) {
750 ESP_LOGD(TAG, "tcpip_perform_acd: ACD probe no longer pending - skipping");
751 return true; // Return true to allow IP assignment (ACD was cancelled or completed)
752 }
753
754 // Initialize callback tracking: timeout without callback means probe sequence hasn't completed yet
755 // Only explicit callback (ACD_IP_OK, ACD_RESTART_CLIENT, or ACD_DECLINE) indicates completion
756 s_acd_callback_received = false;
757 s_acd_last_state = ACD_IP_OK; // Default assumption, but won't be used unless callback_received is true
758 CipTcpIpSetLastAcdActivity(2);
759
760 // Verify netif is still valid (may have been invalidated)
761 if (netif == NULL) {
762 ESP_LOGW(TAG, "tcpip_perform_acd: netif became NULL - ACD cancelled");
763 return true; // Return true to allow IP assignment (can't perform ACD without netif)
764 }
765
766 // Allocate context on heap: tcpip_callback_with_block executes asynchronously,
767 // so stack-allocated context would be corrupted
768 AcdStartContext *ctx = (AcdStartContext *)malloc(sizeof(AcdStartContext));
769 if (ctx == NULL) {
770 ESP_LOGE(TAG, "tcpip_perform_acd: Failed to allocate ACD context");
771 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
772 CipTcpIpSetLastAcdActivity(3);
773 return false;
774 }
775
776 ctx->netif = netif;
777 ctx->ip = *ip;
778 ctx->err = ERR_OK;
779
780 ESP_LOGD(TAG, "tcpip_perform_acd: Registering ACD client for IP " IPSTR, IP2STR(ip));
781
782 // Create registration semaphore to wait for callback to complete registration
783 if (s_acd_registration_sem == NULL) {
784 s_acd_registration_sem = xSemaphoreCreateBinary();
785 if (s_acd_registration_sem == NULL) {
786 ESP_LOGE(TAG, "Failed to create ACD registration semaphore");
787 free(ctx);
788 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
789 CipTcpIpSetLastAcdActivity(3);
790 return false;
791 }
792 }
793
794 // Clear any stale signals
795 while (xSemaphoreTake(s_acd_registration_sem, 0) == pdTRUE) {
796 /* flush any stale signals */
797 }
798
799 // Try direct registration first (faster), fallback to callback if needed
800 if (!s_acd_registered) {
801 ESP_LOGD(TAG, "tcpip_perform_acd: Attempting direct ACD registration");
802 netif->acd_list = NULL;
803 memset(&s_static_ip_acd, 0, sizeof(s_static_ip_acd));
804 err_t add_err = acd_add(netif, &s_static_ip_acd, tcpip_acd_conflict_callback);
805 if (add_err == ERR_OK) {
806 s_acd_registered = true;
807 ESP_LOGD(TAG, "tcpip_perform_acd: Direct ACD registration succeeded");
808 free(ctx); // Free context since we didn't use callback
809 } else {
810 ESP_LOGW(TAG, "tcpip_perform_acd: Direct registration failed (err=%d), trying callback", (int)add_err);
811 // Fall through to callback method
812 }
813 }
814
815 // If direct registration failed, try via callback (ensures thread safety)
816 if (!s_acd_registered) {
817 ESP_LOGD(TAG, "tcpip_perform_acd: Registering ACD client via callback");
818 err_t callback_err = tcpip_callback_with_block(tcpip_acd_start_cb, ctx, 1);
819 // ctx is now freed by the callback
820
821 if (callback_err != ERR_OK) {
822 ESP_LOGE(TAG, "Failed to register ACD client (callback_err=%d)", (int)callback_err);
823 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
824 CipTcpIpSetLastAcdActivity(3);
825 return false;
826 }
827
828 // Wait for registration callback to complete (ensures s_acd_registered is set)
829 TickType_t registration_timeout = pdMS_TO_TICKS(500); // 500ms timeout
830 if (xSemaphoreTake(s_acd_registration_sem, registration_timeout) != pdTRUE) {
831 ESP_LOGW(TAG, "ACD registration callback timed out - trying direct registration as fallback");
832 // Last resort: try direct registration again
833 if (!s_acd_registered) {
834 netif->acd_list = NULL;
835 memset(&s_static_ip_acd, 0, sizeof(s_static_ip_acd));
836 err_t add_err = acd_add(netif, &s_static_ip_acd, tcpip_acd_conflict_callback);
837 if (add_err == ERR_OK) {
838 s_acd_registered = true;
839 ESP_LOGI(TAG, "tcpip_perform_acd: Fallback direct registration succeeded");
840 } else {
841 ESP_LOGE(TAG, "ACD registration failed via both callback and direct methods");
842 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
843 CipTcpIpSetLastAcdActivity(3);
844 return false;
845 }
846 }
847 }
848
849 if (!s_acd_registered) {
850 ESP_LOGE(TAG, "ACD registration callback completed but registration failed");
851 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
852 CipTcpIpSetLastAcdActivity(3);
853 return false;
854 }
855 }
856
857 // Start ACD probe directly (we're on tcpip thread or direct registration succeeded)
858 if (s_acd_probe_pending && s_acd_registered) {
859 ESP_LOGD(TAG, "tcpip_perform_acd: Starting ACD probe for IP " IPSTR, IP2STR(ip));
860 err_t acd_start_err = acd_start(netif, &s_static_ip_acd, *ip);
861 if (acd_start_err == ERR_OK) {
862 ESP_LOGD(TAG, "tcpip_perform_acd: ACD probe started");
863 } else {
864 ESP_LOGE(TAG, "tcpip_perform_acd: acd_start() failed with err=%d", (int)acd_start_err);
865 // Try via callback as fallback
866 AcdStartProbeContext *probe_ctx = (AcdStartProbeContext *)malloc(sizeof(AcdStartProbeContext));
867 if (probe_ctx == NULL) {
868 ESP_LOGE(TAG, "Failed to allocate probe context");
869 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
870 CipTcpIpSetLastAcdActivity(3);
871 return false;
872 }
873
874 probe_ctx->netif = netif;
875 probe_ctx->ip = *ip;
876 probe_ctx->err = ERR_OK;
877
878 err_t callback_err = tcpip_callback_with_block(acd_start_probe_cb, probe_ctx, 1);
879 if (callback_err != ERR_OK) {
880 ESP_LOGE(TAG, "tcpip_perform_acd: acd_start() callback failed (callback_err=%d)",
881 (int)callback_err);
882 free(probe_ctx);
883 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
884 CipTcpIpSetLastAcdActivity(3);
885 return false;
886 }
887 ESP_LOGI(TAG, "tcpip_perform_acd: ACD probe started via callback");
888 }
889 } else {
890 ESP_LOGW(TAG, "tcpip_perform_acd: Cannot start ACD probe - probe_pending=%d, registered=%d",
891 s_acd_probe_pending, s_acd_registered);
892 }
893
894 // Wait for ACD to complete - probe phase takes ~600-800ms (3 probes 200ms + wait times)
895 // Announce phase takes ~8s (4 announcements 2s), but we can assign IP after probes complete
896 // Total probe phase: PROBE_WAIT (0-200ms) + 3 probes (200ms each) + ANNOUNCE_WAIT (2000ms) = ~2.8-3s max
897 // But we can assign after probe phase completes, so wait ~1.5s for probes + initial announce
898 TickType_t wait_ticks = pdMS_TO_TICKS(2000); // Increased from 500ms to allow full probe sequence
899
900 ESP_LOGD(TAG, "Waiting for ACD probe sequence to complete (timeout: 2000ms)...");
901 if (xSemaphoreTake(s_acd_sem, wait_ticks) == pdTRUE) {
902 ESP_LOGI(TAG, "ACD completed with state=%d", (int)s_acd_last_state);
903 if (s_acd_last_state == ACD_IP_OK) {
904 CipTcpIpSetLastAcdActivity(0);
905 return true;
906 }
907 // If we got a callback but not ACD_IP_OK, it might be a conflict
908 if (s_acd_last_state == ACD_DECLINE || s_acd_last_state == ACD_RESTART_CLIENT) {
909 ESP_LOGE(TAG, "ACD detected conflict (state=%d) - IP should not be assigned", (int)s_acd_last_state);
910 CipTcpIpSetLastAcdActivity(3);
911 return false;
912 }
913 } else if (s_acd_callback_received && s_acd_last_state == ACD_IP_OK) {
914 // MODIFICATION: ACD callback was received but semaphore wait timed out
915 // Added by: Adam G. Sweeney <agsweeney@gmail.com>
916 // This is OK - callback set state to IP_OK, so we can safely continue
917 // The timeout occurred because callback came after semaphore wait started,
918 // but the state change confirms ACD completed successfully
919 ESP_LOGI(TAG, "ACD callback received (state=IP_OK) - semaphore timeout was harmless, continuing with IP assignment");
920 CipTcpIpSetLastAcdActivity(0);
921 return true;
922 }
923
924 // Timeout - check if callback set conflict state during wait
925 // Only explicit ACD_DECLINE/ACD_RESTART_CLIENT from callback indicates conflict
926 // Timeout without callback means no conflict (s_acd_last_state remains ACD_IP_OK)
927 if (s_acd_last_state == ACD_RESTART_CLIENT || s_acd_last_state == ACD_DECLINE) {
928 // This state only occurs if callback was received, so it's a real conflict
929 ESP_LOGE(TAG, "ACD conflict detected during probe phase (state=%d) - IP should not be assigned", (int)s_acd_last_state);
930 CipTcpIpSetLastAcdActivity(3);
931 tcpip_callback_with_block(tcpip_acd_stop_cb, NULL, 1);
932 return false;
933 }
934
935 // Timeout without callback - ACD probe sequence is still in progress
936 // Don't assign IP until callback confirms completion (ACD_IP_OK)
937 // The callback will fire when announce phase completes (~6-10 seconds total)
938 // Return true here to indicate "no conflict detected yet, waiting for callback"
939 // This is different from returning false (which indicates actual conflict)
940 ESP_LOGW(TAG, "ACD probe wait timed out (state=%d) - callback not received yet (probe sequence still running)", (int)s_acd_last_state);
941 ESP_LOGW(TAG, "Note: ACD probe sequence can take 6-10 seconds (probes + announcements). Waiting for callback...");
942 ESP_LOGW(TAG, "IP assignment will occur when ACD_IP_OK callback is received.");
943 // Return true to indicate "no conflict, but waiting for callback to assign IP"
944 // The callback will trigger IP assignment when it fires (see tcpip_acd_conflict_callback)
945 return true;
946}

948static void tcpip_try_pending_acd(esp_netif_t *netif, struct netif *lwip_netif) {
949 ESP_LOGI(TAG, "tcpip_try_pending_acd: called - probe_pending=%d, netif=%p, lwip_netif=%p",
950 s_acd_probe_pending, netif, lwip_netif);
951 if (!s_acd_probe_pending || netif == NULL || lwip_netif == NULL) {
952 ESP_LOGW(TAG, "tcpip_try_pending_acd: Skipping - probe_pending=%d, netif=%p, lwip_netif=%p",
953 s_acd_probe_pending, netif, lwip_netif);
954 return;
955 }
956 if (!netif_has_valid_hwaddr(lwip_netif)) {
957 ESP_LOGI(TAG, "ACD deferred until MAC address is available");
958 return;
959 }
960 // Check if link is actually up - sometimes netif_is_link_up() can be delayed
961 // Use a small delay to allow netif to fully initialize after ETHERNET_EVENT_CONNECTED
962 if (!netif_is_link_up(lwip_netif)) {
963 ESP_LOGI(TAG, "ACD deferred until link is up (link status: %d) - will retry", netif_is_link_up(lwip_netif));
964 // Note: "invalid static ip" error from esp_netif_handlers is expected and harmless.
965 // IP hasn't been assigned yet (waiting for ACD). Error disappears once IP is assigned.
966 // Retry after short delay - link should be up soon after ETHERNET_EVENT_CONNECTED
967 sys_timeout(100, tcpip_retry_acd_deferred, netif);
968 return;
969 }
970 ESP_LOGI(TAG, "tcpip_try_pending_acd: All conditions met, starting ACD...");
971
972 /* Legacy ACD flow: Perform ACD BEFORE setting IP (better conflict detection) */
973 ESP_LOGW(TAG, "Using legacy ACD mode - ACD runs before IP assignment");
974 ip4_addr_t desired_ip = { .addr = s_pending_static_ip_cfg.ip.addr };
975 CipTcpIpSetLastAcdActivity(2);
976 ESP_LOGD(TAG, "Legacy ACD: Starting probe sequence for IP " IPSTR " BEFORE IP assignment", IP2STR(&desired_ip));
977
978 // Run ACD BEFORE assigning IP to ensure conflicts are detected first
979 (void)tcpip_perform_acd(lwip_netif, &desired_ip);
980
981 // Check if callback was received and indicates conflict
982 if (s_acd_callback_received && (s_acd_last_state == ACD_DECLINE || s_acd_last_state == ACD_RESTART_CLIENT)) {
983 ESP_LOGE(TAG, "ACD conflict detected for " IPSTR " - NOT assigning IP", IP2STR(&desired_ip));
984 ESP_LOGW(TAG, "IP assignment cancelled due to ACD conflict");
985 g_tcpip.status |= kTcpipStatusAcdStatus | kTcpipStatusAcdFault;
986 CipTcpIpSetLastAcdActivity(3);
987 s_acd_probe_pending = false;
988 // Stop ACD and cancel any pending retry timers
989 tcpip_callback_with_block(tcpip_acd_stop_cb, NULL, 1);
990 // Don't assign IP if conflict detected
991 return;
992 }
993
994 // If callback was received with ACD_IP_OK, assign IP now
995 // Otherwise (timeout without callback), IP will be assigned when callback fires
996 if (s_acd_callback_received && s_acd_last_state == ACD_IP_OK) {
997 ESP_LOGI(TAG, "Legacy ACD: No conflict detected - assigning IP " IPSTR, IP2STR(&desired_ip));
998 ESP_ERROR_CHECK(esp_netif_set_ip_info(netif, &s_pending_static_ip_cfg));
999 opener_configure_dns(netif);
1000 s_acd_probe_pending = false;
1001 } else {
1002 // Timeout without callback - probe sequence still running
1003 // IP will be assigned when ACD_IP_OK callback fires (see tcpip_acd_conflict_callback)
1004 ESP_LOGI(TAG, "Legacy ACD: Probe sequence in progress - IP will be assigned when callback fires");
1005 }
1006
1007 // ACD_IP_OK callback fires AFTER announce phase completes, which means ACD is already in ONGOING state.
1008 // The ACD timer will naturally transition: PROBE_WAIT PROBING ANNOUNCE_WAIT ANNOUNCING ONGOING
1009 // So we don't need to manually transition - ACD is already in ONGOING state and will send periodic defensive ARPs.
1010 // Just set activity = 1 to indicate ongoing defense phase.
1011 CipTcpIpSetLastAcdActivity(1);
1012 ESP_LOGD(TAG, "Legacy ACD: ACD is in ONGOING state (callback fired after announce phase), periodic defense active");
1013
1014 // Cancel any pending retry timers (retry handler checks s_acd_probe_pending and skips gracefully)
1015}

1224 g_tcpip.status &= ~kTcpipStatusIfaceCfgPend;

1513// Comprehensive I2C bus scan function
1514static void scan_i2c_bus(i2c_master_bus_handle_t bus_handle)
1515{
1516 if (bus_handle == NULL) {
1517 ESP_LOGW(TAG, "I2C bus scan: bus handle is NULL");
1518 return;
1519 }
1520
1521 ESP_LOGI(TAG, "Scanning I2C bus for devices...");
1522
1523 uint8_t found_addresses[128] = {0};
1524 int found_count = 0;
1525
1526 // Scan addresses 0x08 to 0x77 (valid I2C address range, excluding reserved addresses)
1527 // Reserved addresses: 0x00-0x07 (general call, reserved), 0x78-0x7F (reserved)
1528 for (uint8_t addr = 0x08; addr <= 0x77; addr++) {
1529 esp_err_t err = i2c_master_probe(bus_handle, addr, 100); // Short timeout for scan
1530 if (err == ESP_OK) {
1531 found_addresses[found_count++] = addr;
1532 }
1533 // Small delay to avoid bus congestion
1534 vTaskDelay(pdMS_TO_TICKS(1));
1535 }
1536
1537 if (found_count == 0) {
1538 ESP_LOGW(TAG, "I2C bus scan: No devices found");
1539 return;
1540 }
1541
1542 ESP_LOGI(TAG, "I2C bus scan: Found %d device(s):", found_count);
1543
1544 // Print found devices in a formatted table
1545 ESP_LOGI(TAG, "");
1546 ESP_LOGI(TAG, " Address Device Type ");
1547 ESP_LOGI(TAG, "");
1548
1549 for (int i = 0; i < found_count; i++) {
1550 uint8_t addr = found_addresses[i];
1551 const char* device_name = "Unknown device";
1552
1553 switch (addr) {
1554 case 0x18:
1555 device_name = "ES8311 (Audio Codec)";
1556 break;
1557 case 0x29:
1558 device_name = "Unknown device";
1559 break;
1560 case 0x2A:
1561 device_name = "Unknown device";
1562 break;
1563 case 0x68:
1564 device_name = "MPU6050 (IMU) - AD0 LOW";
1565 break;
1566 case 0x69:
1567 device_name = "MPU6050 (IMU) - AD0 HIGH";
1568 break;
1569 case 0x6A:
1570 device_name = "LSM6DS3 (IMU) - SA0 LOW";
1571 break;
1572 case 0x6B:
1573 device_name = "LSM6DS3 (IMU) - SA0 HIGH";
1574 break;
1575 default:
1576 device_name = "Unknown device";
1577 break;
1578 }
1579
1580 ESP_LOGI(TAG, " 0x%02X %-35s ", addr, device_name);
1581 }
1582
1583 ESP_LOGI(TAG, "");
1584
1585 // Additional info for known devices
1586 bool has_mpu6050 = false;
1587 bool has_lsm6ds3 = false;
1588
1589 for (int i = 0; i < found_count; i++) {
1590 uint8_t addr = found_addresses[i];
1591 if (addr == 0x68 || addr == 0x69) has_mpu6050 = true;
1592 if (addr == 0x6A || addr == 0x6B) has_lsm6ds3 = true;
1593 }
1594
1595 ESP_LOGI(TAG, "Device summary:");
1596 if (has_mpu6050) ESP_LOGI(TAG, " MPU6050 IMU detected");
1597 if (has_lsm6ds3) ESP_LOGI(TAG, " LSM6DS3 IMU detected");
1598}

1681// Helper function to try initializing MPU6050
1682static bool try_init_mpu6050(i2c_master_bus_handle_t bus_handle)
1683{
1684 // Try both I2C addresses (0x68 and 0x69) - AD0 pin can be LOW or HIGH
1685 uint8_t mpu6050_addr = MPU6050_I2C_ADDR_PRIMARY; // Start with 0x68
1686 bool device_found = false;
1687
1688 esp_err_t probe_err = i2c_master_probe(bus_handle, MPU6050_I2C_ADDR_PRIMARY, 1000);
1689 if (probe_err == ESP_OK) {
1690 mpu6050_addr = MPU6050_I2C_ADDR_PRIMARY;
1691 device_found = true;
1692 } else {
1693 // Try secondary address
1694 probe_err = i2c_master_probe(bus_handle, MPU6050_I2C_ADDR_SECONDARY, 1000);
1695 if (probe_err == ESP_OK) {
1696 mpu6050_addr = MPU6050_I2C_ADDR_SECONDARY;
1697 device_found = true;
1698 }
1699 }
1700
1701 if (!device_found) {
1702 ESP_LOGI(TAG, "MPU6050: Device not detected at either address (0x%02X or 0x%02X)",
1703 MPU6050_I2C_ADDR_PRIMARY, MPU6050_I2C_ADDR_SECONDARY);
1704 return false;
1705 }
1706
1707 // Configure I2C device for MPU6050
1708 i2c_device_config_t dev_cfg = {
1709 .dev_addr_length = I2C_ADDR_BIT_LEN_7,
1710 .device_address = mpu6050_addr,
1711 .scl_speed_hz = 400000,
1712 };
1713
1714 esp_err_t err = i2c_master_bus_add_device(bus_handle, &dev_cfg, &s_mpu6050_dev_handle);
1715 if (err != ESP_OK) {
1716 ESP_LOGE(TAG, "MPU6050: Failed to add I2C device: %s", esp_err_to_name(err));
1717 return false;
1718 }
1719
1720 vTaskDelay(pdMS_TO_TICKS(200)); // Give I2C bus a moment to stabilize after adding device
1721
1722 // Initialize MPU6050
1723 if (!mpu6050_init(&s_mpu6050, s_mpu6050_dev_handle)) {
1724 ESP_LOGE(TAG, "MPU6050: Failed to initialize device structure");
1725 i2c_master_bus_rm_device(s_mpu6050_dev_handle);
1726 s_mpu6050_dev_handle = NULL;
1727 return false;
1728 }
1729
1730 // Read WHO_AM_I register to verify connection with retries
1731 uint8_t who_am_i = 0;
1732 bool who_am_i_success = false;
1733 esp_err_t last_err = ESP_OK;
1734
1735 for (int retry = 0; retry < 10; retry++) {
1736 err = mpu6050_read_who_am_i(&s_mpu6050, &who_am_i);
1737 last_err = err;
1738
1739 if (err == ESP_OK) {
1740 who_am_i_success = true;
1741 break;
1742 }
1743
1744 // Retry on timeout, invalid state, or any I2C error
1745 if (err == ESP_ERR_TIMEOUT || err == ESP_ERR_INVALID_STATE ||
1746 err == ESP_ERR_INVALID_RESPONSE || err == ESP_FAIL) {
1747 int delay_ms = 200 * (retry + 1);
1748 vTaskDelay(pdMS_TO_TICKS(delay_ms));
1749 } else {
1750 ESP_LOGE(TAG, "MPU6050: Unexpected error type, not retrying: %s", esp_err_to_name(err));
1751 break;
1752 }
1753 }
1754
1755 if (!who_am_i_success) {
1756 ESP_LOGE(TAG, "MPU6050: Failed to read WHO_AM_I after %d attempts. Last error: %s",
1757 10, esp_err_to_name(last_err));
1758 i2c_master_bus_rm_device(s_mpu6050_dev_handle);
1759 s_mpu6050_dev_handle = NULL;
1760 return false;
1761 }
1762
1763 if (who_am_i == MPU6050_WHO_AM_I_VALUE) {
1764 ESP_LOGI(TAG, "MPU6050: Standard chip detected (WHO_AM_I: 0x%02X)", who_am_i);
1765 } else if (who_am_i == 0x98) {
1766 ESP_LOGI(TAG, "MPU6050: Clone chip detected (WHO_AM_I: 0x%02X) - continuing with initialization", who_am_i);
1767 } else {
1768 ESP_LOGW(TAG, "MPU6050: Unexpected WHO_AM_I value: 0x%02X (expected 0x%02X or 0x98 for clone)",
1769 who_am_i, MPU6050_WHO_AM_I_VALUE);
1770 }
1771
1772 // Reset and wake up the device
1773 err = mpu6050_reset(&s_mpu6050);
1774 if (err != ESP_OK) {
1775 ESP_LOGW(TAG, "MPU6050: Reset failed: %s (continuing anyway)", esp_err_to_name(err));
