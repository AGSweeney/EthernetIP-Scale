195unsigned int GetPaddedLogicalPath(const EipUint8 **logical_path_segment) {
196 unsigned int padded_logical_path = *(*logical_path_segment)++;
197
198 if( (padded_logical_path & 3) == 0 ) {
199 padded_logical_path = *(*logical_path_segment)++;
200 } else if( (padded_logical_path & 3) == 1 ) {
201 (*logical_path_segment)++; /* skip pad */
202 padded_logical_path = *(*logical_path_segment)++;
203 padded_logical_path |= *(*logical_path_segment)++ << 8;
204 } else {
205 OPENER_TRACE_ERR("illegal logical path segment\n");
206 }
207 return padded_logical_path;
208}

241static void EncodeConnectionEntryList(const void *const data,
242 ENIPMessage *const outgoing_message) {
243 (void)data; /* Suppress unused parameter warning */
244
245 /* Count active connections - include all connections in the list
246 * Both explicit (Class 3) and I/O connections should be in connection_list */
247 CipUint num_entries = 0;
248 CipUint total_connections = 0;
249 DoublyLinkedListNode *node = connection_list.first;
250 while (node != NULL) {
251 CipConnectionObject *conn = node->data;
252 if (conn != NULL) {
253 total_connections++;
254 ConnectionObjectState state = ConnectionObjectGetState(conn);
255 ConnectionObjectInstanceType inst_type = ConnectionObjectGetInstanceType(conn);
256 /* Count connections that are Established (both explicit and I/O should be Established when active) */
257 if (state == kConnectionObjectStateEstablished) {
258 num_entries++;
259 OPENER_TRACE_INFO("Connection Entry List: Found Established connection, type=%d, state=%d\n",
260 (int)inst_type, (int)state);
261 } else {
262 OPENER_TRACE_INFO("Connection Entry List: Found non-Established connection, type=%d, state=%d\n",
263 (int)inst_type, (int)state);
264 }
265 }
266 node = node->next;
267 }
268 OPENER_TRACE_INFO("Connection Entry List: Total connections=%d, Established=%d\n",
269 (int)total_connections, (int)num_entries);
270
271 /* Encode: UINT num_conn_entries */
272 AddIntToMessage(num_entries, outgoing_message);
273
274 if (num_entries > 0) {
275 /* Calculate bytes needed for bit array (round up) */
276 CipUint bits_bytes = (num_entries + 7) / 8;
277 CipOctet *bits = (CipOctet *)outgoing_message->current_message_position;
278
279 /* Clear bits array */
280 memset(bits, 0, bits_bytes);
281
282 /* Set bits for active connections */
283 CipUint bit_index = 0;
284 node = connection_list.first;
285 while (node != NULL) {
286 CipConnectionObject *conn = node->data;
287 if (conn != NULL) {
288 ConnectionObjectState state = ConnectionObjectGetState(conn);
289 /* Set bit for Established connections (both explicit and I/O) */
290 if (state == kConnectionObjectStateEstablished) {
291 bits[bit_index / 8] |= (1 << (bit_index % 8));
292 bit_index++;
293 }
294 }
295 node = node->next;
296 }
297
298 /* Move message pointer past the bits array */
299 outgoing_message->current_message_position += bits_bytes;
300 outgoing_message->used_message_length += bits_bytes;
301 }
302
303 /* Ensure message length is updated (AddIntToMessage already updated it, but be explicit) */
304 /* No additional update needed - AddIntToMessage handles the UINT, and we handle the bits array above */
305}

433EipStatus HandleReceivedConnectedData(const EipUint8 *const data,
434 int data_length,
435 struct sockaddr_in *from_address) {
436
437 if( (CreateCommonPacketFormatStructure(data, data_length,
438 &g_common_packet_format_data_item) ) ==
439 kEipStatusError ) {
440 return kEipStatusError;
441 } else {
442 /* check if connected address item or sequenced address item received, otherwise it is no connected message and should not be here */
443 if( (g_common_packet_format_data_item.address_item.type_id ==
444 kCipItemIdConnectionAddress)
445 || (g_common_packet_format_data_item.address_item.type_id ==
446 kCipItemIdSequencedAddressItem) ) { /* found connected address item or found sequenced address item -> for now the sequence number will be ignored */
447 if(g_common_packet_format_data_item.data_item.type_id ==
448 kCipItemIdConnectedDataItem) { /* connected data item received */
449
450 CipConnectionObject *connection_object = GetConnectedObject(
451 g_common_packet_format_data_item.address_item.data.connection_identifier);
452 if(connection_object == NULL) {
453 return kEipStatusError;
454 }
455
456 /* only handle the data if it is coming from the originator */
457 if(connection_object->originator_address.sin_addr.s_addr ==
458 from_address->sin_addr.s_addr) {
459 ConnectionObjectResetLastPackageInactivityTimerValue(connection_object);
460
461 if(SEQ_GT32(g_common_packet_format_data_item.address_item.data.
462 sequence_number,
463 connection_object->eip_level_sequence_count_consuming) ||
464 !connection_object->eip_first_level_sequence_count_received) {
465 /* reset the watchdog timer */
466 ConnectionObjectResetInactivityWatchdogTimerValue(connection_object);
467
468 /* only inform assembly object if the sequence counter is greater or equal */
469 connection_object->eip_level_sequence_count_consuming =
470 g_common_packet_format_data_item.address_item.data.sequence_number;
471 connection_object->eip_first_level_sequence_count_received = true;
472
473 if(NULL != connection_object->connection_receive_data_function) {
474 return connection_object->connection_receive_data_function(
475 connection_object,
476 g_common_packet_format_data_item.data_item.data,
477 g_common_packet_format_data_item.data_item.length);
478 }
479 }
480 } else {
481 OPENER_TRACE_WARN(
482 "Connected Message Data Received with wrong address information\n");
483 }
484 }
485 }
486 }
487 return kEipStatusOk;
488}

595EipStatus HandleNonNullNonMatchingForwardOpenRequest(
596 CipConnectionObject *connection_object,
597 CipInstance *instance,
598 CipMessageRouterRequest *message_router_request,
599 CipMessageRouterResponse *message_router_response) {
600 /* Suppress unused parameter compiler warning. */
601 (void) connection_object;
602 (void) instance;
603
604 EipUint16 connection_status = kConnectionManagerExtendedStatusCodeSuccess;
605
606 /*check if the trigger type value is invalid or ok */
607 if(kConnectionObjectTransportClassTriggerProductionTriggerInvalid ==
608 ConnectionObjectGetTransportClassTriggerProductionTrigger(&
609 g_dummy_connection_object) )
610 {
611 return AssembleForwardOpenResponse(&g_dummy_connection_object,
612 message_router_response,
613 kCipErrorConnectionFailure,
614 kConnectionManagerExtendedStatusCodeErrorTransportClassAndTriggerCombinationNotSupported);
615 }
616
617 EipUint32 temp = ParseConnectionPath(&g_dummy_connection_object,
618 message_router_request,
619 &connection_status);
620 if(kEipStatusOk != temp) {
621 return AssembleForwardOpenResponse(&g_dummy_connection_object,
622 message_router_response,
623 temp,
624 connection_status);
625 }
626
627 /*parsing is now finished all data is available and check now establish the connection */
628 ConnectionManagementHandling *connection_management_entry =
629 GetConnectionManagementEntry( /* Gets correct open connection function for the targeted object */
630 g_dummy_connection_object.configuration_path.class_id);
631 if(NULL != connection_management_entry) {
632 if (NULL != connection_management_entry->open_connection_function) {
633 temp = connection_management_entry->open_connection_function(
634 &g_dummy_connection_object, &connection_status);
635 } else {
636 connection_status = kConnectionManagerExtendedStatusCodeMiscellaneous;
637 }
638 } else {
639 temp = kEipStatusError;
640 connection_status =
641 kConnectionManagerExtendedStatusCodeInconsistentApplicationPathCombo;
642 }
643
644 if(kEipStatusOk != temp) {
645 OPENER_TRACE_INFO("connection manager: connect failed\n");
646
647 /* Increment reject counters based on error type */
648 if (connection_status == kConnectionManagerExtendedStatusCodeErrorInvalidSegmentTypeInPath) {
649 g_connection_manager_stats.open_format_rejects++;
650 } else if (connection_status == kConnectionManagerExtendedStatusCodeErrorConnectionInUseOrDuplicateForwardOpen ||
651 connection_status == kConnectionManagerExtendedStatusCodeErrorOwnershipConflict ||
652 connection_status == kConnectionManagerExtendedStatusCodeTargetObjectOutOfConnections ||
653 connection_status == kConnectionManagerExtendedStatusCodeErrorOwnershipConflict) {
654 g_connection_manager_stats.open_resource_rejects++;
655 } else {
656 g_connection_manager_stats.open_other_rejects++;
657 }
658
659 /* in case of error the dummy objects holds all necessary information */
660 return AssembleForwardOpenResponse(&g_dummy_connection_object,
661 message_router_response,
662 temp,
663 connection_status);
664 } else {
665 OPENER_TRACE_INFO("connection manager: connect succeeded\n");
666 g_connection_manager_stats.open_requests++; /* Successful open */
667 /* in case of success the new connection is added at the head of the connection list */
668 return AssembleForwardOpenResponse(connection_list.first->data,
669 message_router_response,
670 kCipErrorSuccess,
671 0);
672 }

764 ConnectionObjectConnectionType o_to_t_connection_type =
765 ConnectionObjectGetOToTConnectionType(&g_dummy_connection_object);
766 ConnectionObjectConnectionType t_to_o_connection_type =
767 ConnectionObjectGetTToOConnectionType(&g_dummy_connection_object);
768
769 /* Check if both connection types are valid, otherwise send error response */
770 if(kConnectionObjectConnectionTypeInvalid == o_to_t_connection_type) {
771 return AssembleForwardOpenResponse(&g_dummy_connection_object,
772 message_router_response,
773 kCipErrorConnectionFailure,
774 kConnectionManagerExtendedStatusCodeErrorInvalidOToTConnectionType);
775 }
776
777 if(kConnectionObjectConnectionTypeInvalid == t_to_o_connection_type) {
778 return AssembleForwardOpenResponse(&g_dummy_connection_object,
779 message_router_response,
780 kCipErrorConnectionFailure,
781 kConnectionManagerExtendedStatusCodeErrorInvalidTToOConnectionType);
782 }
783
784 if(kConnectionObjectConnectionTypeMulticast == t_to_o_connection_type) {
785 /* for multicast, check if IP is within configured net because we send TTL 1 */
786 CipUdint originator_ip =
787 ( (struct sockaddr_in *) originator_address )->sin_addr.s_addr;
788 CipUdint interface_ip = g_network_status.ip_address;
789 CipUdint interface_mask = g_network_status.network_mask;
790 if( (originator_ip & interface_mask) != (interface_ip & interface_mask) ) {
791 return AssembleForwardOpenResponse(&g_dummy_connection_object,
792 message_router_response,
793 kCipErrorConnectionFailure,
794 kConnectionManagerExtendedStatusCodeNotConfiguredForOffSubnetMulticast);
795 }
796 }
797
798 /* Check if request is a Null request or a Non-Null request */
799 if(kConnectionObjectConnectionTypeNull == o_to_t_connection_type &&
800 kConnectionObjectConnectionTypeNull == t_to_o_connection_type) {
801 is_null_request = true;
802 OPENER_TRACE_INFO("We have a Null request\n");
803 } else {
804 is_null_request = false;
805 OPENER_TRACE_INFO("We have a Non-Null request\n");
806 }
807
808 /* Check if we have a matching or non matching request */
809 if(NULL != CheckForExistingConnection(&g_dummy_connection_object) ) {
810 OPENER_TRACE_INFO("We have a Matching request\n");
811 is_matching_request = true;
812
813 } else {
814 OPENER_TRACE_INFO("We have a Non-Matching request\n");
815 is_matching_request = false;
816 }
817
818 HandleForwardOpenRequestFunction choosen_function =
819 handle_forward_open_request_functions[is_null_request][is_matching_request];
820
821 return choosen_function(&g_dummy_connection_object,
822 instance,
823 message_router_request,
824 message_router_response);
825}

1097EipStatus ManageConnections(MilliSeconds elapsed_time) {
1098 //OPENER_TRACE_INFO("Entering ManageConnections\n");
1099 /*Inform application that it can execute */
1100 HandleApplication();
1101 ManageEncapsulationMessages(elapsed_time);
1102
1103 DoublyLinkedListNode *node = connection_list.first;
1104
1105 while(NULL != node) {
1106 //OPENER_TRACE_INFO("Entering Connection Object loop\n");
1107 CipConnectionObject *connection_object = node->data;
1108 if(kConnectionObjectStateEstablished ==
1109 ConnectionObjectGetState(connection_object) ) {
1110 if( (NULL != connection_object->consuming_instance) || /* we have a consuming connection check inactivity watchdog timer */
1111 (kConnectionObjectTransportClassTriggerDirectionServer ==
1112 ConnectionObjectGetTransportClassTriggerDirection(connection_object) ) ) /* all server connections have to maintain an inactivity watchdog timer */
1113 {
1114 if(elapsed_time >= connection_object->inactivity_watchdog_timer) {
1115 /* we have a timed out connection perform watchdog time out action*/
1116 OPENER_TRACE_INFO(">>>>>>>>>>Connection ConnNr: %u timed out\n",
1117 connection_object->connection_serial_number);
1118 g_connection_manager_stats.connection_timeouts++; /* Increment timeout counter */
1119 OPENER_ASSERT(NULL != connection_object->connection_timeout_function);
1120 connection_object->connection_timeout_function(connection_object);
1121 } else {
1122 connection_object->inactivity_watchdog_timer -= elapsed_time;
1123 connection_object->last_package_watchdog_timer -= elapsed_time;
1124 }
1125 }
1126 /* only if the connection has not timed out check if data is to be send */
1127 if(kConnectionObjectStateEstablished ==
1128 ConnectionObjectGetState(connection_object) ) {
1129 /* client connection */
1130 if( (0 != ConnectionObjectGetExpectedPacketRate(connection_object) )
1131 && (kEipInvalidSocket !=
1132 connection_object->socket[kUdpCommuncationDirectionProducing]) ) /* only produce for the master connection */
1133 {
1134 if(kConnectionObjectTransportClassTriggerProductionTriggerCyclic !=
1135 ConnectionObjectGetTransportClassTriggerProductionTrigger(
1136 connection_object) ) {
1137 /* non cyclic connections have to decrement production inhibit timer */
1138 if(elapsed_time <= connection_object->production_inhibit_timer) {
1139 //The connection is allowed to send again
1140 } else {
1141 connection_object->production_inhibit_timer -= elapsed_time;
1142 }
1143 }
1144
1145 if(connection_object->transmission_trigger_timer <= elapsed_time) { /* need to send package */
1146 OPENER_ASSERT(
1147 NULL != connection_object->connection_send_data_function);
1148 EipStatus eip_status =
1149 connection_object->connection_send_data_function(connection_object);
1150 if(eip_status == kEipStatusError) {
1151 OPENER_TRACE_ERR(
1152 "sending of UDP data in manage Connection failed\n");
1153 }
1154 /* add the RPI to the timer value */
1155 connection_object->transmission_trigger_timer +=
1156 ConnectionObjectGetRequestedPacketInterval(connection_object);
1157 /* decrecment the elapsed time from timer value, if less than timer value */
1158 if (connection_object->transmission_trigger_timer > elapsed_time) {
1159 connection_object->transmission_trigger_timer -= elapsed_time;
1160 } else { /* elapsed time was longer than RPI */
1161 connection_object->transmission_trigger_timer = 0;
1162 OPENER_TRACE_INFO("elapsed time: %lu ms was longer than RPI: %u ms\n",
1163 elapsed_time,
1164 ConnectionObjectGetRequestedPacketInterval(connection_object));
1165 }
1166 if(kConnectionObjectTransportClassTriggerProductionTriggerCyclic !=
1167 ConnectionObjectGetTransportClassTriggerProductionTrigger(
1168 connection_object) ) {
1169 /* non cyclic connections have to reload the production inhibit timer */
1170 ConnectionObjectResetProductionInhibitTimer(connection_object);
1171 }
1172 } else {
1173 connection_object->transmission_trigger_timer -= elapsed_time;
1174 }
1175 }
1176 }
1177 }
1178 node = node->next;
1179 }
1180 return kEipStatusOk;
1181}

1194EipStatus AssembleForwardOpenResponse(CipConnectionObject *connection_object,
1195 CipMessageRouterResponse *message_router_response,
1196 EipUint8 general_status,
1197 EipUint16 extended_status) {
1198 /* write reply information in CPF struct dependent of pa_status */
1199 CipCommonPacketFormatData *cip_common_packet_format_data =
1200 &g_common_packet_format_data_item;
1201 cip_common_packet_format_data->item_count = 2;
1202 cip_common_packet_format_data->data_item.type_id =
1203 kCipItemIdUnconnectedDataItem;
1204
1205 AddNullAddressItem(cip_common_packet_format_data);
1206
1207 CIPServiceCode service_code = kForwardOpen;
1208 if(connection_object->is_large_forward_open) {
1209 service_code = kLargeForwardOpen;
1210 }
1211
1212 message_router_response->reply_service = (0x80 | service_code);
1213 message_router_response->general_status = general_status;
1214
1215 if(kCipErrorSuccess == general_status) {
1216 OPENER_TRACE_INFO("assembleFWDOpenResponse: sending success response\n");
1217 /* if there is no application specific data, total length should be 26 */
1218 message_router_response->size_of_additional_status = 0;
1219
1220 if(cip_common_packet_format_data->address_info_item[0].type_id != 0) {
1221 cip_common_packet_format_data->item_count = 3;
1222 if(cip_common_packet_format_data->address_info_item[1].type_id != 0) {
1223 cip_common_packet_format_data->item_count = 4; /* there are two sockaddrinfo items to add */
1224 }
1225 }
1226
1227 AddDintToMessage(connection_object->cip_consumed_connection_id,
1228 &message_router_response->message);
1229 AddDintToMessage(connection_object->cip_produced_connection_id,
1230 &message_router_response->message);
1231 } else {
1232 /* we have an connection creation error */
1233 OPENER_TRACE_WARN("AssembleForwardOpenResponse: sending error response, general/extended status=%d/%d\n", general_status, extended_status);
1234 ConnectionObjectSetState(connection_object,
1235 kConnectionObjectStateNonExistent);
1236 /* Expected data length is 10 octets */
1237
1238 switch(general_status) {
1239 case kCipErrorNotEnoughData:
1240 case kCipErrorTooMuchData: {
1241 message_router_response->size_of_additional_status = 0;
1242 break;
1243 }
1244
1245 default: {
1246 switch(extended_status) {
1247 case
1248 kConnectionManagerExtendedStatusCodeErrorInvalidOToTConnectionSize:
1249 {
1250 message_router_response->size_of_additional_status = 2;
1251 message_router_response->additional_status[0] = extended_status;
1252 message_router_response->additional_status[1] =
1253 connection_object->correct_originator_to_target_size;
1254 break;
1255 }
1256
1257 case
1258 kConnectionManagerExtendedStatusCodeErrorInvalidTToOConnectionSize:
1259 {
1260 message_router_response->size_of_additional_status = 2;
1261 message_router_response->additional_status[0] = extended_status;
1262 message_router_response->additional_status[1] =
1263 connection_object->correct_target_to_originator_size;
1264 break;
1265 }
1266
1267 default: {
1268 message_router_response->size_of_additional_status = 1;
1269 message_router_response->additional_status[0] = extended_status;
1270 break;
1271 }
1272 }
1273 break;
1274 }
1275 }
1276 }
1277
1278 AddIntToMessage(connection_object->connection_serial_number,
1279 &message_router_response->message);
1280 AddIntToMessage(connection_object->originator_vendor_id,
1281 &message_router_response->message);
1282 AddDintToMessage(connection_object->originator_serial_number,
1283 &message_router_response->message);
1284
1285 if(kCipErrorSuccess == general_status) {
1286 /* set the actual packet rate to requested packet rate */
1287 AddDintToMessage(connection_object->o_to_t_requested_packet_interval,
1288 &message_router_response->message);
1289 AddDintToMessage(connection_object->t_to_o_requested_packet_interval,
1290 &message_router_response->message);
1291 }
1292
1293 AddSintToMessage(0, &message_router_response->message); /* remaining path size - for routing devices relevant */
1294 AddSintToMessage(0, &message_router_response->message); /* reserved */
1295
1296 return kEipStatusOkSend; /* send reply */
1297}
