38EipStatus NotifyCommonPacketFormat(const EncapsulationData *const received_data,
39 const struct sockaddr *const originator_address,
40 ENIPMessage *const outgoing_message) {
41 EipStatus return_value = kEipStatusError;
42 CipMessageRouterResponse message_router_response;
43 InitializeMessageRouterResponse(&message_router_response);
44
45 if(kEipStatusError
46 == (return_value =
47 CreateCommonPacketFormatStructure(received_data->
48 current_communication_buffer_position,
49 received_data->data_length,
50 &
51 g_common_packet_format_data_item) ) )
52 {
53 OPENER_TRACE_ERR("notifyCPF: error from createCPFstructure\n");
54 } else {
55 return_value = kEipStatusOkSend; /* In cases of errors we normally need to send an error response */
56 if(g_common_packet_format_data_item.address_item.type_id ==
57 kCipItemIdNullAddress) /* check if NullAddressItem received, otherwise it is no unconnected message and should not be here*/
58 { /* found null address item*/
59 if(g_common_packet_format_data_item.data_item.type_id ==
60 kCipItemIdUnconnectedDataItem) { /* unconnected data item received*/
61 return_value = NotifyMessageRouter(
62 g_common_packet_format_data_item.data_item.data,
63 g_common_packet_format_data_item.data_item.length,
64 &message_router_response,
65 originator_address,
66 received_data->session_handle);
67 if(return_value != kEipStatusError) {
68 SkipEncapsulationHeader(outgoing_message);
69 /* TODO: Here we get the status. What to do? kEipStatusError from AssembleLinearMessage().
70 * Its not clear how to transport this error information to the requester. */
71 EipStatus status = AssembleLinearMessage(&message_router_response,
72 &g_common_packet_format_data_item,
73 outgoing_message);
74 (void)status; /* Suppress unused variable warning. */
75
76 /* Save pointer and move to start for Encapusulation Header */
77 CipOctet *buffer = outgoing_message->current_message_position;
78 outgoing_message->current_message_position =
79 outgoing_message->message_buffer;
80 GenerateEncapsulationHeader(received_data,
81 outgoing_message->used_message_length,
82 received_data->session_handle,
83 kEncapsulationProtocolSuccess,
84 outgoing_message);
85 /* Move pointer back to last octet */
86 outgoing_message->current_message_position = buffer;
87 return_value = kEipStatusOkSend;
88 }
89 } else {
90 /* wrong data item detected*/
91 OPENER_TRACE_ERR(
92 "notifyCPF: got something besides the expected CIP_ITEM_ID_UNCONNECTEDMESSAGE\n");
93 GenerateEncapsulationHeader(received_data,
94 0,
95 received_data->session_handle,
96 kEncapsulationProtocolIncorrectData,
97 outgoing_message);
98 return_value = kEipStatusOkSend;
99 }
100 } else {
101 OPENER_TRACE_ERR(
102 "notifyCPF: got something besides the expected CIP_ITEM_ID_NULL\n");
103 GenerateEncapsulationHeader(received_data,
104 0,
105 received_data->session_handle,
106 kEncapsulationProtocolIncorrectData,
107 outgoing_message);
108 return_value = kEipStatusOkSend;
109 }
110 }
111 return return_value;
112}

114EipStatus NotifyConnectedCommonPacketFormat(
115 const EncapsulationData *const received_data,
116 const struct sockaddr *const originator_address,
117 ENIPMessage *const outgoing_message) {
118
119 EipStatus return_value = CreateCommonPacketFormatStructure(
120 received_data->current_communication_buffer_position,
121 received_data->data_length,
122 &g_common_packet_format_data_item);
123
124 if(kEipStatusError == return_value) {
125 OPENER_TRACE_ERR("notifyConnectedCPF: error from createCPFstructure\n");
126 } else {
127 return_value = kEipStatusError; /* For connected explicit messages status always has to be 0*/
128 if(g_common_packet_format_data_item.address_item.type_id ==
129 kCipItemIdConnectionAddress) /* check if ConnectedAddressItem received, otherwise it is no connected message and should not be here*/
130 { /* ConnectedAddressItem item */
131 CipConnectionObject *connection_object = GetConnectedObject(
132 g_common_packet_format_data_item.address_item.data.connection_identifier);
133 if(NULL != connection_object) {
134 /* reset the watchdog timer */
135 ConnectionObjectResetInactivityWatchdogTimerValue(connection_object);
136
137 /*TODO check connection id and sequence count */
138 if(g_common_packet_format_data_item.data_item.type_id ==
139 kCipItemIdConnectedDataItem) { /* connected data item received*/
140 EipUint8 *buffer = g_common_packet_format_data_item.data_item.data;
141 g_common_packet_format_data_item.address_item.data.sequence_number =
142 GetUintFromMessage( (const EipUint8 **const ) &buffer );
143 OPENER_TRACE_INFO(
144 "Class 3 sequence number: %" PRIu32 ", last sequence number: %u\n",
145 g_common_packet_format_data_item.address_item.data.sequence_number,
146 (unsigned int)connection_object->sequence_count_consuming);
147 if(connection_object->sequence_count_consuming ==
148 g_common_packet_format_data_item.address_item.data.sequence_number)
149 {
150 memcpy(outgoing_message,
151 &(connection_object->last_reply_sent),
152 sizeof(ENIPMessage) );
153 outgoing_message->current_message_position =
154 outgoing_message->message_buffer;
155 /* Regenerate encapsulation header for new message */
156 outgoing_message->used_message_length -=
157 ENCAPSULATION_HEADER_LENGTH;
158 GenerateEncapsulationHeader(received_data,
159 outgoing_message->used_message_length,
160 received_data->session_handle,
161 kEncapsulationProtocolSuccess,
162 outgoing_message);
163 outgoing_message->current_message_position = buffer;
164 /* End regenerate encapsulation header for new message */
165 return kEipStatusOkSend;
166 }
167 connection_object->sequence_count_consuming =
168 g_common_packet_format_data_item.address_item.data.sequence_number;
169
170 ConnectionObjectResetInactivityWatchdogTimerValue(connection_object);
171
172 CipMessageRouterResponse message_router_response;
173 InitializeMessageRouterResponse(&message_router_response);
174 return_value = NotifyMessageRouter(buffer,
175 g_common_packet_format_data_item.data_item.length - 2,
176 &message_router_response,
177 originator_address,
178 received_data->session_handle);
179
180 if(return_value != kEipStatusError) {
181 g_common_packet_format_data_item.address_item.data.
182 connection_identifier =
183 connection_object->cip_produced_connection_id;
184 SkipEncapsulationHeader(outgoing_message);
185 /* TODO: Here we get the status. What to do? kEipStatusError from AssembleLinearMessage().
186 * Its not clear how to transport this error information to the requester. */
187 EipStatus status = AssembleLinearMessage(&message_router_response,
188 &g_common_packet_format_data_item,
189 outgoing_message);
190 (void)status; /* Suppress unused variable warning. */
191
192 CipOctet *pos = outgoing_message->current_message_position;
193 outgoing_message->current_message_position =
194 outgoing_message->message_buffer;
195 GenerateEncapsulationHeader(received_data,
196 outgoing_message->used_message_length,
197 received_data->session_handle,
198 kEncapsulationProtocolSuccess,
199 outgoing_message);
200 outgoing_message->current_message_position = pos;
201 memcpy(&connection_object->last_reply_sent,
202 outgoing_message,
203 sizeof(ENIPMessage) );
204 return_value = kEipStatusOkSend;
205 }
206 } else {
207 /* wrong data item detected*/
208 OPENER_TRACE_ERR(
209 "notifyConnectedCPF: got something besides the expected CIP_ITEM_ID_UNCONNECTEDMESSAGE\n");
210 }
211 } else {
212 OPENER_TRACE_ERR(
213 "notifyConnectedCPF: connection with given ID could not be found\n");
214 }
215 } else {
216 OPENER_TRACE_ERR(
217 "notifyConnectedCPF: got something besides the expected CIP_ITEM_ID_NULL\n");
218 }
219 }
220 // return outgoing_message->used_message_length;
221 return (0 !=
222 outgoing_message->used_message_length ? kEipStatusOkSend :
223 kEipStatusOk); /* TODO: What would the right EipStatus to return? */
224}

235EipStatus CreateCommonPacketFormatStructure(const EipUint8 *data,
236 size_t data_length,
237 CipCommonPacketFormatData *common_packet_format_data)
238{
239
240 common_packet_format_data->address_info_item[0].type_id = 0;
241 common_packet_format_data->address_info_item[1].type_id = 0;
242
243 size_t length_count = 0;
244 CipUint item_count = GetUintFromMessage(&data);
245 //OPENER_ASSERT(4U >= item_count);/* Sanitizing data - probably needs to be changed for productive code */
246 common_packet_format_data->item_count = item_count;
247 length_count += 2;
248 if(common_packet_format_data->item_count >= 1U) {
249 common_packet_format_data->address_item.type_id = GetUintFromMessage(&data);
250 common_packet_format_data->address_item.length = GetUintFromMessage(&data);
251 length_count += 4;
252 if(common_packet_format_data->address_item.length >= 4) {
253 common_packet_format_data->address_item.data.connection_identifier =
254 GetUdintFromMessage(&data);
255 length_count += 4;
256 }
257 if(common_packet_format_data->address_item.length == 8) {
258 common_packet_format_data->address_item.data.sequence_number =
259 GetUdintFromMessage(&data);
260 length_count += 4;
261 }
262 }
263 if(common_packet_format_data->item_count >= 2) {
264 common_packet_format_data->data_item.type_id = GetUintFromMessage(&data);
265 common_packet_format_data->data_item.length = GetUintFromMessage(&data);
266 common_packet_format_data->data_item.data = (EipUint8 *) data;
267 if(data_length >=
268 length_count + 4 + common_packet_format_data->data_item.length) {
269 data += common_packet_format_data->data_item.length;
270 length_count += (4 + common_packet_format_data->data_item.length);
271 } else {
272 return kEipStatusError;
273 }
274
275 /* Data type per CIP Volume 2, Edition 1.4, Table 2-6.1. */
276 CipUint address_item_count = (CipUint)(common_packet_format_data->item_count - 2U);
277
278 for(size_t j = 0; j < (address_item_count > 2 ? 2 : address_item_count);
279 j++) /* TODO there needs to be a limit check here???*/
280 {
281 common_packet_format_data->address_info_item[j].type_id =
282 GetIntFromMessage(&data);
283 OPENER_TRACE_INFO("Sockaddr type id: %x\n",
284 common_packet_format_data->address_info_item[j].type_id);
285 length_count += 2;
286 if( (common_packet_format_data->address_info_item[j].type_id ==
287 kCipItemIdSocketAddressInfoOriginatorToTarget)
288 || (common_packet_format_data->address_info_item[j].type_id ==
289 kCipItemIdSocketAddressInfoTargetToOriginator) ) {
290 common_packet_format_data->address_info_item[j].length =
291 GetIntFromMessage(&data);
292 common_packet_format_data->address_info_item[j].sin_family =
293 GetIntFromMessage(&data);
294 common_packet_format_data->address_info_item[j].sin_port =
295 GetIntFromMessage(&data);
296 common_packet_format_data->address_info_item[j].sin_addr =
297 GetUdintFromMessage(&data);
298 for(size_t i = 0; i < 8; i++) {
299 common_packet_format_data->address_info_item[j].nasin_zero[i] = *data;
300 data++;
301 }
302 length_count += 18;
303 } else { /* no sockaddr item found */
304 common_packet_format_data->address_info_item[j].type_id = 0; /* mark as not set */
305 data -= 2;
306 }
307 }
308 }
309 /* set the addressInfoItems to not set if they were not received */
310 if(common_packet_format_data->item_count < 4) {
311 common_packet_format_data->address_info_item[1].type_id = 0;
312 if(common_packet_format_data->item_count < 3) {
313 common_packet_format_data->address_info_item[0].type_id = 0;
314 }
315 }
316 if(length_count == data_length) { /* length of data is equal to length of Addr and length of Data */
317 return kEipStatusOk;
318 } else {
319 OPENER_TRACE_WARN(
320 "something is wrong with the length in Message Router @ CreateCommonPacketFormatStructure\n");
321 if(common_packet_format_data->item_count > 2) {
322 /* there is an optional packet in data stream which is not sockaddr item */
323 return kEipStatusOk;
324 } else { /* something with the length was wrong */
325 return kEipStatusError;
326 }
327 }
328}

616EipStatus AssembleLinearMessage(
617 const CipMessageRouterResponse *const message_router_response,
618 const CipCommonPacketFormatData *const common_packet_format_data_item,
619 ENIPMessage *const outgoing_message) {
620
621 if(message_router_response) {
622 /* add Interface Handle and Timeout = 0 -> only for SendRRData and SendUnitData necessary */
623 AddDintToMessage(0, outgoing_message);
624 AddIntToMessage(0, outgoing_message);
625 }
626
627 EncodeItemCount(common_packet_format_data_item, outgoing_message);
628
629 /* process Address Item */
630 switch(common_packet_format_data_item->address_item.type_id) {
631 case kCipItemIdNullAddress: {
632 EncodeNullAddressItem(outgoing_message);
633 break;
634 }
635 case kCipItemIdConnectionAddress: {
636 EncodeConnectedAddressItem(common_packet_format_data_item,
637 outgoing_message);
638 break;
639 }
640 case kCipItemIdSequencedAddressItem: {
641 EncodeSequencedAddressItem(common_packet_format_data_item,
642 outgoing_message);
643 break;
644 }
645 default:
646 OPENER_TRACE_INFO("Unknown CIP Item in AssembleLinearMessage");
647 return kEipStatusError;
648 }
649
650 /* process Data Item */
651 if( (common_packet_format_data_item->data_item.type_id ==
652 kCipItemIdUnconnectedDataItem)
653 || (common_packet_format_data_item->data_item.type_id ==
654 kCipItemIdConnectedDataItem) ) {
655
656 if(message_router_response) {
657 EncodeDataItemType(common_packet_format_data_item, outgoing_message);
658
659 if(common_packet_format_data_item->data_item.type_id ==
660 kCipItemIdConnectedDataItem) { /* Connected Item */
661 EncodeConnectedDataItemLength(message_router_response,
662 outgoing_message);
663 EncodeSequenceNumber(&g_common_packet_format_data_item,
664 outgoing_message);
665
666 } else { /* Unconnected Item */
667 EncodeUnconnectedDataItemLength(message_router_response,
668 outgoing_message);
669 }
670
671 /* write message router response into linear memory */
672 EncodeReplyService(message_router_response, outgoing_message);
673 EncodeReservedFieldOfLengthByte(message_router_response,
674 outgoing_message);
675 EncodeGeneralStatus(message_router_response, outgoing_message);
676 EncodeExtendedStatus(message_router_response, outgoing_message);
677 EncodeMessageRouterResponseData(message_router_response,
678 outgoing_message);
679 } else { /* connected IO Message to send */
680 EncodeDataItemType(common_packet_format_data_item, outgoing_message);
681
682 EncodeDataItemLength(common_packet_format_data_item, outgoing_message);
683
684 EncodeDataItemData(common_packet_format_data_item, outgoing_message);
685 }
686 }
687
688 /* process SockAddr Info Items */
689 /* make sure first the O->T and then T->O appears on the wire.
690 * EtherNet/IP specification doesn't demand it, but there are EIP
691 * devices which depend on CPF items to appear in the order of their
692 * ID number */
693 for(int type = kCipItemIdSocketAddressInfoOriginatorToTarget;
694 type <= kCipItemIdSocketAddressInfoTargetToOriginator; type++) {
695 for(int j = 0; j < 2; j++) {
696 if(common_packet_format_data_item->address_info_item[j].type_id == type) {
697 EncodeSockaddrInfoItemTypeId(j,
698 common_packet_format_data_item,
699 outgoing_message);
700
701 EncodeSockaddrInfoLength(j,
702 common_packet_format_data_item,
703 outgoing_message);
704
705 EncapsulateIpAddress(
706 common_packet_format_data_item->address_info_item[j].sin_port,
707 common_packet_format_data_item->address_info_item[j].sin_addr,
708 outgoing_message);
709
710 FillNextNMessageOctetsWithValueAndMoveToNextPosition(0,
711 8,
712 outgoing_message);
713 break;
714 }
715 }
716 }
717 return kEipStatusOk;
718}
