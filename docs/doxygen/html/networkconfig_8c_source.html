43EipStatus IfaceGetConfiguration(TcpIpInterface *iface,
44 CipTcpIpInterfaceConfiguration *iface_cfg) {
45 CipTcpIpInterfaceConfiguration local_cfg;
46 EipStatus status;
47
48 memset(&local_cfg, 0x00, sizeof local_cfg);
49
50 status = GetIpAndNetmaskFromInterface(iface, &local_cfg);
51 if (kEipStatusOk == status) {
52 status = GetGatewayFromRoute(iface, &local_cfg);
53 }
54 if (kEipStatusOk == status) {
55 ClearCipString(&iface_cfg->domain_name);
56 *iface_cfg = local_cfg;
