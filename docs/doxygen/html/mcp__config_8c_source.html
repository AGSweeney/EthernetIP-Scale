37void mcp_config_get_defaults(mcp_device_config_t *config, uint8_t i2c_address, uint8_t device_type)
38{
39 if (config == NULL) {
40 return;
41 }
42
43 memset(config, 0, sizeof(mcp_device_config_t));
44 config->i2c_address = i2c_address;
45 config->device_type = device_type;
46 config->enabled = true;
47
48 // Default: all pins as outputs (0x0000 for MCP23017, 0x00 for MCP23008)
49 config->pin_directions = 0;
50 config->input_byte_start = 0;
51 config->output_byte_start = 0;
52 config->output_logic_inverted = false; // Default: standard logic (active high)
53
54 // Initialize pin mappings to disabled
55 for (int i = 0; i < MCP_MAX_PINS_MCP23017; i++) {
56 config->pin_mappings[i].enabled = false;
57 config->pin_mappings[i].assembly_type = 0;
58 config->pin_mappings[i].byte_offset = 0;
59 config->pin_mappings[i].bit_offset = 0;
60 }
61}

63bool mcp_config_load_all(mcp_config_t *config)
64{
65 if (config == NULL) {
66 return false;
67 }
68
69 nvs_handle_t handle;
70 esp_err_t err = nvs_open(NVS_NAMESPACE, NVS_READONLY, &handle);
71 if (err != ESP_OK) {
72 if (err == ESP_ERR_NVS_NOT_FOUND) {
73 ESP_LOGI(TAG, "No saved MCP configurations found, using defaults");
74 config->device_count = 0;
75 return false;
76 }
77 ESP_LOGE(TAG, "Failed to open NVS namespace: %s", esp_err_to_name(err));
78 return false;
79 }
80
81 // Load device count
82 uint8_t device_count = 0;
83 size_t required_size = sizeof(uint8_t);
84 err = nvs_get_blob(handle, NVS_KEY_COUNT, &device_count, &required_size);
85 if (err == ESP_ERR_NVS_NOT_FOUND) {
86 ESP_LOGI(TAG, "No device count found, assuming no devices");
87 config->device_count = 0;
88 nvs_close(handle);
89 return false;
90 }
91
92 if (err != ESP_OK) {
93 ESP_LOGE(TAG, "Failed to load device count: %s", esp_err_to_name(err));
94 nvs_close(handle);
95 return false;
96 }
97
98 if (device_count > MCP_MAX_DEVICES) {
99 ESP_LOGW(TAG, "Device count %d exceeds maximum %d, clamping", device_count, MCP_MAX_DEVICES);
100 device_count = MCP_MAX_DEVICES;
101 }
102
103 config->device_count = device_count;
104
105 // Load device configurations
106 required_size = sizeof(mcp_device_config_t) * device_count;
107 err = nvs_get_blob(handle, NVS_KEY_DEVICES, config->devices, &required_size);
108 nvs_close(handle);
109
110 if (err == ESP_ERR_NVS_NOT_FOUND) {
111 ESP_LOGI(TAG, "No device configurations found");
112 config->device_count = 0;
113 return false;
114 }
115
116 if (err != ESP_OK) {
117 ESP_LOGE(TAG, "Failed to load device configurations: %s", esp_err_to_name(err));
118 return false;
119 }
120
121 ESP_LOGI(TAG, "Loaded %d MCP device configuration(s) from NVS", device_count);
122 return true;
123}

125bool mcp_config_save_all(const mcp_config_t *config)
126{
127 if (config == NULL) {
128 return false;
129 }
130
131 nvs_handle_t handle;
132 esp_err_t err = nvs_open(NVS_NAMESPACE, NVS_READWRITE, &handle);
133 if (err != ESP_OK) {
134 ESP_LOGE(TAG, "Failed to open NVS namespace: %s", esp_err_to_name(err));
135 return false;
136 }
137
138 // Save device count
139 err = nvs_set_blob(handle, NVS_KEY_COUNT, &config->device_count, sizeof(uint8_t));
140 if (err != ESP_OK) {
141 ESP_LOGE(TAG, "Failed to save device count: %s", esp_err_to_name(err));
142 nvs_close(handle);
143 return false;
144 }
145
146 // Save device configurations
147 if (config->device_count > 0) {
148 err = nvs_set_blob(handle, NVS_KEY_DEVICES, config->devices,
149 sizeof(mcp_device_config_t) * config->device_count);
150 if (err != ESP_OK) {
151 ESP_LOGE(TAG, "Failed to save device configurations: %s", esp_err_to_name(err));
152 nvs_close(handle);
153 return false;
154 }
155 }
156
157 err = nvs_commit(handle);
158 nvs_close(handle);
159
160 if (err != ESP_OK) {
161 ESP_LOGE(TAG, "Failed to commit MCP configurations: %s", esp_err_to_name(err));
162 return false;
163 }
164
165 ESP_LOGI(TAG, "Saved %d MCP device configuration(s) to NVS", config->device_count);
166 return true;
167}

169bool mcp_config_load_device(uint8_t i2c_address, mcp_device_config_t *config)
170{
171 if (config == NULL) {
172 return false;
173 }
174
175 mcp_config_t all_config;
176 if (!mcp_config_load_all(&all_config)) {
177 return false;
178 }
179
180 mcp_device_config_t *found = mcp_config_find_device(&all_config, i2c_address);
181 if (found == NULL) {
182 return false;
183 }
184
185 memcpy(config, found, sizeof(mcp_device_config_t));
186 return true;
187}

189bool mcp_config_save_device(const mcp_device_config_t *config)
190{
191 if (config == NULL) {
192 return false;
193 }
194
195 mcp_config_t all_config;
196 mcp_config_load_all(&all_config); // Load existing configs (may fail, that's OK)
197
198 // Find existing device or add new one
199 mcp_device_config_t *existing = mcp_config_find_device(&all_config, config->i2c_address);
200 if (existing != NULL) {
201 // Update existing device
202 memcpy(existing, config, sizeof(mcp_device_config_t));
203 } else {
204 // Add new device
205 if (all_config.device_count >= MCP_MAX_DEVICES) {
206 ESP_LOGE(TAG, "Maximum number of devices (%d) reached", MCP_MAX_DEVICES);
207 return false;
208 }
209 memcpy(&all_config.devices[all_config.device_count], config, sizeof(mcp_device_config_t));
210 all_config.device_count++;
211 }
212
213 return mcp_config_save_all(&all_config);
214}

216bool mcp_config_delete_device(uint8_t i2c_address)
217{
218 mcp_config_t all_config;
219 if (!mcp_config_load_all(&all_config)) {
220 return false;
221 }
222
223 // Find and remove device
224 for (int i = 0; i < all_config.device_count; i++) {
225 if (all_config.devices[i].i2c_address == i2c_address) {
226 // Shift remaining devices down
227 for (int j = i; j < all_config.device_count - 1; j++) {
228 memcpy(&all_config.devices[j], &all_config.devices[j + 1], sizeof(mcp_device_config_t));
229 }
230 all_config.device_count--;
231 return mcp_config_save_all(&all_config);
232 }
233 }
234
235 return false;
236}

238mcp_device_config_t* mcp_config_find_device(mcp_config_t *config, uint8_t i2c_address)
239{
240 if (config == NULL) {
241 return NULL;
242 }
243
244 for (int i = 0; i < config->device_count; i++) {
245 if (config->devices[i].i2c_address == i2c_address) {
246 return &config->devices[i];
247 }
248 }
249
250 return NULL;
251}
