201void EncapsulateIpAddress(EipUint16 port,
202 EipUint32 address,
203 ENIPMessage *const outgoing_message) {
204 if(kOpENerEndianessLittle == g_opener_platform_endianess) {
205 AddIntToMessage(htons(AF_INET), outgoing_message);
206 AddIntToMessage(port, outgoing_message);
207 AddDintToMessage(address, outgoing_message);
208
209 } else {
210 if(kOpENerEndianessBig == g_opener_platform_endianess) {
211
212 AddIntToMessage(htons(AF_INET), outgoing_message);
213
214 AddSintToMessage( (unsigned char) (port >> 8), outgoing_message );
215 AddSintToMessage( (unsigned char) port, outgoing_message );
216
217 AddSintToMessage( (unsigned char) address, outgoing_message );
218 AddSintToMessage( (unsigned char) (address >> 8), outgoing_message );
219 AddSintToMessage( (unsigned char) (address >> 16), outgoing_message );
220 AddSintToMessage( (unsigned char) (address >> 24), outgoing_message );
221
222 } else {
223 fprintf(stderr,
224 "No endianess detected! Probably the DetermineEndianess function was not executed!");
225 exit(EXIT_FAILURE);
226 }
227 }
