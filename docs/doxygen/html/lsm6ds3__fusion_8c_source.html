35static void madgwick_ahrs_update(lsm6ds3_madgwick_filter_t *filter, float gx, float gy, float gz, float ax, float ay, float az, float dt)
36{
37 float q0 = filter->quaternion.q0;
38 float q1 = filter->quaternion.q1;
39 float q2 = filter->quaternion.q2;
40 float q3 = filter->quaternion.q3;
41
42 float recipNorm;
43 float s0, s1, s2, s3;
44 float qDot1, qDot2, qDot3, qDot4;
45 float _2q0, _2q1, _2q2, _2q3, _4q0, _4q1, _4q2, _8q1, _8q2, q0q0, q1q1, q2q2, q3q3;
46
47 if (!filter->initialized) {
48 float norm = sqrtf(ax * ax + ay * ay + az * az);
49 if (norm > 0.0f) {
50 ax /= norm;
51 ay /= norm;
52 az /= norm;
53
54 float roll = atan2f(ay, az);
55 float pitch = asinf(-ax);
56
57 float cy = cosf(roll * 0.5f);
58 float sy = sinf(roll * 0.5f);
59 float cp = cosf(pitch * 0.5f);
60 float sp = sinf(pitch * 0.5f);
61
62 q0 = cy * cp;
63 q1 = cy * sp;
64 q2 = sy * cp;
65 q3 = sy * sp;
66
67 filter->quaternion.q0 = q0;
68 filter->quaternion.q1 = q1;
69 filter->quaternion.q2 = q2;
70 filter->quaternion.q3 = q3;
71 filter->initialized = true;
72 }
73 return;
74 }
75
76 gx *= DEG_TO_RAD;
77 gy *= DEG_TO_RAD;
78 gz *= DEG_TO_RAD;
79
80 qDot1 = 0.5f * (-q1 * gx - q2 * gy - q3 * gz);
81 qDot2 = 0.5f * (q0 * gx + q2 * gz - q3 * gy);
82 qDot3 = 0.5f * (q0 * gy - q1 * gz + q3 * gx);
83 qDot4 = 0.5f * (q0 * gz + q1 * gy - q2 * gx);
84
85 if (!((ax == 0.0f) && (ay == 0.0f) && (az == 0.0f))) {
86 recipNorm = 1.0f / sqrtf(ax * ax + ay * ay + az * az);
87 ax *= recipNorm;
88 ay *= recipNorm;
89 az *= recipNorm;
90
91 _2q0 = 2.0f * q0;
92 _2q1 = 2.0f * q1;
93 _2q2 = 2.0f * q2;
94 _2q3 = 2.0f * q3;
95 _4q0 = 4.0f * q0;
96 _4q1 = 4.0f * q1;
97 _4q2 = 4.0f * q2;
98 _8q1 = 8.0f * q1;
99 _8q2 = 8.0f * q2;
100 q0q0 = q0 * q0;
101 q1q1 = q1 * q1;
102 q2q2 = q2 * q2;
103 q3q3 = q3 * q3;
104
105 s0 = _4q0 * q2q2 + _2q2 * ax + _4q0 * q1q1 - _2q1 * ay;
106 s1 = _4q1 * q3q3 - _2q3 * ax + 4.0f * q0q0 * q1 - _2q0 * ay - _4q1 + _8q1 * q1q1 + _8q1 * q2q2 + _4q1 * az;
107 s2 = 4.0f * q0q0 * q2 + _2q0 * ax + _4q2 * q3q3 - _2q3 * ay - _4q2 + _8q2 * q1q1 + _8q2 * q2q2 + _4q2 * az;
108 s3 = 4.0f * q1q1 * q3 - _2q1 * ax + 4.0f * q2q2 * q3 - _2q2 * ay;
109
110 float s_norm = sqrtf(s0 * s0 + s1 * s1 + s2 * s2 + s3 * s3);
111 if (s_norm > 0.0f) {
112 recipNorm = 1.0f / s_norm;
113 s0 *= recipNorm;
114 s1 *= recipNorm;
115 s2 *= recipNorm;
116 s3 *= recipNorm;
117 }
118
119 qDot1 -= filter->beta * s0;
120 qDot2 -= filter->beta * s1;
121 qDot3 -= filter->beta * s2;
122 qDot4 -= filter->beta * s3;
123 }
124
125 q0 += qDot1 * dt;
126 q1 += qDot2 * dt;
127 q2 += qDot3 * dt;
128 q3 += qDot4 * dt;
129
130 float q_norm = sqrtf(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3);
131 if (q_norm > 0.0f) {
132 recipNorm = 1.0f / q_norm;
133 q0 *= recipNorm;
134 q1 *= recipNorm;
135 q2 *= recipNorm;
136 q3 *= recipNorm;
137 }
138
139 filter->quaternion.q0 = q0;
140 filter->quaternion.q1 = q1;
141 filter->quaternion.q2 = q2;
142 filter->quaternion.q3 = q3;
143}

145esp_err_t lsm6ds3_madgwick_init(lsm6ds3_madgwick_filter_t *filter, float beta, float sample_rate_hz)
146{
147 if (filter == NULL) {
148 return ESP_ERR_INVALID_ARG;
149 }
150
151 // Validate beta parameter (typically 0.01 to 0.2)
152 if (beta < 0.0f || beta > 1.0f) {
153 return ESP_ERR_INVALID_ARG;
154 }
155
156 // Validate sample rate
157 if (sample_rate_hz <= 0.0f || sample_rate_hz > 10000.0f) {
158 return ESP_ERR_INVALID_ARG;
159 }
160
161 memset(filter, 0, sizeof(lsm6ds3_madgwick_filter_t));
162 filter->beta = beta;
163 filter->sample_rate_hz = sample_rate_hz;
164 filter->quaternion.q0 = 1.0f;
165 filter->initialized = false;
166
167 return ESP_OK;
168}

170esp_err_t lsm6ds3_madgwick_update(lsm6ds3_madgwick_filter_t *filter, const float accel[3], const float gyro[3], float dt)
171{
172 if (filter == NULL || accel == NULL || gyro == NULL) {
173 return ESP_ERR_INVALID_ARG;
174 }
175
176 // Validate dt to prevent division by zero or invalid updates
177 if (dt <= 0.0f || dt > 1.0f) {
178 return ESP_ERR_INVALID_ARG;
179 }
180
181 memcpy(filter->accel, accel, sizeof(float) * 3);
182 memcpy(filter->gyro, gyro, sizeof(float) * 3);
183
184 madgwick_ahrs_update(filter, gyro[0], gyro[1], gyro[2], accel[0], accel[1], accel[2], dt);
185
186 return ESP_OK;
187}

189esp_err_t lsm6ds3_madgwick_get_euler(lsm6ds3_madgwick_filter_t *filter, lsm6ds3_euler_angles_t *angles)
190{
191 if (filter == NULL || angles == NULL) {
192 return ESP_ERR_INVALID_ARG;
193 }
194
195 float q0 = filter->quaternion.q0;
196 float q1 = filter->quaternion.q1;
197 float q2 = filter->quaternion.q2;
198 float q3 = filter->quaternion.q3;
199
200 angles->roll = atan2f(q0 * q1 + q2 * q3, 0.5f - q1 * q1 - q2 * q2) * RAD_TO_DEG;
201
202 // Clamp argument to [-1, 1] to prevent NaN from asinf()
203 float pitch_arg = 2.0f * (q0 * q2 - q3 * q1);
204 pitch_arg = fmaxf(-1.0f, fminf(1.0f, pitch_arg));
205 angles->pitch = asinf(pitch_arg) * RAD_TO_DEG;
206
207 angles->yaw = atan2f(q0 * q3 + q1 * q2, 0.5f - q2 * q2 - q3 * q3) * RAD_TO_DEG;
208
209 return ESP_OK;
210}

236esp_err_t lsm6ds3_complementary_update(lsm6ds3_complementary_filter_t *filter, const float accel[3], const float gyro[3], float dt)
237{
238 if (filter == NULL || accel == NULL || gyro == NULL) {
239 return ESP_ERR_INVALID_ARG;
240 }
241
242 // Validate dt to prevent invalid updates
243 if (dt <= 0.0f || dt > 1.0f) {
244 return ESP_ERR_INVALID_ARG;
245 }
246
247 float accel_norm = sqrtf(accel[0] * accel[0] + accel[1] * accel[1] + accel[2] * accel[2]);
248
249 if (accel_norm > 0.0f) {
250 float accel_x = accel[0] / accel_norm;
251 float accel_y = accel[1] / accel_norm;
252 float accel_z = accel[2] / accel_norm;
253
254 float accel_roll = atan2f(accel_y, accel_z) * RAD_TO_DEG;
255 float accel_pitch = asinf(-accel_x) * RAD_TO_DEG;
256
257 if (!filter->initialized) {
258 filter->roll = accel_roll;
259 filter->pitch = accel_pitch;
260 filter->initialized = true;
261 } else {
262 filter->roll = filter->alpha * (filter->roll + gyro[0] * dt) + (1.0f - filter->alpha) * accel_roll;
263 filter->pitch = filter->alpha * (filter->pitch + gyro[1] * dt) + (1.0f - filter->alpha) * accel_pitch;
264 }
265 } else {
266 if (filter->initialized) {
267 filter->roll += gyro[0] * dt;
268 filter->pitch += gyro[1] * dt;
269 }
270 }
271
272 return ESP_OK;
273}

420float lsm6ds3_calculate_angle_from_vertical(float roll_deg, float pitch_deg)
421{
422 float roll_rad = roll_deg * DEG_TO_RAD;
423 float pitch_rad = pitch_deg * DEG_TO_RAD;
424
425 float cos_roll = cosf(roll_rad);
426 float cos_pitch = cosf(pitch_rad);
427 float cos_angle = cos_roll * cos_pitch;
428
429 cos_angle = fmaxf(-1.0f, fminf(1.0f, cos_angle));
430
431 float angle_rad = acosf(cos_angle);
432 float angle_deg = angle_rad * RAD_TO_DEG;
433
434 return angle_deg;
435}
