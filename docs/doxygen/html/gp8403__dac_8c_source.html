45// GP8403 I2C register addresses
46// Note: These are typical DAC register addresses. Actual addresses may vary
47// based on GP8403 datasheet. Adjust if needed after verifying datasheet.
48#define GP8403_REG_CHANNEL0_DATA 0x00 // Channel 0 data register (12-bit)
49#define GP8403_REG_CHANNEL1_DATA 0x01 // Channel 1 data register (12-bit)
50#define GP8403_REG_CONTROL 0x02 // Control register (if exists)

52// I2C transaction timeout (ms)
53#define GP8403_I2C_TIMEOUT_MS 100

58static inline float clamp_voltage(float voltage)
59{
60 if (voltage < GP8403_DAC_MIN_VOLTAGE) {
61 return GP8403_DAC_MIN_VOLTAGE;
62 }
63 if (voltage > GP8403_DAC_MAX_VOLTAGE) {
64 return GP8403_DAC_MAX_VOLTAGE;
65 }
66 return voltage;
67}

83static esp_err_t gp8403_write_register(gp8403_dac_handle_t *handle, uint8_t reg, uint16_t data)
84{
85 if (handle == NULL || !handle->initialized) {
86 ESP_LOGE(TAG, "Invalid handle or not initialized");
87 return ESP_ERR_INVALID_STATE;
88 }
89
90 // Prepare data: 12-bit value, typically sent as 2 bytes (MSB first)
91 // Format: [reg_addr] [data_high] [data_low]
92 // Data format: 12-bit value in bits [11:0], upper 4 bits typically unused
93 uint8_t write_data[3];
94 write_data[0] = reg;
95 write_data[1] = (data >> 8) & 0x0F; // Upper 4 bits of 12-bit value
96 write_data[2] = data & 0xFF; // Lower 8 bits
97
98 esp_err_t err = i2c_master_transmit(
99 handle->dev_handle,
100 write_data,
101 sizeof(write_data),
102 pdMS_TO_TICKS(GP8403_I2C_TIMEOUT_MS)
103 );
104
105 if (err != ESP_OK) {
106 ESP_LOGE(TAG, "I2C write failed: %s (reg=0x%02X, data=0x%03X)",
107 esp_err_to_name(err), reg, data);
108 return err;
109 }
110
111 ESP_LOGD(TAG, "Write reg=0x%02X, data=0x%03X", reg, data);
112 return ESP_OK;
113}

115bool gp8403_dac_init(const gp8403_dac_config_t *config, gp8403_dac_handle_t **handle)
116{
117 if (config == NULL || handle == NULL) {
118 ESP_LOGE(TAG, "Invalid parameters");
119 return false;
120 }
121
122 if (config->bus_handle == NULL) {
123 ESP_LOGE(TAG, "I2C bus handle is NULL");
124 return false;
125 }
126
127 // Validate I2C address (should be in range 0x58-0x5F for GP8403)
128 uint8_t i2c_addr = config->i2c_addr;
129 if (i2c_addr == 0) {
130 i2c_addr = GP8403_DAC_DEFAULT_I2C_ADDR;
131 }
132
133 if (i2c_addr < 0x58 || i2c_addr > 0x5F) {
134 ESP_LOGW(TAG, "I2C address 0x%02X may be out of valid range (0x58-0x5F)", i2c_addr);
135 }
136
137 // Allocate handle
138 gp8403_dac_handle_t *dev_handle = calloc(1, sizeof(gp8403_dac_handle_t));
139 if (dev_handle == NULL) {
140 ESP_LOGE(TAG, "Failed to allocate device handle");
141 return false;
142 }
143
144 // Configure I2C device
145 i2c_device_config_t dev_cfg = {
146 .dev_addr_length = I2C_ADDR_BIT_LEN_7,
147 .device_address = i2c_addr,
148 .scl_speed_hz = 100000, // 100kHz default, can be increased if needed
149 };
150
151 esp_err_t err = i2c_master_bus_add_device(
152 config->bus_handle,
153 &dev_cfg,
154 &dev_handle->dev_handle
155 );
156
157 if (err != ESP_OK) {
158 ESP_LOGE(TAG, "Failed to add I2C device: %s", esp_err_to_name(err));
159 free(dev_handle);
160 return false;
161 }
162
163 // Initialize handle
164 dev_handle->i2c_addr = i2c_addr;
165 dev_handle->initialized = true;
166 dev_handle->channel0_voltage = 0.0f;
167 dev_handle->channel1_voltage = 0.0f;
168
169 // Initialize both channels to 0V
170 gp8403_write_register(dev_handle, GP8403_REG_CHANNEL0_DATA, 0x000);
171 gp8403_write_register(dev_handle, GP8403_REG_CHANNEL1_DATA, 0x000);
172
173 *handle = dev_handle;
174
175 ESP_LOGI(TAG, "GP8403 DAC initialized at I2C address 0x%02X", i2c_addr);
176 return true;
177}

179void gp8403_dac_deinit(gp8403_dac_handle_t **handle)
180{
181 if (handle == NULL || *handle == NULL) {
182 return;
183 }
184
185 gp8403_dac_handle_t *dev_handle = *handle;
186
187 // Set both channels to 0V before deinitializing
188 if (dev_handle->initialized) {
189 gp8403_write_register(dev_handle, GP8403_REG_CHANNEL0_DATA, 0x000);
190 gp8403_write_register(dev_handle, GP8403_REG_CHANNEL1_DATA, 0x000);
191 }
192
193 // Remove I2C device
194 if (dev_handle->dev_handle != NULL) {
195 i2c_master_bus_rm_device(dev_handle->dev_handle);
196 }
197
198 // Free handle
199 free(dev_handle);
200 *handle = NULL;
201
202 ESP_LOGI(TAG, "GP8403 DAC deinitialized");
203}

205bool gp8403_dac_set_voltage(gp8403_dac_handle_t *handle, gp8403_channel_t channel, float voltage)
206{
207 if (handle == NULL || !handle->initialized) {
208 ESP_LOGE(TAG, "Invalid handle or not initialized");
209 return false;
210 }
211
212 if (channel >= GP8403_CHANNEL_MAX) {
213 ESP_LOGE(TAG, "Invalid channel: %d", channel);
214 return false;
215 }
216
217 // Clamp voltage to valid range
218 float clamped_voltage = clamp_voltage(voltage);
219 if (clamped_voltage != voltage) {
220 ESP_LOGW(TAG, "Voltage %.3fV clamped to %.3fV", voltage, clamped_voltage);
221 }
222
223 // Convert voltage to raw DAC value
224 uint16_t dac_value = gp8403_dac_voltage_to_raw(clamped_voltage);
225
226 // Select register based on channel
227 uint8_t reg = (channel == GP8403_CHANNEL_0) ? GP8403_REG_CHANNEL0_DATA : GP8403_REG_CHANNEL1_DATA;
228
229 // Write to DAC
230 esp_err_t err = gp8403_write_register(handle, reg, dac_value);
231 if (err != ESP_OK) {
232 return false;
233 }
234
235 // Update cached voltage
236 if (channel == GP8403_CHANNEL_0) {
237 handle->channel0_voltage = clamped_voltage;
238 } else {
239 handle->channel1_voltage = clamped_voltage;
240 }
241
242 ESP_LOGD(TAG, "Channel %d set to %.3fV (DAC=0x%03X)", channel, clamped_voltage, dac_value);
243 return true;
244}

325uint16_t gp8403_dac_voltage_to_raw(float voltage)
326{
327 float clamped = clamp_voltage(voltage);
328
329 // Convert voltage (0-10V) to 12-bit value (0x000-0xFFF)
330 // Formula: dac_value = (voltage / 10.0) 4095
331 float normalized = clamped / GP8403_DAC_MAX_VOLTAGE;
332 uint16_t dac_value = (uint16_t)roundf(normalized * GP8403_DAC_MAX_VALUE);
333
334 return clamp_dac_value(dac_value);
335}
336
337float gp8403_dac_raw_to_voltage(uint16_t dac_value)
338{
339 uint16_t clamped = clamp_dac_value(dac_value);
340
341 // Convert 12-bit value (0x000-0xFFF) to voltage (0-10V)
342 // Formula: voltage = (dac_value / 4095) 10.0
343 float normalized = (float)clamped / GP8403_DAC_MAX_VALUE;
344 return normalized * GP8403_DAC_MAX_VOLTAGE;
345}
