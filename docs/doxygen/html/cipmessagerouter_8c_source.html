51void InitializeCipMessageRouterClass(CipClass *cip_class) {
52
53 CipClass *meta_class = cip_class->class_instance.cip_class;
54
55 InsertAttribute( (CipInstance *) cip_class, 1, kCipUint, EncodeCipUint, NULL,
56 (void *) &cip_class->revision, kGetableSingleAndAll ); /* revision */
57 InsertAttribute( (CipInstance *) cip_class, 2, kCipUint, EncodeCipUint, NULL,
58 (void *) &cip_class->number_of_instances, kGetableSingle ); /* largest instance number */
59 InsertAttribute( (CipInstance *) cip_class, 3, kCipUint, EncodeCipUint, NULL,
60 (void *) &cip_class->number_of_instances, kGetableSingle ); /* number of instances currently existing*/
61 InsertAttribute( (CipInstance *) cip_class, 4, kCipUint, EncodeCipUint, NULL,
62 (void *) &kCipUintZero, kGetableAll ); /* optional attribute list - default = 0 */
63 InsertAttribute( (CipInstance *) cip_class, 5, kCipUint, EncodeCipUint, NULL,
64 (void *) &kCipUintZero, kGetableAll ); /* optional service list - default = 0 */
65 InsertAttribute( (CipInstance *) cip_class, 6, kCipUint, EncodeCipUint, NULL,
66 (void *) &meta_class->highest_attribute_number,
67 kGetableSingleAndAll ); /* max class attribute number*/
68 InsertAttribute( (CipInstance *) cip_class, 7, kCipUint, EncodeCipUint, NULL,
69 (void *) &cip_class->highest_attribute_number,
70 kGetableSingleAndAll ); /* max instance attribute number*/
71
72 InsertService(meta_class,
73 kGetAttributeAll,
74 &GetAttributeAll,
75 "GetAttributeAll"); /* bind instance services to the metaclass*/
76 InsertService(meta_class,
77 kGetAttributeSingle,
78 &GetAttributeSingle,
79 "GetAttributeSingle");
80}

158EipStatus RegisterCipClass(CipClass *cip_class) {
159 CipMessageRouterObject **message_router_object = &g_first_object;
160
161 while(*message_router_object) {
162 message_router_object = &(*message_router_object)->next; /* follow the list until p points to an empty link (list end)*/
163
164 }
165 *message_router_object =
166 (CipMessageRouterObject *) CipCalloc(1, sizeof(CipMessageRouterObject) ); /* create a new node at the end of the list*/
167 if(*message_router_object == 0) {
168 return kEipStatusError; /* check for memory error*/
169
170 }
171 (*message_router_object)->cip_class = cip_class; /* fill in the new node*/
172 (*message_router_object)->next = NULL;
173
174 return kEipStatusOk;
175}

177EipStatus NotifyMessageRouter(EipUint8 *data,
178 int data_length,
179 CipMessageRouterResponse *message_router_response,
180 const struct sockaddr *const originator_address,
181 const CipSessionHandle encapsulation_session) {
182 EipStatus eip_status = kEipStatusOkSend;
183 CipError status = kCipErrorSuccess;
184
185 OPENER_TRACE_INFO("NotifyMessageRouter: routing unconnected message\n");
186 if(kCipErrorSuccess !=
187 (status =
188 CreateMessageRouterRequestStructure(data, data_length,
189 &g_message_router_request) ) ) { /* error from create MR structure*/
190 OPENER_TRACE_ERR(
191 "NotifyMessageRouter: error from createMRRequeststructure\n");
192 message_router_response->general_status = status;
193 message_router_response->size_of_additional_status = 0;
194 message_router_response->reserved = 0;
195 message_router_response->reply_service =
196 (0x80 | g_message_router_request.service);
197 } else {
198 /* forward request to appropriate Object if it is registered*/
199 CipMessageRouterObject *registered_object = GetRegisteredObject(
200 g_message_router_request.request_path.class_id);
201 if(registered_object == 0) {
202 OPENER_TRACE_ERR(
203 "NotifyMessageRouter: sending CIP_ERROR_OBJECT_DOES_NOT_EXIST reply, class id 0x%x is not registered\n",
204 (unsigned ) g_message_router_request.request_path.class_id);
205 message_router_response->general_status = kCipErrorPathDestinationUnknown; /*according to the test tool this should be the correct error flag instead of CIP_ERROR_OBJECT_DOES_NOT_EXIST;*/
206 message_router_response->size_of_additional_status = 0;
207 message_router_response->reserved = 0;
208 message_router_response->reply_service =
209 (0x80 | g_message_router_request.service);
210 } else {
211 /* call notify function from Object with ClassID (gMRRequest.RequestPath.ClassID)
212 object will or will not make an reply into gMRResponse*/
213 message_router_response->reserved = 0;
214 OPENER_ASSERT(NULL != registered_object->cip_class); OPENER_TRACE_INFO(
215 "NotifyMessageRouter: calling notify function of class '%s'\n",
216 registered_object->cip_class->class_name);
217 eip_status = NotifyClass(registered_object->cip_class,
218 &g_message_router_request,
219 message_router_response,
220 originator_address,
221 encapsulation_session);
222
223#ifdef OPENER_TRACE_ENABLED
224 if (eip_status == kEipStatusError) {
225 OPENER_TRACE_ERR(
226 "notifyMR: notify function of class '%s' returned an error\n",
227 registered_object->cip_class->class_name);
228 } else if (eip_status == kEipStatusOk) {
229 OPENER_TRACE_INFO(
230 "notifyMR: notify function of class '%s' returned no reply\n",
231 registered_object->cip_class->class_name);
232 } else {
233 OPENER_TRACE_INFO(
234 "notifyMR: notify function of class '%s' returned a reply\n",
235 registered_object->cip_class->class_name);
236 }
237#endif
238 }
239 }
240 return eip_status;
241}

243CipError CreateMessageRouterRequestStructure(const EipUint8 *data,
244 EipInt16 data_length,
245 CipMessageRouterRequest *message_router_request)
246{
247
248 message_router_request->service = *data;
249 data++;
250 data_length--;
251
252 size_t number_of_decoded_bytes;
253 const EipStatus path_result =
254 DecodePaddedEPath(&(message_router_request->request_path),
255 &data,
256 &number_of_decoded_bytes);
257 if(path_result != kEipStatusOk) {
258 return kCipErrorPathSegmentError;
259 }
260
261 if(number_of_decoded_bytes > data_length) {
262 return kCipErrorPathSizeInvalid;
263 } else {
264 message_router_request->data = data;
265 message_router_request->request_data_size = data_length -
266 number_of_decoded_bytes;
267 return kCipErrorSuccess;
268 }

271void DeleteAllClasses(void) {
272 CipMessageRouterObject *message_router_object = g_first_object; /* get pointer to head of class registration list */
273 CipMessageRouterObject *message_router_object_to_delete = NULL;
274 CipInstance *instance = NULL;
275 CipInstance *instance_to_delete = NULL;
276
277 while(NULL != message_router_object) {
278 message_router_object_to_delete = message_router_object;
279 message_router_object = message_router_object->next;
280
281 instance = message_router_object_to_delete->cip_class->instances;
282 while(NULL != instance) {
283 instance_to_delete = instance;
284 instance = instance->next;
285 if(message_router_object_to_delete->cip_class->number_of_attributes) /* if the class has instance attributes */
286 { /* then free storage for the attribute array */
287 CipFree(instance_to_delete->attributes);
288 }
289 CipFree(instance_to_delete);
290 }
291
292 /* free meta class data*/
293 CipClass *meta_class =
294 message_router_object_to_delete->cip_class->class_instance.cip_class;
295 CipFree(meta_class->class_name);
296 CipFree(meta_class->services);
297 CipFree(meta_class->get_single_bit_mask);
298 CipFree(meta_class->set_bit_mask);
299 CipFree(meta_class->get_all_bit_mask);
300 CipFree(meta_class);
301
302 /* free class data*/
303 CipClass *cip_class = message_router_object_to_delete->cip_class;
304 CipFree(cip_class->class_name);
305 CipFree(cip_class->get_single_bit_mask);
306 CipFree(cip_class->set_bit_mask);
307 CipFree(cip_class->get_all_bit_mask);
308 CipFree(cip_class->class_instance.attributes);
309 CipFree(cip_class->services);
310 CipFree(cip_class);
311 /* free message router object */
312 CipFree(message_router_object_to_delete);
313 }
314 g_first_object = NULL;
315}
