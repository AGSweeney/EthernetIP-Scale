153// POST /api/ota/update - Trigger OTA update (supports both URL and file upload)
154static esp_err_t api_ota_update_handler(httpd_req_t *req)
155{
156 ESP_LOGI(TAG, "OTA update request received");
157
158 // Check content type
159 char content_type[256];
160 esp_err_t ret = httpd_req_get_hdr_value_str(req, "Content-Type", content_type, sizeof(content_type));
161 if (ret != ESP_OK) {
162 ESP_LOGE(TAG, "Missing Content-Type header (ret=%d)", ret);
163 return send_json_error(req, "Missing Content-Type", 400);
164 }
165
166 ESP_LOGI(TAG, "OTA update request, Content-Type: %s", content_type);
167
168 // Handle file upload (multipart/form-data) - Use streaming to avoid memory issues
169 if (strstr(content_type, "multipart/form-data") != NULL) {
170 // Get content length - may be 0 if chunked transfer
171 size_t content_len = req->content_len;
172 ESP_LOGI(TAG, "Content-Length: %d", content_len);
173
174 // Validate size
175 if (content_len > 2 * 1024 * 1024) { // Max 2MB for safety
176 ESP_LOGW(TAG, "Content length too large: %d", content_len);
177 return send_json_error(req, "File too large (max 2MB)", 400);
178 }
179
180 // Parse multipart boundary first
181 const char *boundary_str = strstr(content_type, "boundary=");
182 if (boundary_str == NULL) {
183 ESP_LOGW(TAG, "No boundary found in Content-Type");
184 return send_json_error(req, "Invalid multipart data: no boundary", 400);
185 }
186 boundary_str += 9; // Skip "boundary="
187
188 // Extract boundary value
189 char boundary[128];
190 int boundary_len = 0;
191 while (*boundary_str && *boundary_str != ';' && *boundary_str != ' ' && *boundary_str != '\r' && *boundary_str != '\n' && boundary_len < 127) {
192 boundary[boundary_len++] = *boundary_str++;
193 }
194 boundary[boundary_len] = '\0';
195 ESP_LOGI(TAG, "Multipart boundary: %s", boundary);
196
197 // Use a small buffer to read multipart headers (64KB should be enough)
198 const size_t header_buffer_size = 64 * 1024;
199 char *header_buffer = malloc(header_buffer_size);
200 if (header_buffer == NULL) {
201 ESP_LOGE(TAG, "Failed to allocate memory for header buffer");
202 return send_json_error(req, "Failed to allocate memory", 500);
203 }
204
205 // Read enough to find the data separator (\r\n\r\n)
206 size_t header_read = 0;
207 bool found_separator = false;
208 while (header_read < header_buffer_size - 1) {
209 int ret = httpd_req_recv(req, header_buffer + header_read, header_buffer_size - header_read - 1);
210 if (ret <= 0) {
211 if (ret == HTTPD_SOCK_ERR_TIMEOUT) {
212 continue;
213 }
214 ESP_LOGE(TAG, "Error reading headers: %d", ret);
215 free(header_buffer);
216 return send_json_error(req, "Failed to read request headers", 500);
217 }
218 header_read += ret;
219 header_buffer[header_read] = '\0'; // Null terminate for string search
220
221 // Look for data separator
222 if (strstr(header_buffer, "\r\n\r\n") != NULL || strstr(header_buffer, "\n\n") != NULL) {
223 found_separator = true;
224 break;
225 }
226 }
227
228 if (!found_separator) {
229 ESP_LOGW(TAG, "Could not find data separator in multipart headers");
230 free(header_buffer);
231 return send_json_error(req, "Invalid multipart format: no data separator", 400);
232 }
233
234 // Find where data starts
235 char *data_start = strstr(header_buffer, "\r\n\r\n");
236 size_t header_len = 0;
237 if (data_start != NULL) {
238 header_len = (data_start - header_buffer) + 4;
239 } else {
240 data_start = strstr(header_buffer, "\n\n");
241 if (data_start != NULL) {
242 header_len = (data_start - header_buffer) + 2;
243 } else {
244 free(header_buffer);
245 return send_json_error(req, "Invalid multipart format", 400);
246 }
247 }
248
249 // Calculate how much data we already have in the buffer
250 size_t data_in_buffer = header_read - header_len;
251
252 // Start streaming OTA update
253 // Estimate firmware size: Content-Length minus multipart headers (typically ~1KB)
254 // This gives us a reasonable estimate for progress tracking
255 size_t estimated_firmware_size = 0;
256 if (content_len > 0) {
257 // Subtract estimated multipart header overhead (boundary + headers ~1KB)
258 estimated_firmware_size = (content_len > 1024) ? (content_len - 1024) : content_len;
259 }
260 esp_ota_handle_t ota_handle = ota_manager_start_streaming_update(estimated_firmware_size);
261 if (ota_handle == 0) {
262 ESP_LOGE(TAG, "Failed to start streaming OTA update - check serial logs for details");
263 free(header_buffer);
264 return send_json_error(req, "Failed to start OTA update. Check device logs for details.", 500);
265 }
266
267 // Prepare boundary strings for detection
268 char start_boundary[256];
269 char end_boundary[256];
270 snprintf(start_boundary, sizeof(start_boundary), "--%s", boundary);
271 snprintf(end_boundary, sizeof(end_boundary), "--%s--", boundary);
272
273 bool done = false;
274
275 // Write data we already have in buffer (check for boundary first)
276 if (data_in_buffer > 0) {
277 // Check if boundary is already in the initial data
278 char *boundary_in_header = strstr((char *)(header_buffer + header_len), start_boundary);
279
280 if (boundary_in_header != NULL) {
281 // Boundary found in initial data - only write up to it
282 size_t initial_to_write = boundary_in_header - (header_buffer + header_len);
283 // Remove trailing \r\n
284 while (initial_to_write > 0 &&
285 (header_buffer[header_len + initial_to_write - 1] == '\r' ||
286 header_buffer[header_len + initial_to_write - 1] == '\n')) {
287 initial_to_write--;
288 }
289 if (initial_to_write > 0) {
290 if (!ota_manager_write_streaming_chunk(ota_handle, (const uint8_t *)(header_buffer + header_len), initial_to_write)) {
291 ESP_LOGE(TAG, "Failed to write initial chunk");
292 free(header_buffer);
293 return send_json_error(req, "Failed to write firmware data", 500);
294 }
295 data_in_buffer = initial_to_write;
296 } else {
297 data_in_buffer = 0; // No data to write, boundary was at start
298 }
299 done = true; // We're done, boundary found
300 } else {
301 // No boundary in initial data, write it all
302 if (!ota_manager_write_streaming_chunk(ota_handle, (const uint8_t *)(header_buffer + header_len), data_in_buffer)) {
303 ESP_LOGE(TAG, "Failed to write initial chunk");
304 free(header_buffer);
305 return send_json_error(req, "Failed to write firmware data", 500);
306 }
307 }
308 }
309
310 free(header_buffer); // Free header buffer, we'll use a smaller chunk buffer now
311
312 // If boundary was found in initial data, we're done
313 if (done) {
314 ESP_LOGI(TAG, "Streamed %d bytes to OTA partition", data_in_buffer);
315
316 // Finish OTA update
317 cJSON *response = cJSON_CreateObject();
318 cJSON_AddStringToObject(response, "status", "ok");
319 cJSON_AddStringToObject(response, "message", "Firmware uploaded successfully. Finishing update and rebooting...");
320
321 esp_err_t resp_err = send_json_response(req, response, ESP_OK);
322 vTaskDelay(pdMS_TO_TICKS(100));
323
324 if (!ota_manager_finish_streaming_update(ota_handle)) {
325 ESP_LOGE(TAG, "Failed to finish streaming OTA update");
326 return ESP_FAIL;
327 }
328 return resp_err;
329 }
330
331 // Stream remaining data in chunks (64KB chunks)
332 const size_t chunk_size = 64 * 1024;
333 char *chunk_buffer = malloc(chunk_size);
334 if (chunk_buffer == NULL) {
335 ESP_LOGE(TAG, "Failed to allocate chunk buffer");
336 return send_json_error(req, "Failed to allocate memory", 500);
337 }
338
339 size_t total_written = data_in_buffer;
340
341 while (!done) {
342 int ret = httpd_req_recv(req, chunk_buffer, chunk_size);
343 if (ret <= 0) {
344 if (ret == HTTPD_SOCK_ERR_TIMEOUT) {
345 continue;
346 }
347 // Connection closed or error
348 done = true;
349 break;
350 }
351
352 // Search for boundary markers in this chunk
353 // Look for both start boundary (--boundary) and end boundary (--boundary--)
354 char *boundary_pos = NULL;
355 size_t to_write = ret;
356 bool found_end = false;
357
358 // First check for end boundary (--boundary--)
359 char *search_start = chunk_buffer;
360 while ((search_start = strstr(search_start, end_boundary)) != NULL) {
361 // Check if this is actually a boundary (should be preceded by \r\n or at start)
362 if (search_start == chunk_buffer ||
363 (search_start > chunk_buffer &&
364 (search_start[-1] == '\n' || (search_start[-1] == '\r' && search_start > chunk_buffer + 1 && search_start[-2] == '\n')))) {
365 boundary_pos = search_start;
366 found_end = true;
367 break;
368 }
369 search_start++;
370 }
371
372 // If not found, check for start boundary (--boundary) - this indicates next part
373 if (boundary_pos == NULL) {
374 search_start = chunk_buffer;
375 while ((search_start = strstr(search_start, start_boundary)) != NULL) {
376 // Check if this is actually a boundary (should be preceded by \r\n or at start)
377 if (search_start == chunk_buffer ||
378 (search_start > chunk_buffer &&
379 (search_start[-1] == '\n' || (search_start[-1] == '\r' && search_start > chunk_buffer + 1 && search_start[-2] == '\n')))) {
380 // Make sure it's not the end boundary (which is longer)
381 size_t end_boundary_len = strlen(end_boundary);
382 if (strncmp(search_start, end_boundary, end_boundary_len) != 0) {
383 boundary_pos = search_start;
384 break;
385 }
386 }
387 search_start++;
388 }
389 }
390
391 if (boundary_pos != NULL) {
392 // Found boundary - only write up to it (excluding the boundary itself and leading \r\n)
393 to_write = boundary_pos - chunk_buffer;
394
395 // Remove any trailing \r\n before the boundary (multipart boundaries are preceded by \r\n)
396 while (to_write > 0 && (chunk_buffer[to_write - 1] == '\r' || chunk_buffer[to_write - 1] == '\n')) {
397 to_write--;
398 }
399
400 // Also check if there's a \r\n sequence before boundary_pos that we should exclude
401 if (to_write >= 2 && chunk_buffer[to_write - 2] == '\r' && chunk_buffer[to_write - 1] == '\n') {
402 to_write -= 2;
403 } else if (to_write >= 1 && chunk_buffer[to_write - 1] == '\n') {
404 to_write -= 1;
405 }
406
407 done = true;
408 }
409
410 if (to_write > 0) {
411 if (!ota_manager_write_streaming_chunk(ota_handle, (const uint8_t *)chunk_buffer, to_write)) {
412 ESP_LOGE(TAG, "Failed to write chunk at offset %d", total_written);
413 free(chunk_buffer);
414 return send_json_error(req, "Failed to write firmware data", 500);
415 }
416 total_written += to_write;
417 }
418
419 // If we found the end boundary, we're done
420 if (found_end) {
421 done = true;
422 }
423 }
424
425 free(chunk_buffer);
426
427 ESP_LOGI(TAG, "Streamed %d bytes to OTA partition", total_written);
428
429 // Finish OTA update (this will set boot partition and reboot)
430 // Send HTTP response BEFORE finishing, as the device will reboot
431 cJSON *response = cJSON_CreateObject();
432 cJSON_AddStringToObject(response, "status", "ok");
433 cJSON_AddStringToObject(response, "message", "Firmware uploaded successfully. Finishing update and rebooting...");
434
435 // Send response first
436 esp_err_t resp_err = send_json_response(req, response, ESP_OK);
437
438 // Small delay to ensure response is sent
439 vTaskDelay(pdMS_TO_TICKS(100));
440
441 // Now finish the update (this will reboot)
442 if (!ota_manager_finish_streaming_update(ota_handle)) {
443 ESP_LOGE(TAG, "Failed to finish streaming OTA update");
444 // Response already sent, but update failed - device will not reboot
445 return ESP_FAIL;
446 }
447
448 // This should never be reached as ota_manager_finish_streaming_update() reboots
449 return resp_err;
450 }
451
452 // Handle URL-based update (existing JSON method)
453 if (strstr(content_type, "application/json") == NULL) {
454 ESP_LOGW(TAG, "Unsupported Content-Type for OTA update: %s", content_type);
455 return send_json_error(req, "Unsupported Content-Type. Use multipart/form-data for file upload or application/json for URL", 400);
456 }
457
458 char content[256];
459 int bytes_received = httpd_req_recv(req, content, sizeof(content) - 1);
460 if (bytes_received <= 0) {
461 ESP_LOGE(TAG, "Failed to read request body");
462 return send_json_error(req, "Failed to read request body", 500);
463 }
464 content[bytes_received] = '\0';
465
466 cJSON *json = cJSON_Parse(content);
467 if (json == NULL) {
468 ESP_LOGW(TAG, "Invalid JSON in request");
469 return send_json_error(req, "Invalid JSON", 400);
470 }
471
472 cJSON *item = cJSON_GetObjectItem(json, "url");
473 if (item == NULL || !cJSON_IsString(item)) {
474 cJSON_Delete(json);
475 return send_json_error(req, "Missing or invalid URL", 400);
476 }
477
478 const char *url = cJSON_GetStringValue(item);
479 if (url == NULL) {
480 cJSON_Delete(json);
481 return send_json_error(req, "Invalid URL", 400);
482 }
483 cJSON_Delete(json);
484
485 ESP_LOGI(TAG, "Starting OTA update from URL: %s", url);
486 bool success = ota_manager_start_update(url);
487
488 cJSON *response = cJSON_CreateObject();
489 if (success) {
490 cJSON_AddStringToObject(response, "status", "ok");
491 cJSON_AddStringToObject(response, "message", "OTA update started");
492 } else {
493 cJSON_AddStringToObject(response, "status", "error");
494 cJSON_AddStringToObject(response, "message", "Failed to start OTA update");
495 }
496
497 return send_json_response(req, response, success ? ESP_OK : ESP_FAIL);
498}

618// GET /api/status - Get assembly data for status pages
619static esp_err_t api_get_status_handler(httpd_req_t *req)
620{
621 SemaphoreHandle_t mutex = sample_application_get_assembly_mutex();
622 if (mutex == NULL) {
623 return send_json_error(req, "Assembly mutex not available", 500);
624 }
625
626 if (xSemaphoreTake(mutex, pdMS_TO_TICKS(1000)) != pdTRUE) {
627 return send_json_error(req, "Failed to acquire assembly mutex", 500);
