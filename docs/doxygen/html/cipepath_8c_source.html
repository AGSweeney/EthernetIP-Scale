19/*** Path Segment ***/
20SegmentType GetPathSegmentType(const CipOctet *const cip_path) {
21 const unsigned int kSegmentTypeMask = 0xE0;
22 const unsigned int segment_type = *cip_path & kSegmentTypeMask;
23 SegmentType result = kSegmentTypeInvalid;
24 switch(segment_type) {
25 case SEGMENT_TYPE_PORT_SEGMENT:
26 result = kSegmentTypePortSegment;
27 break;
28 case SEGMENT_TYPE_LOGICAL_SEGMENT:
29 result = kSegmentTypeLogicalSegment;
30 break;
31 case SEGMENT_TYPE_NETWORK_SEGMENT:
32 result = kSegmentTypeNetworkSegment;
33 break;
34 case SEGMENT_TYPE_SYMBOLIC_SEGMENT:
35 result = kSegmentTypeSymbolicSegment;
36 break;
37 case SEGMENT_TYPE_DATA_SEGMENT:
38 result = kSegmentTypeDataSegment;
39 break;
40 case SEGMENT_TYPE_DATA_TYPE_CONSTRUCTED:
41 result = kSegmentTypeDataTypeConstructed;
42 break;
43 case SEGMENT_TYPE_DATA_TYPE_ELEMENTARTY:
44 result = kSegmentTypeDataTypeElementary;
45 break;
46 case SEGMENT_TYPE_SEGMENT_RESERVED:
47 result = kSegmentTypeReserved;
48 break;
49 default:
50 OPENER_TRACE_ERR(
51 "Invalid Segment type in the message! We should never come here!\n");
52 break;
53 }
54 return result;
55}

132unsigned int GetPathPortSegmentExtendedPortNumber(
133 const unsigned char *const cip_path) {
134 /* OPENER_ASSERT(kPortSegmentExtendedPort == GetPathPortSegmentPortIdentifier(cip_path),
135 "There is no extended port available!\n");*/
136 OPENER_ASSERT(kPortSegmentExtendedPort ==
137 GetPathPortSegmentPortIdentifier(cip_path) );
138 const unsigned int kExtendedPortSegmentPosition =
139 GetPathPortSegmentExtendedLinkAddressSizeBit(cip_path) == true ? 2 : 1;
140 return cip_path[kExtendedPortSegmentPosition] +
141 (cip_path[kExtendedPortSegmentPosition + 1] << 8);
142}

280CipDword CipEpathGetLogicalValue(const EipUint8 **message) {
281 LogicalSegmentLogicalFormat logical_format =
282 GetPathLogicalSegmentLogicalFormat(*message);
283 CipDword data = kLogicalSegmentLogicalFormatInvalid;
284 (*message) += 1; /* Move to logical value */
285 switch(logical_format) {
286 case kLogicalSegmentLogicalFormatEightBit:
287 data = GetByteFromMessage(message);
288 break;
289 case kLogicalSegmentLogicalFormatSixteenBit:
290 (*message) += 1; /* Pad byte needs to be skipped */
291 data = GetWordFromMessage(message);
292 break;
293 case kLogicalSegmentLogicalFormatThirtyTwoBit:
294 (*message) += 1; /* Pad byte needs to be skipped */
295 data = GetDwordFromMessage(message);
296 break;
297 default:
298 OPENER_TRACE_ERR(
299 "Logical segment/logical type: Invalid logical value detected!\n");
300 break;
301 }
302 return data;
303}

305void CipEpathSetLogicalValue(const CipDword logical_value,
306 const LogicalSegmentLogicalFormat logical_format,
307 CipMessageRouterResponse *const message) {
308 switch(logical_format) {
309 case kLogicalSegmentLogicalFormatEightBit:
310 OPENER_ASSERT( (logical_value <= UINT8_MAX) ); /* Sanity check before casting to a smaller integer. */
311 AddSintToMessage( (EipUint8)logical_value, &message->message );
312 break;
313 case kLogicalSegmentLogicalFormatSixteenBit:
314 MoveMessageNOctets(1, &message->message); /* Needed for padding */
315 OPENER_ASSERT( (logical_value <= UINT16_MAX) ); /* Sanity check before casting to a smaller integer. */
316 AddIntToMessage( (EipUint16)logical_value, &message->message );
317 break;
318 case kLogicalSegmentLogicalFormatThirtyTwoBit:
319 MoveMessageNOctets(1, &message->message); /* Needed for padding */
320 AddDintToMessage(logical_value, &message->message);
321 break;
322 default:
323 OPENER_ASSERT(false); /* This should never happen! */
324 break;
325 }
326}
