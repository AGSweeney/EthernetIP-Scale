40static esp_err_t root_handler(httpd_req_t *req)
41{
42 const char *html = webui_get_index_html();
43
44 // Calculate actual length by scanning for null terminator
45 // This handles cases where strlen() might stop early due to embedded nulls
46 size_t html_len = 0;
47 const char *p = html;
48 while (*p != '\0' && html_len < 200000) { // Safety limit of 200KB
49 html_len++;
50 p++;
51 }
52
53 ESP_LOGI(TAG, "HTML string length: %zu bytes", html_len);
54
55 // Verify the string ends with </html>
56 const char *html_end = "</html>";
57 size_t html_end_len = strlen(html_end);
58 if (html_len < html_end_len || strncmp(html + html_len - html_end_len, html_end, html_end_len) != 0) {
59 ESP_LOGW(TAG, "HTML string may be truncated! Length: %zu, Last 30 bytes: %.30s", html_len, html + (html_len > 30 ? html_len - 30 : 0));
60 } else {
61 ESP_LOGI(TAG, "HTML string appears complete, ends with </html>");
62 }
63
64 // Check if we hit the safety limit (indicates potential issue)
65 if (html_len >= 200000) {
66 ESP_LOGE(TAG, "HTML string appears to be too long or unterminated!");
67 return ESP_ERR_INVALID_SIZE;
68 }
69
70 // Check for null bytes in the string (which would cause strlen to stop early)
71 for (size_t i = 0; i < html_len && i < 50000; i++) {
72 if (html[i] == '\0' && i < html_len - 1) {
73 ESP_LOGW(TAG, "Found null byte at position %zu (before end of string)", i);
74 break;
75 }
76 }
77
78 httpd_resp_set_type(req, "text/html; charset=utf-8");
79
80 // Use chunked transfer encoding for large responses
81 // This avoids buffer size limits in httpd_resp_send
82 httpd_resp_set_hdr(req, "Transfer-Encoding", "chunked");
83
84 const size_t chunk_size = 4096;
85 size_t sent = 0;
86 esp_err_t ret = ESP_OK;
87
88 while (sent < html_len && ret == ESP_OK) {
89 size_t to_send = (html_len - sent > chunk_size) ? chunk_size : (html_len - sent);
90 ret = httpd_resp_send_chunk(req, html + sent, to_send);
91 if (ret != ESP_OK) {
92 ESP_LOGE(TAG, "Failed to send HTML chunk at offset %zu/%zu: %s", sent, html_len, esp_err_to_name(ret));
93 return ret;
94 }
95 sent += to_send;
96 }
97
98 // Finalize chunked transfer
99 ret = httpd_resp_send_chunk(req, NULL, 0);
100 if (ret != ESP_OK) {
101 ESP_LOGE(TAG, "Failed to finalize chunked transfer: %s", esp_err_to_name(ret));
102 return ret;
103 }
104
105 ESP_LOGI(TAG, "Successfully sent %zu bytes of HTML using chunked encoding", sent);
106 return ESP_OK;
107}

152bool webui_init(void)
153{
154 if (server_handle != NULL) {
155 ESP_LOGW(TAG, "Web UI server already initialized");
156 return true;
157 }
158
159 httpd_config_t config = HTTPD_DEFAULT_CONFIG();
160 config.server_port = 80;
161 config.max_uri_handlers = 35; // Increased to accommodate all API endpoints (currently 34 handlers: 4 HTML + 30 API)
162 config.max_open_sockets = 7;
163 config.stack_size = 20480; // Increased to 20KB for large HTML pages and file uploads
164 config.task_priority = 5;
165 config.core_id = 1; // Run on Core 1 with sensor task
166 config.max_req_hdr_len = 2048; // Increased to 2KB for larger multipart headers
167
168 ESP_LOGI(TAG, "Starting HTTP server on port %d", config.server_port);
169
170 // Note: TCP_NODELAY (Nagle's algorithm disabled) would improve Web API responsiveness
171 // but ESP-IDF httpd manages sockets internally and doesn't expose a socket callback.
172 // To enable TCP_NODELAY for HTTP server, use an lwIP hook or patch ESP-IDF httpd component.
173 // For now, TCP_NODELAY is enabled for Modbus TCP and EtherNet/IP (critical services).
174
175 if (httpd_start(&server_handle, &config) == ESP_OK) {
176 ESP_LOGI(TAG, "HTTP server started");
177
178 // Register URI handlers
179 httpd_register_uri_handler(server_handle, &root_uri);
180 httpd_register_uri_handler(server_handle, &ota_uri);
181 httpd_register_uri_handler(server_handle, &favicon_uri);
182
183 // Register API handlers
184 webui_register_api_handlers(server_handle);
185
186 return true;
187 }
188
189 ESP_LOGE(TAG, "Failed to start HTTP server");
190 return false;
191}

199 }
