218void CheckIoConnectionEvent(unsigned int output_assembly_id,
219 unsigned int input_assembly_id,
220 IoConnectionEvent io_connection_event) {
221
222 (void) output_assembly_id;
223 (void) input_assembly_id;
224
225 switch (io_connection_event) {
226 case kIoConnectionEventOpened:
227 if (s_active_io_connections++ == 0) {
228 IdentityEnter(kStateStandby,
229 kAtLeastOneIoConnectionEstablishedAllInIdleMode);
230 }
231 break;
232 case kIoConnectionEventTimedOut:
233 case kIoConnectionEventClosed:
234 if (s_active_io_connections > 0) {
235 s_active_io_connections--;
236 }
237 if (s_active_io_connections == 0) {
238 s_io_activity_seen = false;
239 IdentityEnter(kStateStandby, kNoIoConnectionsEstablished);
240 }
241 break;
242 default:
243 break;
244 }
245}

248 EipStatus status = kEipStatusOk;
249
250 switch (instance->instance_number) {
251 case DEMO_APP_OUTPUT_ASSEMBLY_NUM:
252 /* Process output assembly data (LED control only) */
253 gpio_set_level(kStatusLedGpio,
254 (g_assembly_data096[0] & 0x01) ? 1 : 0);
255 IdentityNoteIoActivity();
256 break;
257 case DEMO_APP_CONFIG_ASSEMBLY_NUM:
258 status = kEipStatusOk;
259 break;
260 default:
261 OPENER_TRACE_INFO(
262 "Unknown assembly instance ind AfterAssemblyDataReceived");
263 break;
264 }
265 return status;
266}

308EipStatus EthLnkPreGetCallback(CipInstance *instance,
309 CipAttributeStruct *attribute,
310 CipByte service) {
311 (void)service;
312 if (instance == NULL || attribute == NULL) {
313 return kEipStatusOk;
314 }
315
316 if (instance->instance_number == 0 ||
317 instance->instance_number > OPENER_ETHLINK_INSTANCE_CNT) {
318 return kEipStatusOk;
319 }
320
321 size_t idx = instance->instance_number - 1U;
322 switch (attribute->attribute_number) {
323 case 4: {
324 const NetworkInterfaceCounters *src = NetworkGetInterfaceCounters();
325 CipEthernetLinkInterfaceCounters *dst = &g_ethernet_link[idx].interface_cntrs;
326 dst->ul.in_octets = src->in_octets;
327 dst->ul.in_ucast = src->in_ucast_packets;
328 dst->ul.in_nucast = src->in_nucast_packets;
329 dst->ul.in_discards = src->in_discards;
330 dst->ul.in_errors = src->in_errors;
331 dst->ul.in_unknown_protos = src->in_unknown_protos;
332 dst->ul.out_octets = src->out_octets;
333 dst->ul.out_ucast = src->out_ucast_packets;
334 dst->ul.out_nucast = src->out_nucast_packets;
335 dst->ul.out_discards = src->out_discards;
336 dst->ul.out_errors = src->out_errors;
337 OPENER_TRACE_INFO("EthCntr Pre: inst=%u in_oct=%" PRIu32 " in_ucast=%" PRIu32 " out_ucast=%" PRIu32 " out_oct=%" PRIu32 "\n",
338 (unsigned)instance->instance_number,
339 src->in_octets,
340 src->in_ucast_packets,
341 src->out_ucast_packets,
342 src->out_octets);
343 break;
344 }
345 case 5: {
346 CipEthernetLinkMediaCounters *dst = &g_ethernet_link[idx].media_cntrs;
347 ZeroMediaCounters(dst);
348 break;
349 }
350 default:
351 break;
352 }
353
354 return kEipStatusOk;
355}

357EipStatus EthLnkPostGetCallback(CipInstance *instance,
358 CipAttributeStruct *attribute,
359 CipByte service) {
360 if (instance == NULL || attribute == NULL) {
361 return kEipStatusOk;
362 }
363
364 if ((service & 0x7FU) != kEthLinkGetAndClear) {
365 return kEipStatusOk;
366 }
367
368 if (instance->instance_number == 0 ||
369 instance->instance_number > OPENER_ETHLINK_INSTANCE_CNT) {
370 return kEipStatusOk;
371 }
372
373 size_t idx = instance->instance_number - 1U;
374 switch (attribute->attribute_number) {
375 case 4:
376 ZeroInterfaceCounters(&g_ethernet_link[idx].interface_cntrs);
377 NetworkResetInterfaceCounters();
378 break;
379 case 5:
380 ZeroMediaCounters(&g_ethernet_link[idx].media_cntrs);
381 break;
382 default:
383 break;
384 }
385
386 return kEipStatusOk;
387}

418void RunIdleChanged(EipUint32 run_idle_value) {
419 OPENER_TRACE_INFO("Run/Idle handler triggered\n");
420 if ((0x0001 & run_idle_value) == 1) {
421 IdentityNoteIoActivity();
422 } else if (s_active_io_connections == 0) {
423 IdentityEnter(kStateStandby, kNoIoConnectionsEstablished);
424 } else if (!s_io_activity_seen) {
425 IdentityEnter(kStateStandby,
426 kAtLeastOneIoConnectionEstablishedAllInIdleMode);
427 }
428 (void) run_idle_value;
429}
